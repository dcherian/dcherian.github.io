1//! Mid-level IR — core lower-level vocabulary for expressing ops on a
2//! chunk grid.
3//!
4//! `Expr` is an expression describing an operation on a Brick grid.
5//! It carries axis names (`dims`) and chunk counts (`grid`) only — element
6//! extents, dtype, and brick kind all live on `ArrayExpr` (the array-level
7//! IR). The lowering pass [`crate::array::lower::lower`] is the boundary
8//! between the two and bakes the element-sized `ChunkGrid` into each
9//! concrete `Source` / `Kernel` / `Accumulator` impl.
10

28// ---------------------------------------------------------------------------
29// Frame — per-axis brick-space framing (read by the planner)
30// ---------------------------------------------------------------------------
31
32/// How a consumer frames a producer's bricks along one axis — the brick-space
33/// analog of the array-level [`crate::array::expr::Frame`], in chunk coords. This is
34/// a structural iteration-space transform: the planner expands it
35/// into an operation applied to the input [`crate::brix::layout::NamedLayout`]. What the Op does
36/// with the resulting tile (sum it in a Reduce, map over it in a Blockwise)
37/// is the Op's concern, not the frame's. Never a point
38/// transform (the affine view + [`AxisOverride`] cover those); identity
39/// (pass-through / broadcast) is the absence of an entry.
40#[derive(Debug, Clone)]
41pub enum Frame {
42    /// The whole axis is one frame. `keepdims = false` drops it from the
43    /// output (the planner appends a fold loop over all producer bricks);
44    /// `keepdims = true` keeps a size-1 survivor (a `logical_divide` by `n_in`).
45    WholeAxis { keepdims: bool },
46    /// Regular frames: output chunk `c` covers producer chunks
47    /// `[c·stride, c·stride + size)`, **clamped to the producer extent**. The
48    /// planner `logical_divide`s the axis into `(n_out, size)` (coarsen when
49    /// `size == stride`).
50    Window { size: usize, stride: usize },
51    /// Arbitrary per-output-chunk producer ranges (e.g. resampling).
52    Ranges(Arc<[Range<usize>]>),
53}
54
55// ---------------------------------------------------------------------------
56// AxisOverride — per-axis point transform (applied after the affine pass)
57// ---------------------------------------------------------------------------
58
59/// A per-axis point override: a `usize -> usize` applied to the
60/// affine-resolved producer coord at [`AccessRelation::resolve_coord`] time.
61///
62/// These are the **non-affine residual** of an edge — what the affine view
63/// ([`crate::brix::layout::NamedLayout::compose`]) can't express.
64///
65/// These overrides are applied when necessary after composing an input NamedLayout
66/// with a [`crate::brix::expr::Frame`].
67///
68/// [`AccessRelation::resolve_coord`]: crate::brix::schedule::AccessRelation::resolve_coord
69#[derive(Debug, Clone)]
70// ANCHOR: axis_override
71pub enum AxisOverride {
72    /// Constant chunk-coord shift (slice): `c + k`.
73    /// Affine, but kept as an override for symmetry with oindex & vindex.
74    Offset(usize),
75    /// Per-iteration lookup table (gather: oindex & vindex): `table[c]`.
76    /// that assumes the keys are 0..len(Vec)
77    /// TODO: eventually we'll likely need an inverted index to reverse this;
78    ///       and thus a more complex type.
79    Lookup(Vec<usize>),
80}
81// ANCHOR_END: axis_override
82
83impl AxisOverride {
84    /// Map a single-axis chunk coord through this override.
85    pub fn translate(&self, coord: usize) -> usize {
86        match self {
87            AxisOverride::Offset(k) => coord + k,
88            AxisOverride::Lookup(table) => {
89                // This should not have been expressible.
90                #[allow(clippy::expect_used)]
91                table.get(coord).copied().expect("gather Lookup coord out of table bounds")
92            }
93        }
94    }
95}

97// ---------------------------------------------------------------------------
98// Per-edge access relations
99// ---------------------------------------------------------------------------
100
101/// Per-axis override kind, carried inside [`AxisAccess::Translated`].
102///
103// ANCHOR: translated_access
104/// We need the ability to express iteration spaces as being composed in at
105/// least two ways:
106/// 1. the cartesian product of iteration spaces along each axis of the chunk grid
107/// 2. a zipped iterator over iteration spaces of multiple axes of the chunk grid
108///
109/// For example, with the outer map key in `[brackets]`:
110/// - Slicing: ["i" → Independent(AxisOverride::Offset(slice_start))]
111/// - OIndex:  ["i" → Independent(AxisOverride::Lookup(outer_indices_for_i))]
112/// - VIndex:  ["k" → Zipped({"i": idx_i, "j": idx_j})]
113///   (key `k` = produced/leader axis; followers `i`,`j` are the consumed axes
114///   co-iterated by `k`, each carrying its own gather table.)
115///
116/// As a side-note, neither Mullin's Mathematics of Arrays calculus, nor CuTe layouts,
117/// provide any formalism for oindex/vindex operations. Those two harmonize the affine
118/// access pieces, with MoA apparently folding in the AxisOverride::Offset piece too.
119#[derive(Clone, Debug)]
120pub enum TranslatedAccess {
121    /// Self-driven per-axis override: producer's chunk-coord on this
122    /// axis is derived from this axis's own iter coord.
123    /// Co-iterated follower group: producer's chunk-coord on each
124    /// follower axis is derived from the leader axis's iter coord.
125    Independent(AxisOverride),
126    Zipped(HashMap<AxisName, AxisOverride>),
127}
128// ANCHOR_END: translated_access
129
130/// An **ordered** per-axis override stack. The planner accumulates one per
131/// emission while descending consumer edges (downstream-most edge first),
132/// and [`AccessRelation::resolve_coord`] applies the entries in that order
133/// — so stacked overrides on one axis compose (`+1` then `+2` reads the
134/// producer at `+3`). Order is load-bearing; an axis may appear more than
135/// once.
136///
137/// [`AccessRelation::resolve_coord`]: crate::brix::schedule::AccessRelation::resolve_coord
138pub type TranslatedAccessStack = Vec<(AxisName, TranslatedAccess)>;
139
140/// Per-axis specification of how a consumer reads a producer along one axis.
141///
142/// Each [`Input`] carries a `HashMap<AxisName, AxisAccess>` keyed by axis
143/// name. A given axis is **either** structurally framed (`Composed`) — folded
144/// into the loop nest by composing layouts via
145/// [`crate::brix::layout::NamedLayout::compose`] — **or** per-coord
146/// translated (`Translated`) — handled at runtime by
147/// [`crate::brix::schedule::AccessRelation::resolve_coord`].
148/// Absence from the map is identity (pass-through / broadcast).
149// Note I am using a sum type here to disallow combining the two modes.
150#[derive(Debug, Clone)]
151// ANCHOR: axis_access
152pub enum AxisAccess {
153    /// Affine transformations on this axis. The planner absorbs it into the
154    /// access layout via `NamedLayout::compose`; nothing about this axis
155    /// survives to runtime resolution.
156    Composed(Frame),
157    /// Per-coord override on this axis (slice offset, gather lookup,
158    /// zipped gather followers). Carried through to runtime
159    /// `resolve_coord` after the affine pass.
160    Translated(TranslatedAccess),
161    // Identity is assumed for any axis not in Input::access
162}
163// ANCHOR_END: axis_access

166// State — shared trait for serializable op state
167// ---------------------------------------------------------------------------
168
169// ANCHOR: state
170/// Opaque serializable state for accumulators (and other stateful ops).
171///
172/// Each accumulator defines its own concrete state struct (e.g., `SumState`,
173/// `MeanState`) and downcasts internally. The executor never inspects state
174/// — it only passes it between accumulator methods and serializes at stage
175/// boundaries.
176pub trait State: Debug + Send + Any {
177    /// Serialize for inter-process exchange
178    fn to_bytes(&self) -> BrixResult<Vec<u8>>;
179    fn as_any(&self) -> &dyn Any;
180    fn as_any_mut(&mut self) -> &mut dyn Any;
181}
182// ANCHOR_END: state

184// ---------------------------------------------------------------------------
185// Source — narrow trait
186// ---------------------------------------------------------------------------
187
188/// A source of brick payloads. Concrete impls (in `array/`) bake their
189/// element-sized `ChunkGrid` in as a private field; `get_brick_payload`
190/// returns just the payload for the given chunk coord, and the executor
191/// re-attaches `coord` to build a `Brick`.
192///
193/// brix never names `ChunkGrid` directly — the planner reads only chunk
194/// counts from the surrounding [`Expr::space`].
195// ANCHOR: source_trait
196pub trait Source: Debug + Send + Sync {
197    fn get_brick_payload(&self, coord: &BrickCoordinate) -> BrixResult<Arc<dyn BrickPayload>>;
198    fn cost(&self, coord: &BrickCoordinate) -> Cost;
199}
200// ANCHOR_END: source_trait
201
202/// A sink receives bricks as they are produced by the executor.
203/// Not part of the expression DAG — passed to `execute()` as a destination.
204///
205/// `&self` on `write_brick` so writes can be concurrent (e.g., Zarr chunk
206/// writes to independent keys). Implementations needing mutation use interior
207/// mutability.
208// ANCHOR: sink_trait
209pub trait Sink: Send + Sync {
210    fn init(&self) -> BrixResult<()>;
211    fn write_brick(&self, brick: Brick) -> BrixResult<()>;
212    fn finalize(&self) -> BrixResult<()>;
213}
214// ANCHOR_END: sink_trait

216// ---------------------------------------------------------------------------
217// Kernel trait — opaque user-defined functions
218// ---------------------------------------------------------------------------
219
220/// Opaque blockwise computation unit.
221/// Each input is a located [`Brick`] carrying its own [`BrickCoordinate`] and [`BrickPayload`].
222/// the output's chunk `coord` is supplied separately.
223/// Concrete impls include their element-sized `ChunkGrid` when lowering.
224/// See `array/kernels/` for examples.
225pub trait Kernel: Debug + Send + Sync {
226    /// Compute one output brick's payload from the input bricks.
227    /// The Brick coordinate `coord` is supplied for kernels
228    /// that need element-space context (e.g. Array Slice/Gather);
229    /// The executor wraps the returned payload into a `Brick { coord, data }`
230    /// and passes that on.
231    fn apply(
232        &self,
233        inputs: &[&Brick],
234        coord: &BrickCoordinate,
235    ) -> BrixResult<Arc<dyn BrickPayload>>;
236
237    /// Per-brick cost closure (exact cost for one output brick at the
238    /// given coordinate). Concrete kernels capture the grid + any other
239    /// state needed to compute brick shape from coord.
240    fn cost(&self, coord: &BrickCoordinate) -> Cost;
241
242    /// Short display prefix used by the planner when naming this node in
243    /// `Schedule` sketches.
244    fn name(&self) -> &'static str;
245}

247// ---------------------------------------------------------------------------
248// Accumulators — trait
249// ---------------------------------------------------------------------------
250
251/// Fold accumulator for reductions.
252///
253/// State is type-erased via `Box<dyn State>` / `&mut dyn State`. Each
254/// accumulator defines its own concrete state struct with typed fields
255/// and downcasts internally.
256pub trait Accumulator: Debug + Send + Sync {
257    /// Create initial state for the given output chunk coord. Concrete
258    /// impls derive bounds internally from their baked-in `ChunkGrid` —
259    /// fallible because deriving the brick shape requires the coord to
260    /// carry every axis the grid expects.
261    fn init(&self, coord: &BrickCoordinate) -> BrixResult<Box<dyn State>>;
262
263    /// Sequential fold over the core (reduced) dimensions.
264    /// I *think* mut dyn State is OK since we are serial within a brick.
265    fn fold_left(
266        &self,
267        state: &mut dyn State,
268        input: &Brick,
269        output_coord: &BrickCoordinate,
270    ) -> BrixResult<()>;
271
272    /// Combine two accumulator states (for tree reduce / parallel merging).
273    fn merge(&self, state: &mut dyn State, other: &dyn State) -> BrixResult<()>;
274
275    /// Extract final brick from accumulator state. `coord` identifies the
276    /// output brick; concrete impls derive element-space `bounds` from
277    /// their baked-in `ChunkGrid` and stamp them on the returned `Brick`.
278    fn finalize(&self, state: Box<dyn State>, coord: &BrickCoordinate) -> BrixResult<Brick>;
279
280    fn deserialize_state(&self, bytes: &[u8]) -> BrixResult<Box<dyn State>>;
281
282    fn is_associative(&self) -> bool;
283    fn is_commutative(&self) -> bool;
284
285    fn cost(&self, coord: &BrickCoordinate) -> Cost;
286}

292// ANCHOR: input_struct
293/// An input to an expression node. Carries the producer `Arc<Expr>` plus a
294/// per-axis [`AxisAccess`] describing how the consumer reads chunks from
295/// this producer along that axis.
296/// Axis transformations are either
297///  - a `Composed` frame: planner applies these via `NamedLayout::compose`
298///    when constructing a [`crate::brix::schedule::Schedule::Domain`] iteration space; or
299///  - `Translated` per-coord override: imposed at runtime with [`crate::brix::schedule::AccessRelation::resolve_coord`].
300#[derive(Debug, Clone)]
301pub struct Input {
302    pub expr: Arc<Expr>,
303    pub access: HashMap<AxisName, AxisAccess>,
304}
305// ANCHOR_END: input_struct

311// ANCHOR: op_enum
312/// The operation performed by an expression node.
313#[derive(Clone)]
314pub enum Op {
315    Source(Arc<dyn Source>),
316    Blockwise(Arc<dyn Kernel>),
317    Reduce(Arc<dyn Accumulator>),
318    // Scan(Arc<dyn Scanner>)
319}
320// ANCHOR_END: op_enum
321
322impl Op {
323    /// Stateful nodes get `Init`/`Apply`/`Finalize` bracketing in the
324    /// schedule; stateless nodes get just `Apply`. The split is structural
325    /// — only `Reduce` carries cross-iteration state.
326    pub fn is_stateful(&self) -> bool {
327        matches!(self, Op::Reduce(_))
328    }
329}
330
331impl Debug for Op {
332    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
333        match self {
334            Op::Source(s) => write!(f, "Source({:?})", s),
335            Op::Blockwise(k) => write!(f, "Blockwise({:?})", k),
336            Op::Reduce(acc) => write!(f, "Reduce({:?})", acc),
337        }
338    }
339}

341// ---------------------------------------------------------------------------
342// Expression node
343// ---------------------------------------------------------------------------
344
345/// Logical expression node — uniform struct with varying `Op`.
346///
347/// Carries chunk counts and axis names via `space` — never element extents.
348/// The corresponding element-sized `ChunkGrid` is baked into each concrete
349/// `Source` / `Kernel` / `Accumulator` impl during lowering; the planner
350/// consumes only `grid` here.
351/// An important choice here is to have the API consumer specify [`Input`]
352/// This allows the consumer to specify patterns of `access` on each [`Input`].
353/// For example, we might be 'reduce-ing' along multiple axes but require
354/// a fold across one, and allow a tree-reduce along another.
355/// (I don't have a concrete example for this).
356/// GroupBy is an example of want the consumer to control `access`.
357/// We might want oindex-style "subsetting" along with a reduction.
358/// (Of course, this is untested at the moment of writing).
359///
360/// `Expr` deliberately does not implement `Clone`: callers share
361/// nodes via `Arc<Expr>`.
362
363#[derive(Debug)]
364// ANCHOR: expr_struct
365pub struct Expr {
366    /// Stable per-instance identity.
367    pub id: ExprId,
368    pub op: Op,
369    pub inputs: Vec<Input>,
370    /// Iteration Space for this node's output. Projected from the
371    /// element-sized `ChunkGrid` chosen at lowering time.
372    pub space: IterationSpace,
373    /// Optional user-supplied display name. When set, the planner prefers
374    /// this as the node's `NodeName` (with a `#N` collision suffix if the
375    /// same label appears more than once).
376    pub label: Option<String>,
377}
378// ANCHOR_END: expr_struct
379
380/// Stable per-instance identity for an [`Expr`].
381#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
382pub struct ExprId(pub u64);
383
384impl ExprId {
385    /// Allocate a fresh id from the process-global counter. Each call
386    /// returns a distinct value; ids are never recycled within a
387    /// process. Use this for every newly-constructed `Expr` — never
388    /// reuse an existing id when minting a new logical node.
389    pub fn fresh() -> Self {
390        static NEXT: AtomicU64 = AtomicU64::new(0);
391        // Relaxed is sufficient — we only require uniqueness, not any
392        // ordering relative to other memory ops.
393        ExprId(NEXT.fetch_add(1, Ordering::Relaxed))
394    }
395}
