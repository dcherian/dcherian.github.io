8/// A brick's position in the producer's chunk grid — a per-axis block
9/// index keyed by axis name, with insertion order preserved. Order
10/// matches the payload's shape, so `iter` yields axes in positional
11/// order; `get` looks up by name. Carrying the names makes bricks
12/// self-describing: kernels can read each input's dim order straight off
13/// `brick.coord` and align/broadcast accordingly, without the kernel
14/// having to memorize producer dims.
15#[derive(Debug, Clone, PartialEq, Eq)]
16pub struct BrickCoordinate(IndexMap<AxisName, usize>);
17
18impl BrickCoordinate {
19    /// Build from parallel positional `names` and `coord` slices.
20    pub fn from_named(names: &[AxisName], coord: &[usize]) -> Self {
21        debug_assert_eq!(
22            names.len(),
23            coord.len(),
24            "BrickCoordinate::from_named: names and coord lengths differ"
25        );
26        Self(names.iter().cloned().zip(coord.iter().copied()).collect())
27    }
28
29    /// Look up the block index for `name`. Returns `None` if this brick
30    /// doesn't carry that axis.
31    pub fn get(&self, name: &AxisName) -> Option<usize> {
32        self.0.get(name).copied()
33    }
34
35    /// Iterate `(name, block_index)` pairs in positional order (matches
36    /// the payload's shape axes).
37    pub fn iter(&self) -> impl Iterator<Item = (&str, usize)> {
38        self.0.iter().map(|(n, &c)| (n.as_str(), c))
39    }
40
41    pub fn rank(&self) -> usize {
42        self.0.len()
43    }
44}

46/// The runtime representation of a chunk's data. Implementations live in
47/// the array layer (see [`crate::array::ArrayPayload`]'s `Dense` / `Fill`
48/// variants); future Arrow-backed bricks will land alongside under `arrow/`.
49///
50/// `Fill`-typed payloads enable runtime short-circuits in kernels (e.g.,
51/// `BinaryOp::Mul` with a `Fill(0)` input returns the Fill directly).
52/// Kernels recover the concrete type via `as_any().downcast_ref::<...>()`.
53// ANCHOR: brick_payload
54pub trait BrickPayload: Debug + Send + Sync + Any {
55    /// Serialize for inter-process exchange (shared storage, DBSP checkpoint).
56    ///
57    /// Format is implementation-specific but conventionally tagged so a
58    /// dispatcher (`array::brick::payload_from_bytes`) can recover the
59    /// concrete type. This is a bootstrapping format; a future version may
60    /// use Arrow IPC, Zarr codecs, or a zero-copy format.
61    fn to_bytes(&self) -> BrixResult<Vec<u8>>;
62
63    fn as_any(&self) -> &dyn Any;
64}
65// ANCHOR_END: brick_payload
