1//! CuTe-inspired hierarchical layout representation (Cecka 2026, arXiv:2603.02298).
2//!
3//! Mirrors CuTe's Shape:Stride representation:
4//! - `IntTuple` = HTuple(Z) from CuTe Definition 2.2
5//! - `Layout` = Shape:Stride from CuTe Definition 2.17
6//!
7//! A layout maps coordinates to offsets into a linear ordering of elements
8//! via `inner_product(coord, stride)`.
9//! Shapes and strides are congruent hierarchical integer tuples.
10//!
11//! The code here was generated by pointing Claude at the paper
12//! and the online docs: <https://docs.nvidia.com/cutlass/latest/media/docs/cpp/cute/index.html>
13//! Unit tests map to worked examples in the paper.
14//! Function names map to existing CuTe function names.
15//! Two major changes:
16//!  - Brix introduces [`NamedLayout`] attaching axis names to Layout.
17//!  - Brix uses row-major (rightmost varies fastest) for index
18//!    decomposition. CuTe uses column-major (leftmost varies fastest).
19

122/// Hierarchical tuple of non-negative integers.
123///
124/// Either a single integer (leaf) or a tuple of IntTuples (modes).
125/// Used for both shapes and strides in CuTe layouts.
126#[derive(Clone, PartialEq, Eq)]
127pub enum IntTuple {
128    Int(usize),
129    Tuple(Vec<IntTuple>),
130}
131
132impl Debug for IntTuple {
133    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
134        match self {
135            IntTuple::Int(v) => write!(f, "{v}"),
136            IntTuple::Tuple(modes) => {
137                write!(f, "(")?;
138                for (i, m) in modes.iter().enumerate() {
139                    if i > 0 {
140                        write!(f, ",")?;
141                    }
142                    write!(f, "{m:?}")?;
143                }
144                write!(f, ")")
145            }
146        }
147    }
148}
149
150impl IntTuple {
151    /// Product of all leaf values. For shapes, this is the domain size.
152    pub fn size(&self) -> usize {
153        match self {
154            IntTuple::Int(v) => *v,
155            IntTuple::Tuple(modes) => modes.iter().map(|m| m.size()).product(),
156        }
157    }
158
159    /// Number of top-level elements. 1 for Int, n for Tuple of n elements.
160    pub fn rank(&self) -> usize {
161        match self {
162            IntTuple::Int(_) => 1,
163            IntTuple::Tuple(modes) => modes.len(),
164        }
165    }
166
167    /// Collect leaf values in pre-order traversal.
168    pub fn flat(&self) -> Vec<usize> {
169        match self {
170            IntTuple::Int(v) => vec![*v],
171            IntTuple::Tuple(modes) => modes.iter().flat_map(|m| m.flat()).collect(),
172        }
173    }
174
175    /// Congruence test (CuTe Definition 2.3): same hierarchical profile.
176    pub fn is_congruent(&self, other: &IntTuple) -> bool {
177        match (self, other) {
178            (IntTuple::Int(_), IntTuple::Int(_)) => true,
179            (IntTuple::Tuple(a), IntTuple::Tuple(b)) => {
180                a.len() == b.len() && a.iter().zip(b).all(|(x, y)| x.is_congruent(y))
181            }
182            _ => false,
183        }
184    }
185}

191/// CuTe Layout = Shape : Stride.
192///
193/// Shape determines the domain (set of valid coordinates).
194/// Stride maps coordinates to offsets via inner_product.
195/// Shape and stride must be congruent (same hierarchical profile).
196#[derive(Clone, PartialEq, Eq)]
197pub struct Layout {
198    pub shape: IntTuple,
199    pub stride: IntTuple,
200}
201
202impl Debug for Layout {
203    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
204        write!(f, "{:?}:{:?}", self.shape, self.stride)
205    }
206}
207
208impl Layout {
209    /// Construct a layout, panicking if shape and stride aren't congruent.
210    pub fn new(shape: IntTuple, stride: IntTuple) -> Self {
211        assert!(
212            shape.is_congruent(&stride),
213            "shape and stride must be congruent: {shape:?} vs {stride:?}"
214        );
215        Layout { shape, stride }
216    }
217
218    /// Domain size (product of all shape leaves).
219    pub fn size(&self) -> usize {
220        self.shape.size()
221    }
222
223    /// Number of top-level modes.
224    pub fn rank(&self) -> usize {
225        self.shape.rank()
226    }
227
228    /// CuTe `crd2idx`: map a coordinate to an offset.
229    ///
230    /// `coord` is an [`IntTuple`] — coordinates live in the same
231    /// hierarchical-integer-tuple algebra as shapes and strides. Three forms,
232    /// dispatched by recursion (see `crd2idx_impl`):
233    /// - **natural** (coord & shape both tuples, congruent): per-mode recursion,
234    ///   summed;
235    /// - **1-D** (coord an `Int`): decomposed against the shape (row-major,
236    ///   rightmost-fastest) via `inner_product_idx`;
237    /// - **mixed**: falls out of the recursion — an `Int` entry at a tuple mode
238    ///   is decomposed; a tuple entry recurses.
239    ///
240    /// Row-major (Brix convention); CuTe is colexicographic.
241    pub fn crd2idx(&self, coord: &IntTuple) -> usize {
242        crd2idx_impl(coord, &self.shape, &self.stride)
243    }
244
245    /// Evaluate from a 1D integral coordinate — `crd2idx` of the `Int` form.
246    ///
247    /// Decomposes `idx` row-major (rightmost leaf varies fastest)
248    /// through the shape tree, then computes inner_product with stride.
249    pub fn eval(&self, idx: usize) -> usize {
250        self.crd2idx(&IntTuple::Int(idx))
251    }
252
253    /// Compose: `self ∘ other` (CuTe Section 3.3).
254    ///
255    /// `other` defines the new domain (its shape becomes the result shape).
256    /// `self` defines the codomain (its strides scale the result strides).
257    ///
258    /// For atoms: `(sA:dA) ∘ (sB:dB) = (sB : dA*dB)`.
259    /// For tuples of same rank: by-mode composition (distributive property,
260    /// CuTe eq. 22 — valid when sublayout images are segregated).
261    pub fn compose(&self, other: &Layout) -> Layout {
262        let (new_shape, new_stride) =
263            compose_impl(&self.shape, &self.stride, &other.shape, &other.stride);
264        Layout::new(new_shape, new_stride)
265    }
266
267    /// `logical_divide` of a 1-D (single-mode) layout by an integer `tile`
268    /// (CuTe Section 3.4), **adapted to Brix row-major**.
269    ///
270    /// CuTe is colexicographic and emits the tile mode first (`(tile, rest)`,
271    /// tile contiguous, stride 1). Brix is row-major (see [`native_layout`] /
272    /// `inner_product_idx`), so we emit **rest-first, tile-last**:
273    ///
274    /// ```text
275    /// (n : s).logical_divide(k) = (ceil(n/k), k) : (k·s, s)
276    /// ```
277    ///
278    /// The tile (rightmost, depth 1) stays contiguous at the original stride
279    /// `s`; the rest mode (leftmost, depth 0) strides by `k·s`. Like CuTe, the
280    /// divide **rounds the rest mode up** (`ceil`) for non-divisible `n`,
281    /// overshooting the domain — coords past the original `n` are masked by
282    /// predication (see `iteration-space.md`), not by this op.
283    ///
284    /// Only the single-mode (atom) case is implemented: the general CuTe
285    /// `A ∘ (B, complement(B, size(A)))` over hierarchical inputs / multi-mode
286    /// tilers is deferred until a use case needs it. Requires an atom layout
287    /// (a layout-algebra invariant, asserted as in [`Layout::new`] /
288    /// [`Layout::compose`]); [`NamedLayout::logical_divide`] guards this before
289    /// calling.
290    pub fn logical_divide(&self, tile: usize) -> Layout {
291        let extents = self.shape.flat();
292        let strides = self.stride.flat();
293        assert_eq!(extents.len(), 1, "logical_divide requires a 1-D layout, got {self:?}");
294        assert!(tile > 0, "logical_divide tile must be positive");
295        let (n, s) = (extents[0], strides[0]);
296        let n_out = n.div_ceil(tile);
297        Layout::new(
298            IntTuple::Tuple(vec![IntTuple::Int(n_out), IntTuple::Int(tile)]),
299            IntTuple::Tuple(vec![IntTuple::Int(tile * s), IntTuple::Int(s)]),
300        )
301    }
302
303    /// CuTe `get<I>`: the `i`-th top-level mode as its own layout, preserving
304    /// its shape/stride sub-tree. Sub-modes nest positionally —
305    /// `get(i0).get(i1)` is CuTe's `get<I0, I1>`. An atom (rank-1) layout has
306    /// only mode 0 (itself). Panics if `i` is out of range (a layout-algebra
307    /// invariant, as in [`Layout::new`] / [`Layout::compose`]).
308    pub fn get(&self, i: usize) -> Layout {
309        match (&self.shape, &self.stride) {
310            (IntTuple::Tuple(shapes), IntTuple::Tuple(strides)) => {
311                assert!(i < shapes.len(), "get: mode {i} out of range (rank {})", shapes.len());
312                Layout::new(shapes[i].clone(), strides[i].clone())
313            }
314            (shape, stride) => {
315                assert_eq!(i, 0, "get: mode {i} out of range for atom layout");
316                Layout::new(shape.clone(), stride.clone())
317            }
318        }
319    }
320
321    /// Evaluate from top-level mode coordinates.
322    ///
323    /// One entry per top-level mode. If a mode is a Tuple, its coord
324    /// entry is treated as an integral coordinate into that mode's
325    /// sub-shape (decomposed row-major).
326    pub fn eval_coord(&self, coord: &[usize]) -> usize {
327        let crd = match &self.shape {
328            IntTuple::Tuple(shapes) => {
329                assert_eq!(
330                    coord.len(),
331                    shapes.len(),
332                    "coord length {} != rank {}",
333                    coord.len(),
334                    shapes.len()
335                );
336                IntTuple::Tuple(coord.iter().copied().map(IntTuple::Int).collect())
337            }
338            IntTuple::Int(_) => {
339                assert_eq!(coord.len(), 1, "coord length {} != rank 1", coord.len());
340                IntTuple::Int(coord[0])
341            }
342        };
343        self.crd2idx(&crd)
344    }
345}

347/// Recursive inner_product with row-major index decomposition.
348///
349/// For `Int(s) : Int(d)`: `d * (idx % s)`.
350/// For `Tuple`: recurses right-to-left (rightmost leaf varies fastest).
351fn inner_product_idx(shape: &IntTuple, stride: &IntTuple, idx: usize) -> usize {
352    match (shape, stride) {
353        (IntTuple::Int(s), IntTuple::Int(d)) => d * (idx % s),
354        (IntTuple::Tuple(shapes), IntTuple::Tuple(strides)) => shapes
355            .iter()
356            .zip(strides)
357            .rev()
358            .scan(idx, |remaining, (s, d)| {
359                let mode_size = s.size();
360                let mode_idx = *remaining % mode_size;
361                *remaining /= mode_size;
362                Some(inner_product_idx(s, d, mode_idx))
363            })
364            .sum(),
365        _ => unreachable!("congruent shape/stride"),
366    }
367}

369/// CuTe `crd2idx` recursion over congruent `(coord, shape, stride)` trees.
370///
371/// - `coord` is an `Int`: a 1-D (or leaf) coordinate — decompose it against
372///   `shape`/`stride` via [`inner_product_idx`]. This covers both the leaf
373///   case (`Int` shape → `d·(c mod s)`) and a 1-D index into a tuple shape
374///   (the row-major odometer), so the *mixed* form needs no special case.
375/// - `coord` is a `Tuple`: a natural/hierarchical coordinate — recurse
376///   per mode against the congruent shape/stride modes and sum.
377fn crd2idx_impl(coord: &IntTuple, shape: &IntTuple, stride: &IntTuple) -> usize {
378    match coord {
379        IntTuple::Int(c) => inner_product_idx(shape, stride, *c),
380        IntTuple::Tuple(coords) => {
381            let (IntTuple::Tuple(shapes), IntTuple::Tuple(strides)) = (shape, stride) else {
382                unreachable!("crd2idx: tuple coord against scalar shape (incongruent)");
383            };
384            assert_eq!(
385                coords.len(),
386                shapes.len(),
387                "crd2idx: coord rank {} != shape rank {}",
388                coords.len(),
389                shapes.len()
390            );
391            coords.iter().zip(shapes).zip(strides).map(|((c, s), d)| crd2idx_impl(c, s, d)).sum()
392        }
393    }
394}

396/// Build the natural coordinate (an [`IntTuple`] congruent to `mode`) for one
397/// named top-level mode, drawing each leaf from `indices[(name, position)]` in
398/// pre-order. A flat (`Int`) mode takes position 0; a tiled mode takes one
399/// position per leaf (CuTe `get<0>, get<1>, …`).
400/// A named coordinate, scattered per mode: `(axis, mode_index)` → component.
401///
402/// This is the iteration coordinate the executor accumulates and that the
403/// access layer (`NamedLayout::eval_axis`,
404/// [`AccessRelation::resolve_coord`]) evaluates a [`NamedLayout`] against. A
405/// split axis contributes two entries — `(axis, 0)` surviving, `(axis, 1)` tile
406/// — kept distinct so they recombine (e.g. coarsen's `c·k + j`).
407///
408/// `mode_index 0` is the leading sub-layout: the only mode of a flat axis, the
409/// survivor of a [`NamedLayout::logical_divide`] (CuTe `get<0>`).
410///
411/// [`AccessRelation::resolve_coord`]: crate::brix::schedule::AccessRelation::resolve_coord
412#[derive(Debug, Default, Clone)]
413pub struct NamedCoord(HashMap<(AxisName, usize), usize>);
414
415impl NamedCoord {
416    /// Set the component at `(axis, mode_index)`.
417    pub fn insert(&mut self, axis: &AxisName, mode_index: usize, value: usize) {
418        self.0.insert((axis.clone(), mode_index), value);
419    }
420
421    /// The component at `(axis, mode_index)`, if present.
422    pub fn get(&self, axis: &AxisName, mode_index: usize) -> Option<usize> {
423        self.0.get(&(axis.clone(), mode_index)).copied()
424    }
425
426    /// The leading (`mode_index 0`) component — the only mode of a flat axis,
427    /// the survivor of a split.
428    pub fn get_leading(&self, axis: &AxisName) -> Option<usize> {
429        self.get(axis, 0)
430    }
431}

433fn natural_coord(
434    mode: &IntTuple,
435    name: &AxisName,
436    pos: usize,
437    indices: &NamedCoord,
438) -> LayoutResult<(IntTuple, usize)> {
439    match mode {
440        IntTuple::Int(_) => {
441            indices.get(name, pos).map(|v| (IntTuple::Int(v), pos + 1)).ok_or_else(|| {
442                LayoutError::UnknownAxis {
443                    axis: format!("{name}@{pos}"),
444                    available: indices.0.keys().map(|(n, p)| format!("{n}@{p}")).collect(),
445                }
446            })
447        }
448        IntTuple::Tuple(subs) => {
449            let (children, next) = subs.iter().try_fold(
450                (Vec::with_capacity(subs.len()), pos),
451                |(mut acc, p), s| {
452                    let (c, np) = natural_coord(s, name, p, indices)?;
453                    acc.push(c);
454                    Ok::<_, LayoutError>((acc, np))
455                },
456            )?;
457            Ok((IntTuple::Tuple(children), next))
458        }
459    }
460}

462/// Recursive composition: (shapeA:strideA) ∘ (shapeB:strideB).
463///
464/// - Atom ∘ Atom: strides multiply, result has B's shape.
465/// - Tuple ∘ Tuple (same rank): by-mode composition.
466/// - Atom ∘ Tuple: distribute atom over B's modes (A is applied to each mode).
467/// - Tuple ∘ Atom: A's modes are accessed sequentially by B's linear index
468///   (general CuTe base case — requires divisibility conditions).
469fn compose_impl(
470    shape_a: &IntTuple,
471    stride_a: &IntTuple,
472    shape_b: &IntTuple,
473    stride_b: &IntTuple,
474) -> (IntTuple, IntTuple) {
475    match (shape_a, stride_a, shape_b, stride_b) {
476        // Atom ∘ Atom: (sA:dA) ∘ (sB:dB) = (sB : dA*dB)
477        (IntTuple::Int(_sa), IntTuple::Int(da), IntTuple::Int(sb), IntTuple::Int(db)) => {
478            (IntTuple::Int(*sb), IntTuple::Int(da * db))
479        }
480
481        // Tuple ∘ Tuple (same rank): by-mode
482        (IntTuple::Tuple(sa), IntTuple::Tuple(da), IntTuple::Tuple(sb), IntTuple::Tuple(db)) => {
483            assert_eq!(sa.len(), sb.len(), "compose: rank mismatch ({} vs {})", sa.len(), sb.len());
484            let (shapes, strides): (Vec<_>, Vec<_>) = sa
485                .iter()
486                .zip(da)
487                .zip(sb.iter().zip(db))
488                .map(|((sa, da), (sb, db))| compose_impl(sa, da, sb, db))
489                .unzip();
490            (IntTuple::Tuple(shapes), IntTuple::Tuple(strides))
491        }
492
493        // Atom ∘ Tuple: distribute — A applied to each mode of B
494        (IntTuple::Int(_sa), IntTuple::Int(_da), IntTuple::Tuple(sb), IntTuple::Tuple(db)) => {
495            let (shapes, strides): (Vec<_>, Vec<_>) =
496                sb.iter().zip(db).map(|(sb, db)| compose_impl(shape_a, stride_a, sb, db)).unzip();
497            (IntTuple::Tuple(shapes), IntTuple::Tuple(strides))
498        }
499
500        // Tuple ∘ Atom: thread B's linear domain through A's modes
501        // (CuTe Section 3.3.2 eq. 19-21).
502        //
503        // Process right-to-left (rightmost = fastest varying in row-major).
504        // B's domain size is decomposed across A's modes: each mode either
505        // fully consumed (mode_size divides remaining) or partially consumed
506        // (remaining < mode_size). Recursion handles hierarchical sub-modes.
507        //
508        // Limitation: requires sB to decompose cleanly across mode sizes
509        // (divisibility at each step). dB > 1 that skips modes (e.g.,
510        // dB == fastest_mode_size) needs the general CuTe base case with
511        // stride-aware flattening — not yet implemented.
512        (IntTuple::Tuple(sa), IntTuple::Tuple(da), IntTuple::Int(sb), IntTuple::Int(db)) => {
513            let n = sa.len();
514            let mut shapes = vec![IntTuple::Int(1); n];
515            let mut strides = vec![IntTuple::Int(0); n];
516            let mut rest = *sb;
517            for i in (0..n).rev() {
518                if rest == 1 {
519                    // B's domain exhausted — remaining (lower-i) modes stay trivial.
520                    break;
521                }
522                let take = sa[i].size().min(rest);
523                assert!(
524                    rest % take == 0,
525                    "compose: Tuple ∘ Atom requires divisibility \
526                     (remaining={rest}, mode_size={})",
527                    sa[i].size()
528                );
529                let (rs, rd) =
530                    compose_impl(&sa[i], &da[i], &IntTuple::Int(take), &IntTuple::Int(*db));
531                shapes[i] = rs;
532                strides[i] = rd;
533                rest /= take;
534            }
535            assert!(rest == 1, "compose: B domain size ({sb}) exceeds A's domain");
536            (IntTuple::Tuple(shapes), IntTuple::Tuple(strides))
537        }
538
539        _ => unreachable!("shape and stride must be congruent"),
540    }
541}

587// NamedLayout — Layout + per-axis names
588// ---------------------------------------------------------------------------
589
590/// A `Layout` with an axis name per top-level mode.
591///
592/// `Layout` is deliberately positional (CuTe convention): shape and stride
593/// are unnamed IntTuples, which is what makes the layout algebra
594/// (composition, tiling, complement) work without dragging names through
595/// every transform.
596///
597/// Scheduling, though, needs to align axes *by name* —
598/// which iter axis corresponds to which producer axis, which stateful's
599/// core dim is which.
600///
601/// `NamedLayout` is the wrapper that pairs them: it owns the naming policy
602/// and will eventually own the name-preserving versions of layout
603/// transforms (tile, split, fuse). For now it's a plain pair — methods
604/// will accrete as the refactor lands.
605///
606/// We also introduce offsets here.
607/// In CuTe, the offset is stored on a Tensor not the Layout.
608///
609/// Invariant: `names.len() == layout.rank()` (one name per top-level mode).
610/// Hierarchical splitting under tiling (e.g. `i` → `(i_outer, i_inner)`)
611/// is deferred — the rule for how names split is a policy choice tied to
612/// whichever rewrite introduces the hierarchy.
613#[derive(Clone, Debug, PartialEq, Eq)]
614pub struct NamedLayout {
615    pub names: Vec<AxisName>,
616    /// Per-axis origin in the producer's global frame.
617    /// Nonzero values arise from pushdown that stops mid-tree
618    pub offsets: Vec<usize>,
619    pub layout: Layout,
620}
621
622/// Error from `NamedLayout::broadcast_with`. Carries the offending axis
623/// name and the conflicting values so callers can build actionable messages.
624#[derive(Debug, Clone, PartialEq, Eq, thiserror::Error)]
625pub enum LayoutMergeError {
626    /// Same axis name, different non-1 extents.
627    #[error("axis {axis:?} has incompatible sizes {lhs} vs {rhs}")]
628    ExtentMismatch { axis: String, lhs: usize, rhs: usize },
629    /// Same axis name and extent, but different non-zero offsets — the two
630    /// layouts see different regions of the producer.
631    #[error("axis {axis:?} has incompatible origins {lhs} vs {rhs}")]
632    OffsetMismatch { axis: String, lhs: usize, rhs: usize },
633}
634
635/// Layout-layer error envelope. Today only wraps [`LayoutMergeError`];
636/// future variants (transpose, reshape, tiling failures) land here.
637#[derive(Debug, Clone, PartialEq, Eq, thiserror::Error)]
638pub enum LayoutError {
639    #[error(transparent)]
640    Merge(#[from] LayoutMergeError),
641    #[error("axis {axis:?} already exists in layout")]
642    DuplicateAxis { axis: String },
643    #[error("axis {axis:?} not in layout (have {available:?})")]
644    UnknownAxis { axis: String, available: Vec<String> },
645    #[error("{detail}")]
646    Unsupported { detail: String },
647}
648
649pub type LayoutResult<T> = Result<T, LayoutError>;

651impl NamedLayout {
652    /// Construct with all offsets defaulted to zero.
653    pub fn new(names: Vec<AxisName>, layout: Layout) -> Self {
654        let offsets = vec![0; names.len()];

658    /// Construct from per-axis (extent, name) with row-major strides and
659    /// zero offsets. Standard array-layer shape — no slicing, no transpose.
660    pub fn from_shape(shape: &[usize], names: &[AxisName]) -> Self {
661        assert_eq!(
662            shape.len(),
663            names.len(),
664            "NamedLayout::from_shape: {} names for {}-axis shape",
665            names.len(),
666            shape.len()
667        );
668        Self::new(names.to_vec(), native_layout(shape))
669    }

671    /// Construct with explicit offsets. `offsets.len()` and `names.len()`
672    /// must match `layout.rank()`.
673    pub fn with_offsets(names: Vec<AxisName>, offsets: Vec<usize>, layout: Layout) -> Self {
674        assert_eq!(
675            names.len(),
676            layout.rank(),
677            "NamedLayout: {} names for rank-{} layout",
678            names.len(),
679            layout.rank()
680        );
681        assert_eq!(
682            offsets.len(),
683            names.len(),
684            "NamedLayout: {} offsets for {} names",
685            offsets.len(),
686            names.len()
687        );

691    /// Iteration extent along the named axis (i.e. `0..extent` is the loop
692    /// range). Returns `None` if the axis isn't in this space.
693    pub fn extent(&self, axis: &AxisName) -> Option<usize> {
694        let pos = self.names.iter().position(|n| n == axis)?;
695        Some(self.layout.shape.flat()[pos])
696    }

699    pub fn extents(&self) -> Vec<usize> {
700        self.layout.shape.flat()
701    }

703    /// Axis names rendered as plain strings, for [`LayoutError::UnknownAxis`]
704    /// `available` payloads (the error layer stays `String`).
705    fn available(&self) -> Vec<String> {
706        self.names.iter().map(Into::into).collect()
707    }

709    /// Append `axis` as a new innermost dimension at `extent`. Errors with
710    /// [`LayoutError::DuplicateAxis`] if the axis is already present.
711    ///
712    /// Produces a **unit-stride per axis** layout (every flat mode has stride 1),
713    /// the form composition expects on the view side of `producer ∘ view`.
714    /// Intended for building iteration-space layouts, not producer layouts.
715    /// Assumes a flat receiver (no tiled axes); [`Self::extents`] flattens
716    /// hierarchical modes.
717    pub fn append(&self, axis: &AxisName, extent: usize) -> LayoutResult<NamedLayout> {
718        if self.names.iter().any(|n| n == axis) {
719            return Err(LayoutError::DuplicateAxis { axis: axis.into() });
720        }
721        let mut names = self.names.clone();
722        let mut extents = self.extents();
723        names.push(axis.clone());
724        extents.push(extent);
725        Ok(NamedLayout::new(names, iter_layout(&extents)))
726    }

728    /// Replace the extent of an existing `axis`. Errors with
729    /// [`LayoutError::UnknownAxis`] if the axis is absent.
730    ///
731    /// Like [`Self::append`], produces a unit-stride iteration-space layout.
732    pub fn update(&self, axis: &AxisName, extent: usize) -> LayoutResult<NamedLayout> {
733        let pos = self.names.iter().position(|n| n == axis).ok_or_else(|| {
734            LayoutError::UnknownAxis { axis: axis.into(), available: self.available() }
735        })?;
736        let mut extents = self.extents();
737        extents[pos] = extent;
738        Ok(NamedLayout::new(self.names.clone(), iter_layout(&extents)))
739    }

741    /// Unify two iteration spaces by axis name, checking `n_chunks` agree
742    /// (with size-1 broadcasting).
743    ///
744    /// Per-axis rules:
745    /// - Name appears in only one side → keep it.
746    /// - Name in both, same `n_chunks` → kept as-is.
747    /// - Name in both, one side is `n_chunks == 1` → the 1 side broadcasts;
748    ///   the non-1 side wins.
749    /// - Name in both, different non-1 extents → `LayoutMergeError::ExtentMismatch`.
750    /// - Name in both, same non-1 extent but different non-zero offsets →
751    ///   `LayoutMergeError::OffsetMismatch` (the two layouts see different
752    ///   regions of the producer; this merger can't represent that).
753    ///
754    /// Per-chunk compatibility (same name, same extent, different chunk
755    /// partitions) isn't the concern here — upstream (`ChunkGrid::new`
756    /// and `broadcast_grids`) keeps the "same name + same extent ⇒ same
757    /// chunks" invariant across a plan.
758    ///
759    /// Result preserves `self`'s axis order, with `other`'s new axes
760    /// appended. Order is merge-deterministic, not semantic — callers treat
761    /// the result as an unordered set.
762    pub fn broadcast_with(&self, other: &NamedLayout) -> Result<NamedLayout, LayoutError> {
763        let init = (self.names.clone(), self.extents(), self.offsets.clone());
764        let (names, extents, offsets) =
765            other.names.iter().zip(other.extents()).zip(&other.offsets).try_fold(
766                init,
767                |(mut names, mut extents, mut offsets),
768                 ((on, oe), &oo)|
769                 -> Result<_, LayoutError> {
770                    match names.iter().position(|n| n == on) {
771                        None => {
772                            names.push(on.clone());
773                            extents.push(oe);
774                            offsets.push(oo);
775                        }
776                        Some(p) => {
777                            let (se, so) = (extents[p], offsets[p]);
778                            if se == 1 && oe != 1 {
779                                extents[p] = oe;
780                                offsets[p] = oo;
781                            } else if se == oe || oe == 1 {
782                                if se != 1 && so != oo {
783                                    return Err(LayoutMergeError::OffsetMismatch {
784                                        axis: on.into(),
785                                        lhs: so,
786                                        rhs: oo,
787                                    }
788                                    .into());
789                                }
790                            } else {
791                                return Err(LayoutMergeError::ExtentMismatch {
792                                    axis: on.into(),
793                                    lhs: se,
794                                    rhs: oe,
795                                }
796                                .into());
797                            }
798                        }
799                    }
800                    Ok((names, extents, offsets))
801                },
802            )?;
803        Ok(NamedLayout::with_offsets(names, offsets, native_layout(&extents)))
804    }

806    /// New `NamedLayout` with extra axes appended (each as `(name, extent)`,
807    /// offset 0). Strides are rebuilt row-major via `native_layout`, matching
808    /// `from_shape` / `broadcast_with`.
809    ///
810    /// Errors with [`LayoutError::DuplicateAxis`] if any appended name
811    /// collides with an existing axis or with another appended axis —
812    /// duplicate names make `extent` / `offset` lookups ambiguous.
813    pub fn extend(
814        &self,
815        axes: impl IntoIterator<Item = (AxisName, usize)>,
816    ) -> Result<NamedLayout, LayoutError> {
817        let init = (self.names.clone(), self.extents(), self.offsets.clone());
818        let (names, extents, offsets) = axes.into_iter().try_fold(
819            init,
820            |(mut names, mut extents, mut offsets), (name, extent)| {
821                if names.contains(&name) {
822                    return Err(LayoutError::DuplicateAxis { axis: name.into() });
823                }
824                names.push(name);
825                extents.push(extent);
826                offsets.push(0);
827                Ok((names, extents, offsets))
828            },
829        )?;
830        Ok(NamedLayout::with_offsets(names, offsets, native_layout(&extents)))
831    }

833    /// New `NamedLayout` with every axis whose name appears in
834    /// `names_to_drop` removed. Errors with [`LayoutError::UnknownAxis`] if
835    /// any name to drop isn't on this layout.
836    pub fn without_named(&self, names_to_drop: &[AxisName]) -> Result<NamedLayout, LayoutError> {
837        if let Some(missing) = names_to_drop.iter().find(|d| !self.names.iter().any(|n| n == *d)) {
838            return Err(LayoutError::UnknownAxis {
839                axis: missing.into(),
840                available: self.available(),
841            });
842        }
843        let (kept_names, kept_rest): (Vec<AxisName>, Vec<(usize, usize)>) = self
844            .names
845            .iter()
846            .zip(self.extents())
847            .zip(self.offsets.iter().copied())
848            .filter(|((n, _), _)| !names_to_drop.iter().any(|d| d == *n))
849            .map(|((n, e), o)| (n.clone(), (e, o)))
850            .unzip();
851        let (kept_extents, kept_offsets): (Vec<usize>, Vec<usize>) = kept_rest.into_iter().unzip();
852        Ok(NamedLayout::with_offsets(kept_names, kept_offsets, native_layout(&kept_extents)))
853    }

855    /// Identity-tiler view over this iteration layout — the per-edge view the
856    /// planner stamps and composes against a producer to translate iteration
857    /// coords into producer chunk coords.
858    ///
859    /// Per top-level axis: flat `n` → `n : 1`; tiled `(n_out, k)` →
860    /// `(n_out, k) : (k, 1)`, so `producer ∘ view` reads `c·k + j` on a
861    /// coarsened axis (broadcast on absent axes falls out of `compose`'s
862    /// name match). Names are preserved.
863    ///
864    /// Pass **stride 0** on `(axis, mode_index)` pairs the access must be
865    /// invariant under (same shape `Schedule::Loop` uses). Use case today:
866    /// a Reduce's `Apply` own-output access must be invariant under each
867    /// fold-tile `Loop` the planner just placed, so the accumulator address
868    /// doesn't move as the fold sweeps. Pass `&[]` for the plain identity-tiler.
869    pub fn ignoring_modes(&self, ignored: &[(AxisName, usize)]) -> NamedLayout {
870        let IntTuple::Tuple(modes) = &self.layout.shape else {
871            unreachable!("NamedLayout invariant: shape is a tuple of top-level modes");
872        };
873        let is_ignored = |name: &AxisName, mode_index: usize| -> bool {
874            ignored.iter().any(|(n, m)| n == name && *m == mode_index)
875        };
876        let (shapes, strides): (Vec<IntTuple>, Vec<IntTuple>) = modes
877            .iter()
878            .zip(self.names.iter())
879            .map(|(mode, name)| match mode {
880                IntTuple::Int(n) => {
881                    let stride = if is_ignored(name, 0) { 0 } else { 1 };
882                    (IntTuple::Int(*n), IntTuple::Int(stride))
883                }
884                IntTuple::Tuple(subs) => {
885                    debug_assert_eq!(subs.len(), 2, "ignoring_modes: nested tiling not supported");
886                    let (n_out, k) = (subs[0].size(), subs[1].size());
887                    let outer_ignored = is_ignored(name, 0);
888                    let inner_ignored = is_ignored(name, 1);
889                    // Eval at (c, j) = c·s_outer + j·s_inner must equal the
890                    // unit-stride logical coord into this axis:
891                    //   neither ignored → c·k + j  → (k, 1)
892                    //   inner ignored   → c        → (1, 0)
893                    //   outer ignored   → j        → (0, 1)
894                    //   both ignored    → 0        → (0, 0)
895                    let s_outer = if outer_ignored {
896                        0
897                    } else if inner_ignored {
898                        1
899                    } else {
900                        k
901                    };
902                    let s_inner = if inner_ignored { 0 } else { 1 };
903                    (
904                        IntTuple::Tuple(vec![IntTuple::Int(n_out), IntTuple::Int(k)]),
905                        IntTuple::Tuple(vec![IntTuple::Int(s_outer), IntTuple::Int(s_inner)]),
906                    )
907                }
908            })
909            .unzip();
910        NamedLayout::new(
911            self.names.clone(),
912            Layout::new(IntTuple::Tuple(shapes), IntTuple::Tuple(strides)),
913        )
914    }

916    /// Compose this layout (a flat producer chunk grid) with a per-edge `view`:
917    /// `self ∘ view`, name-aware. The result maps the view's iteration coords
918    /// to this layout's linear brick index — shapes come from `view` (the new
919    /// domain), strides from `self` (the codomain it addresses).
920    ///
921    /// For each top-level axis of `view`, the producer's `(count, stride)` is
922    /// matched **by name** — size-1 / absent collapses to the broadcast atom
923    /// `(1 : 0)` (stepping it re-reads the same brick) — then composed with the
924    /// view's mode for that axis via [`Layout::compose`]. Transpose falls out of
925    /// name matching; broadcast from the `(1:0)` atom; coarsen / projection from
926    /// the view's mode (`(n_out,k):(k,1)` reads `c·k + j`; `(n_out,k):(1,0)`
927    /// projects to `c`).
928    ///
929    /// The per-edge view (read or projection) is stamped by the planner from
930    /// the iteration space — see [`native_layout`] / [`Self::ignoring_modes`]
931    /// — so the projection-vs-read choice lives there, not in stride arithmetic
932    /// re-derived from producer counts.
933    pub fn compose(&self, view: &NamedLayout) -> Layout {
934        let producer_strides = self.layout.stride.flat();
935        let producer_counts = self.extents();
936        // Precondition: a producer grid is flat
937        // (coalesced), so `.flat()` is 1:1 with `self.names`.
938        debug_assert_eq!(
939            producer_counts.len(),
940            self.names.len(),
941            "compose: producer grid must be flat (coalesced)",
942        );
943        let (IntTuple::Tuple(view_shapes), IntTuple::Tuple(view_strides)) =
944            (&view.layout.shape, &view.layout.stride)
945        else {
946            unreachable!("NamedLayout invariant: shape/stride are tuples of top-level modes");
947        };
948        let (shapes, strides): (Vec<IntTuple>, Vec<IntTuple>) = view
949            .names
950            .iter()
951            .zip(view_shapes)
952            .zip(view_strides)
953            .map(|((name, vshape), vstride)| {
954                // Producer count/stride by name; size-1/absent → broadcast (1, 0).
955                let (count, stride) = match self.names.iter().position(|n| n == name) {
956                    Some(p) if producer_counts[p] > 1 => (producer_counts[p], producer_strides[p]),
957                    _ => (1, 0),
958                };
959                compose_impl(&IntTuple::Int(count), &IntTuple::Int(stride), vshape, vstride)
960            })
961            .unzip();
962        Layout::new(IntTuple::Tuple(shapes), IntTuple::Tuple(strides))
963    }

965    /// Split the named axis by an integer `tile`, in place, keeping the axis
966    /// name and offset. The top-level mode `axis : n` becomes the hierarchical
967    /// `axis : (ceil(n/tile), tile)` via row-major [`Layout::logical_divide`]
968    /// — depth 0 is the surviving rest mode (keeps the name), depth 1 the
969    /// folded tile mode. `names`/`offsets` are unchanged (still one entry per
970    /// top-level mode); the inner [`Layout`]'s shape tuple carries the
971    /// per-depth extents.
972    ///
973    /// Errors with [`LayoutError::UnknownAxis`] if `axis` is absent, or
974    /// [`LayoutError::Unsupported`] if the mode is already tiled (re-splitting
975    /// a hierarchical mode — i.e. nested tiling — is deferred).
976    pub fn logical_divide(&self, axis: &AxisName, tile: usize) -> LayoutResult<NamedLayout> {
977        let pos = self.names.iter().position(|n| n == axis).ok_or_else(|| {
978            LayoutError::UnknownAxis { axis: axis.into(), available: self.available() }
979        })?;
980        let (IntTuple::Tuple(shape_modes), IntTuple::Tuple(stride_modes)) =
981            (&self.layout.shape, &self.layout.stride)
982        else {
983            unreachable!("NamedLayout invariant: shape/stride are tuples of top-level modes");
984        };
985        if !matches!(shape_modes[pos], IntTuple::Int(_)) {
986            return Err(LayoutError::Unsupported {
987                detail: format!(
988                    "logical_divide on already-tiled axis {axis:?}: nested tiling is not yet supported"
989                ),
990            });
991        }
992        let split =
993            Layout::new(shape_modes[pos].clone(), stride_modes[pos].clone()).logical_divide(tile);
994        let mut new_shape = shape_modes.clone();
995        let mut new_stride = stride_modes.clone();
996        new_shape[pos] = split.shape;
997        new_stride[pos] = split.stride;
998        Ok(NamedLayout::with_offsets(
999            self.names.clone(),
1000            self.offsets.clone(),
1001            Layout::new(IntTuple::Tuple(new_shape), IntTuple::Tuple(new_stride)),
1002        ))
1003    }

1005    /// The named top-level mode as its own [`Layout`] — name resolution over
1006    /// CuTe positional [`Layout::get`]. Index into it with `get(i)` for sub-modes (e.g.
1007    /// `get(1)` is the tile mode of a [`Self::logical_divide`]d axis), `size()`
1008    /// for extents, `rank()` for the sub-mode count (the valid `mode_index`
1009    /// range: 1 flat, 2 after one divide).
1010    pub fn mode(&self, axis: &AxisName) -> LayoutResult<Layout> {
1011        let pos = self.names.iter().position(|n| n == axis).ok_or_else(|| {
1012            LayoutError::UnknownAxis { axis: axis.into(), available: self.available() }
1013        })?;
1014        Ok(self.layout.get(pos))
1015    }

1017    /// Evaluate the inner [`Layout`] from a [`NamedCoord`] (iteration indices
1018    /// keyed by `(axis, mode_index)`, where `mode_index` is the CuTe `get<I>`
1019    /// position within the axis's mode).
1020    ///
1021    /// For each top-level mode, assembles its natural coord as an [`IntTuple`]
1022    /// congruent to that mode (each leaf drawn from `(axis, position)`,
1023    /// `get<0>, get<1>, …`), then evaluates via [`Layout::crd2idx`] — the
1024    /// hierarchical coord is fed straight through, no linearization. A flat
1025    /// (untiled) axis is a single position-0 leaf, so this also covers the
1026    /// flat case.
1027    ///
1028    /// Returns [`LayoutError::UnknownAxis`] (keyed `"axis@pos"`) if any required
1029    /// `(axis, mode_index)` is missing from `indices`.
1030    pub(crate) fn eval_axis(&self, axis: &AxisName, indices: &NamedCoord) -> LayoutResult<usize> {
1031        let mode = self.mode(axis)?;
1032        let (coord, _) = natural_coord(&mode.shape, axis, 0, indices)?;
1033        Ok(mode.crd2idx(&coord))
1034    }

1041/// Build a unit-stride iteration-space layout — every axis has stride 1.
1042///
1043/// `compose` works as `(producer ∘ view)(iter) = producer(view(iter))`.
1044/// For each axis, `view.eval(c) = c · view_stride`. If we want the iter
1045/// coord `c` to land directly at producer chunk `c · producer_stride` on
1046/// that axis (with the producer's stride providing the chunk-grid's
1047/// linear addressing), the view's stride must be 1 — anything else
1048/// scales `c` before it reaches the producer.
1049///
1050/// Use this for **loop iter-space layouts**: the planner's `outer_iter`,
1051/// `Schedule::Domain.iter_space`, and the layouts that `compose` consumes
1052/// on the view side. For **producer-side layouts** — each node's chunk
1053/// grid, where the strides encode row-major chunk addressing so
1054/// `unravel_index` can decode the composed result — use [`native_layout`]
1055/// instead.
1056pub fn iter_layout(extents: &[usize]) -> Layout {
1057    let shape = IntTuple::Tuple(extents.iter().copied().map(IntTuple::Int).collect());
1058    let stride = IntTuple::Tuple(std::iter::repeat_n(IntTuple::Int(1), extents.len()).collect());
1059    Layout::new(shape, stride)
1060}
1061
1062/// Build the native layout for a node's chunk grid.
1063///
1064/// Row-major strides, with stride 0 on dims that have only 1 chunk
1065/// (these are invariant — broadcast or trivial). This ensures composition
1066/// propagates stride 0 correctly for broadcast dims.
1067pub fn native_layout(grid: &[usize]) -> Layout {
1068    let strides_rev: Vec<usize> = grid
1069        .iter()
1070        .rev()
1071        .scan(1usize, |acc, &g| {
1072            if g > 1 {
1073                let s = *acc;
1074                *acc *= g;
1075                Some(s)
1076            } else {
1077                Some(0)
1078            }
1079        })
1080        .collect();
1081    let shape = IntTuple::Tuple(grid.iter().copied().map(IntTuple::Int).collect());
1082    let stride = IntTuple::Tuple(strides_rev.into_iter().rev().map(IntTuple::Int).collect());
1083    Layout::new(shape, stride)
1084}

1086/// Decompose a linear index into multi-dim coordinates (row-major).
1087pub fn linear_to_coord(idx: usize, shape: &[usize]) -> Vec<usize> {
1088    let mut coord: Vec<usize> = shape
1089        .iter()
1090        .rev()
1091        .scan(idx, |remaining, &s| {
1092            let c = *remaining % s;
1093            *remaining /= s;
1094            Some(c)
1095        })
1096        .collect();
1097    coord.reverse();
1098    coord
1099}
