1/// Schedule-based execution: the planner produces a Schedule (loop nest +
2/// placement), the executor interprets it. Halide-inspired separation of
3/// algorithm (what to compute) from schedule (how to iterate).
4///
5/// All types here are shared between the planner (constructs) and the
6/// executor (interprets).
7///
8/// ## Core-dim driven three-phase pattern
9///
10/// Every node that declares core dimensions (dims it fully consumes)
11/// follows a three-phase lifecycle:
12///
13/// - **Init**: set up state before iterating core dims (e.g., accumulator
14///   for reduce, running state for scan, buffer for window).
15/// - **Apply** (inside core-dim loop): per-iteration work — fold one brick
16///   into the accumulator, update running state, etc.
17/// - **Finalize**: extract result after all core-dim iterations complete
18///   (e.g., accumulator → output brick).
19///
20/// Nodes with no core dims (sources, elementwise blockwise) are purely
21/// pointwise — just Apply at every iteration, no Init/Finalize.
22///
23/// The schedule builder emits Init before the core-dim loops, Apply inside
24/// them, and Finalize after, based solely on whether a node has core dims.
25/// This generalizes to any future op type (scan, window, etc.) without
26/// special-casing in the schedule.
27use std::collections::HashMap;
28use std::{
29    fmt::{self, Debug},
30    ops::Range,
31    sync::Arc,
32};

39};

71/// A schedule is a tree of nested loops with computation steps placed at
72/// specific loop levels. The executor interprets it — making no decisions
73/// except how/whether to parallelize (if allowed); prefetching of sources, etc.
74///
75/// Every compute step (Init/Apply/Write) carries a precomputed
76/// [`AccessRelation`] — the translation from this band's
77/// iteration space to the producer's iteration space. The planner
78/// computes it once via `producer_grid.layout.compose(view)` at schedule
79/// emission. The executor evaluates it against current loop indices to get the brick coord.
80///
81/// Today access is identity-name-keyed (iter_space is built directly from
82/// the producer grid), but the slot exists so future iter_space rewrites
83/// (strip mining, slice-pushdown, distributed sub-bands) don't require
84/// schema changes.
85///
86/// The three-phase pattern (Init/Apply/Finalize) is driven by core dims:
87/// - Nodes with core dims get all three phases placed at different loop levels.
88/// - Nodes without core dims get only Apply.
89#[derive(Debug)]
90// ANCHOR: schedule
91pub enum Schedule {
92    /// Establish the iteration space for the enclosed body.
93    /// `Loop` nodes inside look up their axis extents from here. One `Domain` per band;
94    /// multi-band plans nest or sequence several.
95    /// A distributed executor may construct Schedules for each worker by simply
96    /// rewriting these Domain nodes to the appropriate subspace the worker is responsible for.
97    Domain { iter_space: IterationSpace, body: Box<Schedule> },
98
99    /// Iterate over a chunk dimension, identified by axis name.
100    /// The range is looked up from the enclosing `Domain` at execute time.
101    /// The enclosing `Domain`'s iteration space may be nested (e.g. for coarsen/rolling)
102    /// So we include a `mode_index` to grab the "i-th mode" or "i-th sublayout"
103    /// (Czeka 2026 Section 2.4.1) of the enclosing Domain's `iter_space`.
104    Loop { axis: AxisName, mode_index: usize, body: Vec<Schedule> },
105
106    /// Co-iterate `axes` over an explicit coordinate table
107    /// (soon to be) filtered by the overarching Domain range (similar to Loop).
108    /// `coords[i]` is the i-th present coord, length `axes.len()`,
109    /// values in `axes` positional order.
110    Zip { axes: Vec<AxisName>, coords: Arc<Vec<Vec<u32>>>, body: Vec<Schedule> },
111
112    /// Execute a list of schedule steps in *specified* order at this loop level.
113    /// Used to express top-level sequences (e.g. InitSink, loop nest,
114    /// FinalizeSink).
115    Sequence(Vec<Schedule>),
116
117    /// Execute a list of schedule steps in *any* order at this loop level.
118    /// Unused today
119    // Set(Vec<Schedule>),
120
121    /// Initialize state for a compute node that has core dimensions.
122    /// Placed before the core-dim loops begin.
123    ///
124    /// Example, for Reduce, this creates accumulator state sized to the output brick.
125    ///
126    /// `access` is this node's iter→global access relation — used to derive the
127    /// output coord that sizes the accumulator's initial state.
128    Init { node: NodeName, access: AccessRelation },
129
130    /// Per-iteration "compute" step. Examples:
131    /// - Source: read brick at current coord into workspace.
132    /// - Blockwise: read inputs from workspace, broadcast, apply op, store result.
133    /// - Reduce: fold `workspace[input]` into accumulator.
134    ///
135    /// `inputs` names the workspace entries this step reads, in port order.
136    /// Each input brick's coordinate was already fixed by the producer's own
137    /// `Apply` (recompute model: the emission immediately precedes this read),
138    /// so the edge carries no access of its own.
139    /// The planner controls placement — a source Apply can be hoisted above
140    /// inner loops for reuse when broadcasting is involved.
141    ///
142    /// `access` is this node's own iter→global translation (e.g. for the accumulator result).
143    ///
144    /// Sinks never appear in Apply — they have their own InitSink / WriteSink /
145    /// FinalizeSink variants.
146    Apply { node: NodeName, inputs: Vec<NodeName>, access: AccessRelation },
147
148    /// Extract result from a compute node's state after core-dim iteration
149    /// completes. Placed after the core-dim loops exit. Produces a brick
150    /// in workspace for routing to the next step.
151    ///
152    /// For Reduce: calls accumulator.finalize() → workspace brick.
153    Finalize { node: NodeName },
154
155    /// FIXME: These sink-specific variants are a wart to me.
156    /// Initialize a sink (allocate buffers, open files, etc.). Placed once,
157    /// before any WriteSink into that sink executes.
158    InitSink { sink: NodeName },
159
160    /// Write `workspace[node]` into the sink at the current brick coord, and
161    /// release `workspace[node]`. This is the explicit memory-release point
162    /// for a brick on its way out of the system.
163    ///
164    /// `access` is this node's iter→global translation — used to derive the
165    /// brick coord at which to write into the sink.
166    WriteSink { node: NodeName, sink: NodeName, access: AccessRelation },
167
168    /// Finalize a sink (commit writes, close files, etc.). Placed once,
169    /// after every WriteSink into that sink has executed.
170    FinalizeSink { sink: NodeName },
171}

174/// The **affine half** of an edge's access relation — the per-axis map from
175/// iter coords to producer chunk coords that any framing / transposition /
176/// broadcast can be expressed as without per-coord lookup tables.
177///
178/// A single [`NamedLayout`] carrying the consumer's iter view with strides
179/// canonicalized for this edge: the planner-supplied `invariant_under`
180/// `(axis, mode_index)` pairs are zero-strided, and every sub-mode of any
181/// size-1 producer axis is also zero-strided (so the eval returns 0 on
182/// that axis regardless of iter coord — the broadcast case). Axes the
183/// view doesn't carry default to 0 in [`Self::eval_axis`]; those are
184/// `Zipped` follower axes whose real coord is stamped by the override
185/// pass.
186#[derive(Debug, Clone)]
187pub struct AffineAccess(NamedLayout);
188
189impl AffineAccess {
190    /// Build the affine access for one consumer→producer edge.
191    ///
192    /// `consumer` is the consumer's loop iter-space layout;
193    /// `invariant_under` lists `(axis, mode_index)` pairs whose
194    /// `Schedule::Loop` this edge's access must be invariant under
195    /// (typically a Reduce's fold-tile modes for its own-output access).
196    /// Size-1 producer axes are folded into the invariant set so the
197    /// view's eval returns 0 on those axes.
198    pub fn for_edge(
199        producer: &IterationSpace,
200        consumer: &NamedLayout,
201        invariant_under: &[(AxisName, usize)],
202    ) -> Self {
203        let mut invariant = invariant_under.to_vec();
204        for (axis, &extent) in producer.names().iter().zip(producer.max_extents().iter()) {
205            if extent == 1
206                && let Ok(mode) = consumer.mode(axis)
207            {
208                for m in 0..mode.rank() {
209                    invariant.push((axis.clone(), m));
210                }
211            }
212        }
213        Self(consumer.ignoring_modes(&invariant))
214    }
215
216    /// Producer chunk coord on `axis` for the given iter `indices`.
217    ///
218    /// Defaults to 0 when `axis` isn't in this view — either a size-1
219    /// broadcast the consumer doesn't iterate, or a `Zipped` follower
220    /// whose real coord is stamped by the override pass. In-view
221    /// broadcasts already evaluate to 0 thanks to the zero-strides baked
222    /// in by [`Self::for_edge`].
223    pub fn eval_axis(&self, axis: &AxisName, indices: &NamedCoord) -> usize {
224        self.0.eval_axis(axis, indices).unwrap_or(0)
225    }
226
227    /// Borrow the inner [`NamedLayout`] — for visualization and tests
228    /// that want to inspect the per-axis mode/stride pattern. Not used on
229    /// the resolution hot path.
230    pub fn layout(&self) -> &NamedLayout {
231        &self.0
232    }
233}

235/// How a Schedule node accesses a producer's chunk grid given the enclosing
236/// `Domain.iter_space`. Split into two halves:
237///
238/// - `affine` — per-axis affine map from iter coords to producer chunk
239///   coords (transposition, broadcast, framing). See [`AffineAccess`].
240/// - `overrides` — per-coord non-affine residual (slice offsets, gather
241///   lookup tables) applied after the affine pass. Ordered: the planner
242///   accumulates entries in descent order (downstream-most edge first)
243///   while threading them down its recursive emission, and they are
244///   applied in that order — stacked overrides on one axis compose,
245///   e.g. two slice offsets `+1` then `+2` resolve to `+3`.
246#[derive(Debug, Clone)]
247pub struct AccessRelation {
248    pub affine: AffineAccess,
249    pub overrides: TranslatedAccessStack,
250}
251
252impl AccessRelation {
253    /// Construct an [`AccessRelation`] from a per-edge affine access and
254    /// the override stack threaded down from downstream consumer edges.
255    pub fn new(affine: AffineAccess, overrides: &[(AxisName, TranslatedAccess)]) -> Self {
256        Self { affine, overrides: overrides.to_vec() }
257    }
258
259    /// Derive the producer's native chunk coordinate for the current
260    /// iteration step.
261    ///
262    /// Two passes:
263    ///
264    /// 1. **Affine pass.** For each producer axis, evaluate
265    ///    [`AffineAccess::eval_axis`] against the iter coords. Axes the
266    ///    affine access doesn't carry (broadcast / size-1) default to 0.
267    ///
268    /// 2. **Override pass.** Apply each [`Self::overrides`] entry, in
269    ///    stack order (downstream first).
270    pub fn resolve_coord(
271        &self,
272        producer_grid: &IterationSpace,
273        indices: &NamedCoord,
274    ) -> BrixResult<BrickCoordinate> {
275        let mut coord: Vec<usize> =
276            producer_grid.names().iter().map(|axis| self.affine.eval_axis(axis, indices)).collect();
277
278        for (axis, entry) in &self.overrides {
279            match entry {
280                TranslatedAccess::Independent(a) => {
281                    let producer_axis = producer_grid.axis_position(axis)?;
282                    coord[producer_axis] = a.translate(coord[producer_axis]);
283                }
284                TranslatedAccess::Zipped(followers) => {
285                    debug_assert!(
286                        producer_grid.axis_position(axis).is_err(),
287                        "Zipped leader axis {axis:?} must not appear on producer grid",
288                    );
289                    let leader_idx = indices
290                        .get_leading(axis)
291                        .expect("invariant: Zipped leader axis indexed by enclosing loop");
292                    // Each follower's `AxisOverride` is driven by the leader's iter coord.
293                    for (follower_axis, follower_access) in followers {
294                        let producer_axis = producer_grid.axis_position(follower_axis)?;
295                        coord[producer_axis] = follower_access.translate(leader_idx);
296                    }
297                }
298            }
299        }
300        Ok(BrickCoordinate::from_named(producer_grid.names(), &coord))
301    }
302}

308/// What a node is and what it holds. Data flow (inputs) is in the Schedule,
309/// not here.
310pub enum NodeInfo {
311    /// Unified source node. Concrete `Source` impls (in `array/`) bake
312    /// their element-sized `ChunkGrid` in and stamp each brick's `bounds`
313    /// at production time; brix sees only the counts-only `IterationSpace`.
314    Source {
315        source: Arc<dyn Source>,
316        space: IterationSpace,
317    },
318    /// Blockwise computation. Per-input chunk-coord routing is carried on
319    /// each edge's `ChunkAccess` (set by lowering) and propagated up via
320    /// the planner's coord transforms onto each producer's `Apply.access`.
321    Blockwise {
322        kernel: Arc<dyn Kernel>,
323        space: IterationSpace,
324    },
325    Reduce {
326        accumulator: Arc<dyn Accumulator>,
327        output_space: IterationSpace,
328    },
329    /// A sink node — records the space the sink expects to receive. The
330    /// concrete `&dyn Sink` is bound at execute time, keyed by this node's
331    /// NodeId. Lifecycle lives in the Schedule (InitSink / WriteSink /
332    /// FinalizeSink) rather than in Apply.
333    Sink {
334        space: IterationSpace,
335    },
336}
337
338impl Debug for NodeInfo {
339    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
340        match self {
341            NodeInfo::Source { source, space } => {
342                write!(f, "Source {{ source: {source:?}, space: {space:?} }}")
343            }
344            NodeInfo::Blockwise { kernel, space } => {
345                write!(f, "Blockwise {{ kernel: {kernel:?}, space: {space:?} }}")
346            }
347            NodeInfo::Reduce { accumulator, output_space } => {
348                write!(
349                    f,
350                    "Reduce {{ accumulator: {accumulator:?}, output_space: {output_space:?} }}"
351                )
352            }
353            NodeInfo::Sink { space } => write!(f, "Sink {{ space: {space:?} }}"),
354        }
355    }
356}
357
358impl NodeInfo {
359    pub fn space(&self) -> &IterationSpace {
360        match self {
361            NodeInfo::Source { space, .. }
362            | NodeInfo::Blockwise { space, .. }
363            | NodeInfo::Sink { space, .. } => space,
364            NodeInfo::Reduce { output_space, .. } => output_space,
365        }
366    }
367}

373// ANCHOR: physical_plan
374/// The physical plan: a node table + a schedule. The planner constructs it,
375/// the executor interprets it.
376///
377/// A node's brick coordinate for a given iteration is *not* stored — it's
378/// computed on demand by the executor from `NodeInfo::grid()` (the producer's
379/// native chunk grid) against the enclosing `Schedule::Domain`'s `iter_space`.
380/// This keeps the plan free of redundant state and lets future schedule
381/// rewrites (distribution, strip mining) change only the `iter_space` on
382/// Domain nodes.
383///
384/// Sinks live in `nodes` as `NodeInfo::Sink` entries — iterate the node
385/// table to find them. There is no separate outputs map.
386#[derive(Debug)]
387pub struct PhysicalPlan {
388    pub nodes: HashMap<NodeName, NodeInfo>,
389    pub schedule: Schedule,
390}
391// ANCHOR_END: physical_plan

394// Sketch — compact structural representation for display and assertions
395// ---------------------------------------------------------------------------

397/// A compact, `PartialEq`-able representation of a schedule tree.
398/// Used for pretty test assertions.
399/// Shows each node's `NodeId` — which is the user-supplied label when
400/// `Array::labeled(...)` was used, otherwise the op-kind prefix (`"source"`,
401/// `"blockwise"`, `"reduce"`, `"sink"`, …). Collisions on the same name
402/// get a `#N` suffix. This lets tests read like the expression algebra
403/// (`Apply("A"), Apply("sum_a", ["A"])`) without depending on positional
404/// counters.
405#[derive(Debug, Clone, PartialEq)]
406pub enum Sketch {
407    /// Domain axes are displayed sorted by name so assertions are
408    /// order-independent (the underlying `HashMap` is unordered).
409    Domain {
410        axes: Vec<(String, Range<usize>)>,
411        body: Box<Sketch>,
412    },
413    Loop {
414        axis: String,
415        mode_index: usize,
416        body: Vec<Sketch>,
417    },
418    /// `Zip { sparse_axes, n_present, body }` — co-iteration over a
419    /// present-coord table. `sparse_axes` is sorted in the sketch so
420    /// assertions are order-independent.
421    Zip {
422        axes: Vec<String>,
423        n_present: usize,
424        body: Vec<Sketch>,
425    },
426    Sequence(Vec<Sketch>),
427    Init(String),
428    Apply(String, Vec<String>),
429    Finalize(String),
430    InitSink(String),
431    WriteSink(String),
432    FinalizeSink(String),
433}
434
435impl Sketch {
436    /// Build a [`Sketch::Domain`], sorting `axes` by name so the result is
437    /// order-independent (the underlying `IterationSpace` is unordered). The
438    /// sort lives here, not in the `schedule!` macro, so every construction
439    /// path — macro, [`PhysicalPlan::sketch`], or hand-built — shares it.
440    pub fn domain(mut axes: Vec<(String, Range<usize>)>, body: Sketch) -> Sketch {
441        axes.sort_by(|a, b| a.0.cmp(&b.0));
442        Sketch::Domain { axes, body: Box::new(body) }
443    }
444}
445
446impl PhysicalPlan {
447    /// Convert the schedule tree to a [`Sketch`] for display, logging, and
448    /// test assertions.
449    pub fn sketch(&self) -> Sketch {
450        self.sketch_step(&self.schedule)
451    }
452
453    fn sketch_step(&self, s: &Schedule) -> Sketch {
454        match s {
455            Schedule::Domain { iter_space, body } => {
456                let extents = iter_space.max_extents();
457                let axes: Vec<(String, Range<usize>)> = iter_space
458                    .names()
459                    .iter()
460                    .zip(extents.iter())
461                    .map(|(n, &k)| (n.to_string(), 0..k))
462                    .collect();
463                Sketch::domain(axes, self.sketch_step(body))
464            }
465            Schedule::Loop { axis, body, mode_index } => Sketch::Loop {
466                axis: axis.to_string(),
467                mode_index: *mode_index,
468                body: body.iter().map(|step| self.sketch_step(step)).collect(),
469            },
470            Schedule::Zip { axes, coords, body } => Sketch::Zip {
471                axes: axes.iter().map(|a| a.to_string()).collect(),
472                n_present: coords.len(),
473                body: body.iter().map(|step| self.sketch_step(step)).collect(),
474            },
475            Schedule::Sequence(body) => {
476                Sketch::Sequence(body.iter().map(|step| self.sketch_step(step)).collect())
477            }
478            Schedule::Init { node, .. } => Sketch::Init(node.0.clone()),
479            Schedule::Apply { node, inputs, .. } => {
480                Sketch::Apply(node.0.clone(), inputs.iter().map(|name| name.0.clone()).collect())
481            }
482            Schedule::Finalize { node } => Sketch::Finalize(node.0.clone()),
483            Schedule::InitSink { sink } => Sketch::InitSink(sink.0.clone()),
484            Schedule::WriteSink { node, .. } => Sketch::WriteSink(node.0.clone()),
485            Schedule::FinalizeSink { sink } => Sketch::FinalizeSink(sink.0.clone()),
486        }
487    }
488}

494/// Validate structural invariants the planner must uphold on a `PhysicalPlan`.
495///
496/// Panics on violation — these are planner bugs, not user errors. Called
497/// under `#[cfg(debug_assertions)]` at the tail of `planner::plan` so the
498/// whole plan is validated once at build time; the executor then trusts it
499/// and short-circuits with `.expect("invariant: ...")` on the same conditions
500/// as defence-in-depth for release builds.
501///
502/// Checks covered:
503/// 1. Every `Loop`/`Init`/`Apply`/`Finalize`/`WriteSink` has a `Domain` ancestor.
504/// 2. Every `Loop { axis }` has `axis` in the nearest enclosing Domain's
505///    `iter_space.names`.
506/// 3. Every `Init`/`Finalize { node }` references a stateful compute node
507/// 4. Every `Apply { node, .. }`: `nodes[node]` is not `NodeInfo::Sink`.
508/// 6. Every `WriteSink { node, sink }`: `nodes[sink]` is `NodeInfo::Sink`,
509///    `nodes[node]` exists.
510/// 7. Every `InitSink`/`FinalizeSink { sink }`: `nodes[sink]` is `NodeInfo::Sink`.
511/// 8. Every `NodeName` referenced exists in `plan.nodes`.
512/// 9. `plan.nodes` contains at least one `NodeInfo::Sink`.
513///
514/// Temporal invariants (Init precedes Apply/Finalize for a stateful node;
515/// WriteSink preceded by a workspace-producing step) are deferred — they
516/// require a traversal-order model worth a separate pass.
517pub fn assert_planner_invariants(plan: &PhysicalPlan) {
518    assert!(
519        plan.nodes.values().any(|n| matches!(n, NodeInfo::Sink { .. })),
520        "invariant: plan must contain at least one Sink node"
521    );
522    check_step(&plan.schedule, &plan.nodes, None);
523}
524
525/// Invariants the planner requires of its input `ExprGraph`.
526///
527/// Called from `planner::plan` before anything else. Violations would
528/// indicate a pipeline-wiring bug, not a user error.
529///
530/// Currently empty: no remaining invariants on the input graph.
531pub fn assert_input_graph_invariants(_graph: &crate::brix::expr_graph::ExprGraph) {}
532
533fn check_step(
534    step: &Schedule,
535    nodes: &HashMap<NodeName, NodeInfo>,
536    iter_space: Option<&IterationSpace>,
537) {
538    match step {
539        Schedule::Domain { iter_space, body } => {
540            check_step(body, nodes, Some(iter_space));
541        }
542        Schedule::Loop { axis, body, mode_index } => {
543            let space = iter_space.expect("invariant: Loop must be inside a Domain");
544            // `mode` errors (UnknownAxis, listing available names) if the axis
545            // is absent from the enclosing Domain's iter_space.
546            let rank = space
547                .mode(axis)
548                .expect("invariant: Loop axis must be in enclosing Domain's iter_space")
549                .rank();
550            assert!(
551                *mode_index < rank,
552                "invariant: Loop on axis {axis:?} addresses mode_index {mode_index}, \
553                 but its mode has rank {rank} (valid mode_index: 0..{rank})",
554            );
555            for sub in body {
556                check_step(sub, nodes, iter_space);
557            }
558        }
559        Schedule::Zip { axes, body, .. } => {
560            let space = iter_space.expect("invariant: Zip must be inside a Domain");
561            for axis in axes {
562                assert!(
563                    space.names().iter().any(|n| n == axis),
564                    "invariant: Zip sparse axis {axis:?} must be in enclosing Domain's iter_space (names: {:?})",
565                    space.names()
566                );
567            }
568            for sub in body {
569                check_step(sub, nodes, iter_space);
570            }
571        }
572        Schedule::Sequence(body) => {
573            for sub in body {
574                check_step(sub, nodes, iter_space);
575            }
576        }
577        Schedule::Init { node, .. } => {
578            iter_space.expect("invariant: Init must be inside a Domain");
579            let info = nodes
580                .get(node)
581                .unwrap_or_else(|| panic!("invariant: Init references unknown node {node:?}"));
582            assert!(
583                matches!(info, NodeInfo::Reduce { .. }),
584                "invariant: Init node {node:?} must be stateful (Reduce)"
585            );
586        }
587        Schedule::Apply { node, inputs, .. } => {
588            iter_space.expect("invariant: Apply must be inside a Domain");
589            let info = nodes
590                .get(node)
591                .unwrap_or_else(|| panic!("invariant: Apply references unknown node {node:?}"));
592            assert!(
593                !matches!(info, NodeInfo::Sink { .. }),
594                "invariant: Apply must not target Sink node {node:?}"
595            );
596            for input in inputs {
597                assert!(
598                    nodes.contains_key(input),
599                    "invariant: Apply input {input:?} not in node table"
600                );
601            }
602        }
603        Schedule::Finalize { node } => {
604            iter_space.expect("invariant: Finalize must be inside a Domain");
605            let info = nodes
606                .get(node)
607                .unwrap_or_else(|| panic!("invariant: Finalize references unknown node {node:?}"));
608            assert!(
609                matches!(info, NodeInfo::Reduce { .. }),
610                "invariant: Finalize node {node:?} must be stateful (Reduce)"
611            );
612        }
613        Schedule::InitSink { sink } => {
614            let info = nodes
615                .get(sink)
616                .unwrap_or_else(|| panic!("invariant: InitSink references unknown node {sink:?}"));
617            assert!(
618                matches!(info, NodeInfo::Sink { .. }),
619                "invariant: InitSink target {sink:?} must be a Sink node"
620            );
621        }
622        Schedule::WriteSink { node, sink, .. } => {
623            iter_space.expect("invariant: WriteSink must be inside a Domain");
624            let sink_info = nodes
625                .get(sink)
626                .unwrap_or_else(|| panic!("invariant: WriteSink references unknown sink {sink:?}"));
627            assert!(
628                matches!(sink_info, NodeInfo::Sink { .. }),
629                "invariant: WriteSink target {sink:?} must be a Sink node"
630            );
631            assert!(
632                nodes.contains_key(node),
633                "invariant: WriteSink source {node:?} not in node table"
634            );
635        }
636        Schedule::FinalizeSink { sink } => {
637            let info = nodes.get(sink).unwrap_or_else(|| {
638                panic!("invariant: FinalizeSink references unknown node {sink:?}")
639            });
640            assert!(
641                matches!(info, NodeInfo::Sink { .. }),
642                "invariant: FinalizeSink target {sink:?} must be a Sink node"
643            );
644        }
645    }
646}

648/// Build a [`Sketch`] from concise syntax. Body elements wrapped in `{}`.
649///
650/// ```ignore
651/// schedule!(Loop("i", 0..2, [
652///     {Apply(DenseSource)},
653///     {Apply(Blockwise, [DenseSource])},
654///     {WriteSink(Blockwise)},
655/// ]))
656/// ```
657/// Axis names are string literals (user data). Node tags like `DenseSource`
658/// remain bare idents (they correspond to `NodeInfo` variants, not user data).
659///
660/// Exported with `#[macro_export]` (not path-based `pub use`): the `brix`
661/// module is `pub(crate)`, so a path export can't cross the crate boundary,
662/// and the integration tests in `tests/` build the expected `Sketch` with this.
663#[macro_export]
664macro_rules! schedule {
665    (@item Apply($tag:ident)) => {
666        $crate::schedule::Sketch::Apply(stringify!($tag).to_string(), vec![])
667    };
668    (@item Apply($tag:ident, [$($inp:ident),* $(,)?])) => {
669        $crate::schedule::Sketch::Apply(
670            stringify!($tag).to_string(),
671            vec![$(stringify!($inp).to_string()),*],
672        )
673    };
674    (@item Init($tag:ident)) => {
675        $crate::schedule::Sketch::Init(stringify!($tag).to_string())
676    };
677    (@item Finalize($tag:ident)) => {
678        $crate::schedule::Sketch::Finalize(stringify!($tag).to_string())
679    };
680    (@item InitSink($tag:ident)) => {
681        $crate::schedule::Sketch::InitSink(stringify!($tag).to_string())
682    };
683    (@item WriteSink($tag:ident)) => {
684        $crate::schedule::Sketch::WriteSink(stringify!($tag).to_string())
685    };
686    (@item FinalizeSink($tag:ident)) => {
687        $crate::schedule::Sketch::FinalizeSink(stringify!($tag).to_string())
688    };
689    // `Loop("a", mode, [..])` pins the sub-layout position; the common
690    // `Loop("a", [..])` form defaults to `mode_index 0` (the flat / surviving
691    // mode), so existing fixtures need no change.
692    (@item Loop($axis:literal, $mode:literal, [$($body:tt),* $(,)?] $(,)?)) => {
693        $crate::schedule::Sketch::Loop {
694            axis: $axis.to_string(),
695            mode_index: $mode,
696            body: vec![$($crate::schedule!(@unwrap $body)),*],
697        }
698    };
699    (@item Loop($axis:literal, [$($body:tt),* $(,)?] $(,)?)) => {
700        $crate::schedule::Sketch::Loop {
701            axis: $axis.to_string(),
702            mode_index: 0,
703            body: vec![$($crate::schedule!(@unwrap $body)),*],
704        }
705    };
706    (@item Domain([$(($ax:literal, $lo:literal..$hi:literal)),* $(,)?], $body:tt $(,)?)) => {
707        $crate::schedule::Sketch::domain(
708            vec![$(($ax.to_string(), $lo..$hi)),*],
709            $crate::schedule!(@unwrap $body),
710        )
711    };
712    (@item Sequence([$($body:tt),* $(,)?] $(,)?)) => {
713        $crate::schedule::Sketch::Sequence(
714            vec![$($crate::schedule!(@unwrap $body)),*],
715        )
716    };
717    (@unwrap { $($inner:tt)+ }) => {
718        $crate::schedule!(@item $($inner)+)
719    };
720    ($($tokens:tt)+) => {
721        $crate::schedule!(@item $($tokens)+)
722    };
723}
