31/// Convert a logical expression graph into a physical plan with a schedule.
32pub fn plan(mid: &Expr) -> BrixResult<PhysicalPlan> {
33    let (graph, root) = build_expr_graph(mid);
34
35    schedule::assert_input_graph_invariants(&graph);
36
37    let (mut table, order) = build_nodes(&graph, root);
38
39    let sink_grid = table.space_of(root).clone();
40    let sink_node_id = table.add_sink(sink_grid);
41
42    let domain_schedule = emit_schedule(&graph, &order, &table, root, &sink_node_id)?;
43
44    let schedule = Schedule::Sequence(
45        once(Schedule::InitSink { sink: sink_node_id.clone() })
46            .chain(once(domain_schedule))
47            .chain(once(Schedule::FinalizeSink { sink: sink_node_id.clone() }))
48            .collect(),
49    );
50
51    let plan = PhysicalPlan { nodes: table.nodes, schedule };
52    schedule::assert_planner_invariants(&plan);
53    Ok(plan)
54}

60#[derive(Debug, Default)]
61struct IdGen {
62    counters: HashMap<String, usize>,
63}
64
65impl IdGen {
66    fn next(&mut self, label: Option<&str>, fallback_prefix: &str) -> NodeName {
67        let base = label.unwrap_or(fallback_prefix);
68        let count = self.counters.entry(base.to_string()).or_insert(0);
69        let id = if *count == 0 {
70            NodeName::new(base.to_string())
71        } else {
72            NodeName::new(format!("{base}#{count}"))
73        };
74        *count += 1;
75        id
76    }
77}

79#[derive(Debug, Default)]
80struct NodeTable {
81    ids: IdGen,
82    nodes: HashMap<NodeName, NodeInfo>,
83    idx_to_name: HashMap<NodeIndex, NodeName>,
84}
85
86impl NodeTable {
87    fn name_of(&self, nidx: NodeIndex) -> &NodeName {
88        &self.idx_to_name[&nidx]
89    }
90
91    fn info(&self, nidx: NodeIndex) -> &NodeInfo {
92        &self.nodes[self.name_of(nidx)]
93    }
94
95    fn space_of(&self, nidx: NodeIndex) -> &IterationSpace {
96        self.info(nidx).space()
97    }
98
99    fn insert(&mut self, nidx: NodeIndex, label: Option<&str>, prefix: &str, info: NodeInfo) {
100        let id = self.ids.next(label, prefix);
101        self.nodes.insert(id.clone(), info);
102        self.idx_to_name.insert(nidx, id);
103    }
104
105    fn add_sink(&mut self, space: IterationSpace) -> NodeName {
106        let id = self.ids.next(None, "sink");
107        self.nodes.insert(id.clone(), NodeInfo::Sink { space });
108        id
109    }
110}

112/// Extend the inherited stack with one edge's `Translated` entries.
113/// `Composed` entries are not threaded — they're consumed locally by
114/// [`extend_iter_space_for_fold`] at the edge they appear on. Entries from
115/// one edge are sorted by axis for deterministic plans (they commute —
116/// distinct axes).
117///
118/// A `Zipped` entry whose leader axis already appears in the inherited
119/// stack would need its leader coord rewritten by the downstream overrides
120/// — not expressible in the per-axis stack. Unreachable today (slicing a
121/// vindex-produced axis is rejected at lowering), so reject loudly.
122fn extend_overrides(
123    inherited: &[(AxisName, TranslatedAccess)],
124    edge_access: &HashMap<AxisName, AxisAccess>,
125) -> BrixResult<TranslatedAccessStack> {
126    let appended = translated_entries(edge_access);
127
128    if let Some((axis, _)) = appended.iter().find(|(axis, entry)| {
129        matches!(entry, TranslatedAccess::Zipped(_)) && inherited.iter().any(|(a, _)| a == axis)
130    }) {
131        return Err(BrixError::Unsupported(format!(
132            "planner: downstream override on Zipped leader axis {axis:?} cannot be \
133             composed with the zipped gather"
134        )));
135    }
136
137    Ok(inherited.iter().cloned().chain(appended).collect())
138}
139
140/// An edge's `Translated` entries, sorted by axis for deterministic plans
141/// (entries on one edge commute — distinct axes). `Composed` entries are
142/// dropped — they're consumed by [`extend_iter_space_for_fold`].
143fn translated_entries(edge_access: &HashMap<AxisName, AxisAccess>) -> TranslatedAccessStack {
144    let mut entries: TranslatedAccessStack = edge_access
145        .iter()
146        .filter_map(|(axis, entry)| match entry {
147            AxisAccess::Translated(e) => Some((axis.clone(), e.clone())),
148            AxisAccess::Composed(_) => None,
149        })
150        .collect();
151    entries.sort_by(|(a, _), (b, _)| a.cmp(b));
152    entries
153}

159fn build_nodes(graph: &ExprGraph, root: NodeIndex) -> (NodeTable, Vec<NodeIndex>) {
160    let reversed = Reversed(graph);
161    let order: Vec<NodeIndex> = DfsPostOrder::new(reversed, root).iter(reversed).collect();
162
163    let mut table = NodeTable::default();
164    for &nidx in &order {
165        let node = &graph[nidx];
166        let (info, prefix) = node_info(graph, nidx);
167        table.insert(nidx, node.label.as_deref(), prefix, info);
168    }
169    (table, order)
170}
171
172fn node_info(graph: &ExprGraph, nidx: NodeIndex) -> (NodeInfo, &'static str) {
173    let node = &graph[nidx];
174    let space = node.space.clone();
175    match &node.op {
176        Op::Source(source) => (NodeInfo::Source { source: Arc::clone(source), space }, "source"),
177        Op::Blockwise(kernel) => {
178            (NodeInfo::Blockwise { kernel: Arc::clone(kernel), space }, kernel.name())
179        }
180        Op::Reduce(accumulator) => (
181            NodeInfo::Reduce { accumulator: Arc::clone(accumulator), output_space: space },
182            "reduce",
183        ),
184    }
185}

187/// Return a node's incoming edges as `(source, access)`, sorted by port.
188fn port_sorted_inputs(
189    graph: &ExprGraph,
190    nidx: NodeIndex,
191) -> Vec<(NodeIndex, &HashMap<AxisName, AxisAccess>)> {
192    let mut by_port: Vec<(usize, NodeIndex, &HashMap<AxisName, AxisAccess>)> = graph
193        .edges_directed(nidx, Incoming)
194        .map(|e| (e.weight().port, e.source(), &e.weight().access))
195        .collect();
196    by_port.sort_by_key(|(p, ..)| *p);
197    by_port.into_iter().map(|(_, src, access)| (src, access)).collect()
198}

204fn emit_schedule(
205    graph: &ExprGraph,
206    _order: &[NodeIndex],
207    table: &NodeTable,
208    root: NodeIndex,
209    sink_node_id: &NodeName,
210) -> BrixResult<Schedule> {
211    let root_grid = table.space_of(root);
212    let outer_iter = &root_grid.layout;
213
214    // The root has no downstream edges — its emission starts with an
215    // empty override stack.
216    let mut body = plan_subtree(root, outer_iter, graph, table, &[])?;
217    body.push(Schedule::WriteSink {
218        node: table.name_of(root).clone(),
219        sink: sink_node_id.clone(),
220        access: AccessRelation::new(AffineAccess::for_edge(root_grid, outer_iter, &[]), &[]),
221    });
222
223    // Wrap the body in iteration steps.
224    // Dense axes get one `Loop` each (in reverse so the first listed axis is outermost).
225    // If the root grid is Sparse, the body is additionally wrapped in a single
226    // `Zip` covering all sparse axes (placed innermost — so dense
227    // outer loops still pin the dense axes before zip enumerates the
228    // present sparse coords).
229    let body = match root_grid.occupancy() {
230        Some(occupancy) => vec![Schedule::Zip {
231            axes: occupancy.sparse_axes.clone(),
232            coords: occupancy.as_coo_coords(),
233            body,
234        }],
235        None => body,
236    };
237    let dense_axes: Vec<AxisName> = root_grid.dense_axes().cloned().collect();
238    let body = dense_axes
239        .into_iter()
240        .rev()
241        // TODO: is this mode_index: 0 always right?
242        .fold(body, |acc, axis| vec![Schedule::Loop { axis, body: acc, mode_index: 0 }]);
243
244    // Note that the Domain describes fullest possible extent of the IterationSpace axes
245    // For sparse axes of IterationSpace, the occupancy information is stored on Schedule::Zip.
246    // FIXME: we probably want to lift that sparsity information up to Domain nodes
247    //        so that we can partition for distributed compute.
248    Ok(Schedule::Domain {
249        iter_space: IterationSpace::from_layout(outer_iter.clone()),
250        body: Box::new(Schedule::Sequence(body)),
251    })
252}

254/// `overrides` is this emission's [`TranslatedAccessStack`]: the
255/// `Translated` entries of the edges walked from the root down to `node`,
256/// in descent order. It parameterizes the whole emitted subtree — under fan-out each
257/// consumer re-emits the producer with its own stack.
258fn plan_subtree(
259    node: NodeIndex,
260    iter_space: &NamedLayout,
261    graph: &ExprGraph,
262    table: &NodeTable,
263    overrides: &[(AxisName, TranslatedAccess)],
264) -> BrixResult<Vec<Schedule>> {
265    if graph[node].op.is_stateful() {
266        plan_stateful(node, iter_space, graph, table, overrides)
267    } else {
268        plan_stateless(node, iter_space, graph, table, overrides)
269    }
270}
271
272// Flatten-order invariant: each input subtree's emission precedes this
273// node's apply, and under fan-out each consumer's emission of a shared
274// producer precedes that consumer's read with no intervening re-emission
275// (read-follows-write on the shared workspace slot). A reordering pass
276// (e.g. hoisting common emissions) would break this silently — guarded
277// empirically by tests/access_threading.rs.
278fn plan_stateless(
279    node: NodeIndex,
280    iter_space: &NamedLayout,
281    graph: &ExprGraph,
282    table: &NodeTable,
283    overrides: &[(AxisName, TranslatedAccess)],
284) -> BrixResult<Vec<Schedule>> {
285    let mut steps: Vec<Schedule> = port_sorted_inputs(graph, node)
286        .into_iter()
287        .map(|(input, input_edge)| {
288            let input_overrides = extend_overrides(overrides, input_edge)?;
289            plan_subtree(input, iter_space, graph, table, &input_overrides)
290        })
291        .collect::<BrixResult<Vec<_>>>()?
292        .into_iter()
293        .flatten()
294        .collect();
295    steps.push(node_to_apply(graph, node, table, iter_space, &[], overrides));
296    Ok(steps)
297}

299// Process one "stateful" node --- these have Init/Apply/Finalize steps
300fn plan_stateful(
301    node: NodeIndex,
302    iter_space: &NamedLayout,
303    graph: &ExprGraph,
304    table: &NodeTable,
305    overrides: &[(AxisName, TranslatedAccess)],
306) -> BrixResult<Vec<Schedule>> {
307    let mut incoming = graph.edges_directed(node, Incoming);
308    let edge = incoming.next().expect("stateful node must have at least one input");
309    debug_assert!(incoming.next().is_none(), "stateful node must have exactly one input");
310    let reduce_edge = edge.weight().access.clone();
311    let input = edge.source();
312    let input_grid = table.space_of(input);
313
314    // Worked example:
315    //   — coarsen axis "i" by 2 on input with space: [8] → output [4]:
316    //   - `extend_iter_space_for_fold` divides "i" into [4, 2]:
317    //     outer block index `c` for mode ("i", 0): the output's surviving coord,
318    //     and inner tile index `j` for mode ("i", 1), the fold-summed coord).
319    //     Here `fold_modes = [("i", 1)]`.
320    let (local_iter, fold_modes, limits) =
321        extend_iter_space_for_fold(iter_space, &reduce_edge, input_grid)?;
322
323    // A downstream override on a fold-divided axis is in *output*-chunk
324    // units, but inside the fold extension that axis evaluates in
325    // *input*-chunk units — applying it raw would shift by the wrong
326    // amount (off by the fold factor). Rewriting it (`Offset(k)` →
327    // `Offset(k × window)`) is future work; fail loudly for now.
328    if let Some((axis, _)) =
329        overrides.iter().find(|(axis, _)| fold_modes.iter().any(|(fold_axis, _)| fold_axis == axis))
330    {
331        return Err(BrixError::Unsupported(format!(
332            "planner: override on fold axis {axis:?} — slicing/gathering a reduced or \
333             coarsened axis below the reduce is not supported yet"
334        )));
335    }
336
337    // The fold extension changes the iteration space, not the override
338    // units on non-fold axes — the inherited stack threads through, plus
339    // this edge's own `Translated` entries.
340    let input_overrides = extend_overrides(overrides, &reduce_edge)?;
341
342    // The Reduce's own-output access must be invariant under each of the
343    // fold loops we're about to wrap `inner` in.
344    let mut inner = plan_subtree(input, &local_iter, graph, table, &input_overrides)?;
345    // Worked example:
346    //   - `node_to_apply` builds the Reduce's apply view with `("i", 1)`
347    //     having zero stride, so the accumulator address depends on `c` only —
348    //     unchanged as `j` sweeps.
349    inner.push(node_to_apply(graph, node, table, &local_iter, &fold_modes, overrides));
350    // Worked example:
351    //   - The `fold` below wraps `inner` in a `Schedule::Loop { axis: "i",
352    //     mode_index: 1 }` so the executor advances `j ∈ 0..2` per accumulator
353    //     output coord `c`.
354    let looped = fold_modes.into_iter().rev().fold(inner, |acc, (axis, mode_index)| {
355        vec![Schedule::Loop { axis, mode_index, body: acc }]
356    });
357
358    // Init runs OUTSIDE the fold loops, so its access must use the parent
359    // `iter_space` (without the fold axes). The accumulator is sized by the
360    // output coord, which only depends on the surviving axes.
361    let init = Schedule::Init {
362        node: table.name_of(node).clone(),
363        access: AccessRelation::new(
364            AffineAccess::for_edge(table.space_of(node), iter_space, &[]),
365            overrides,
366        ),
367    };
368    let finalize = Schedule::Finalize { node: table.name_of(node).clone() };
369    let domain_body: Vec<Schedule> = once(init).chain(looped).chain(once(finalize)).collect();
370
371    Ok(vec![Schedule::Domain {
372        iter_space: IterationSpace::from_layout(local_iter).with_limits(limits),
373        body: Box::new(Schedule::Sequence(domain_body)),
374    }])
375}

377// Given an outer iteration space `outer`, extend it with the fold loops a
378// stateful node (just Reduce today) introduces, and return the fold axes as
379// `(name, mode_index)` (in `input_grid` order, so the `Loop` nest is
380// deterministic).
381//
382// Applies only the `Composed` entries off the edge's per-axis access map;
383// `Translated` entries don't influence the loop nest (they're per-coord
384// overrides handled at runtime).
385//
386// FIXME: The reasoning behind this function is quite fold-specific which need not be necessary.
387// TODO: should we instead just return an "update layout" for concatenation/composition?
388#[allow(clippy::type_complexity)]
389fn extend_iter_space_for_fold(
390    outer: &NamedLayout,
391    access: &HashMap<AxisName, AxisAccess>,
392    input_grid: &IterationSpace,
393) -> BrixResult<(NamedLayout, Vec<(AxisName, usize)>, HashMap<AxisName, usize>)> {
394    let input_counts: HashMap<_, _> = zip(input_grid.names(), input_grid.max_extents()).collect();
395
396    input_grid.names().iter().try_fold(
397        (outer.clone(), Vec::new(), HashMap::new()),
398        |(iter_space, mut fold_axes, mut limits), axis| {
399            let frame = match access.get(axis) {
400                // Not framed on this axis leave the iteration space unchanged.
401                None | Some(AxisAccess::Translated(_)) => {
402                    return Ok((iter_space, fold_axes, limits));
403                }
404                Some(AxisAccess::Composed(f)) => f,
405            };
406            let iter_space = match frame {
407                // Reduced away, absent from `outer`: append it flat; the whole
408                // depth-1 axis folds at mode_index 0.
409                Frame::WholeAxis { keepdims: false } => {
410                    fold_axes.push((axis.clone(), 0));
411                    iter_space.append(axis, input_counts[axis])?
412                }
413                // Whole axis surviving as size-1: divide by `n_in` into
414                // (1, n_in); the tile sub-layout folds at mode_index 1. Exact
415                // divide (n_in / n_in), so no overshoot — no predication limit.
416                Frame::WholeAxis { keepdims: true } => {
417                    let n_in = input_counts[axis];
418                    fold_axes.push((axis.clone(), 1));
419                    iter_space.update(axis, n_in)?.logical_divide(axis, n_in)?
420                }
421                // Surviving framed axis: reset to the input size, then divide
422                // into (n_out, size):(size,1); the tile folds at mode_index 1.
423                // A non-divisible `n_in` makes the divide round up,
424                // record `n_in` as a predication limit.
425                Frame::Window { size, stride } => {
426                    if size != stride {
427                        return Err(BrixError::Unsupported(
428                            "extend_iter_space: sliding windows (size != stride) not supported yet"
429                                .into(),
430                        ));
431                    }
432                    let n_in = input_counts[axis];
433                    fold_axes.push((axis.clone(), 1));
434                    if n_in % size != 0 {
435                        limits.insert(axis.clone(), n_in);
436                    }
437                    iter_space.update(axis, n_in)?.logical_divide(axis, *size)?
438                }
439                other => {
440                    return Err(BrixError::Unsupported(format!(
441                        "extend_iter_space: cannot build a fold loop for {other:?}"
442                    )));
443                }
444            };
445            Ok((iter_space, fold_axes, limits))
446        },
447    )
448}

454fn node_to_apply(
455    graph: &ExprGraph,
456    nidx: NodeIndex,
457    table: &NodeTable,
458    iter_view: &NamedLayout,
459    apply_invariant_under: &[(AxisName, usize)],
460    overrides: &[(AxisName, TranslatedAccess)],
461) -> Schedule {
462    let inputs: Vec<NodeName> = match &graph[nidx].op {
463        Op::Source(_) => vec![],
464        _ => port_sorted_inputs(graph, nidx)
465            .into_iter()
466            .map(|(src, _)| table.name_of(src).clone())
467            .collect(),
468    };
469    Schedule::Apply {
470        node: table.name_of(nidx).clone(),
471        inputs,
472        access: AccessRelation::new(
473            AffineAccess::for_edge(table.space_of(nidx), iter_view, apply_invariant_under),
474            overrides,
475        ),
476    }
477}
