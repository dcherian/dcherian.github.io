16/// Shared positional primitive — a contiguous-with-stride 1-D range.
17///
18/// Used by the array-level indexer ([`array::indexer::Indexer::Slicer`])
19/// and the mid-IR's lowering targets (e.g. [`array::SliceKernel`]).
20/// Pure value type; no dtype, no shape, no semantics beyond `start..end`
21/// with `step`.
22#[derive(Debug, Clone, Copy, PartialEq, Eq)]
23pub struct Slice {
24    pub start: usize,
25    pub step: usize,
26    pub end: usize,
27}
28
29impl Slice {
30    pub fn new(start: usize, end: usize, step: usize) -> Self {
31        Slice { start, end, step }
32    }
33
34    /// Convenience: `start..end` with `step = 1`.
35    pub fn contiguous(start: usize, end: usize) -> Self {
36        Slice { start, end, step: 1 }
37    }
38
39    /// Output extent when this slice is applied to an axis of size `extent`.
40    /// Returns [`BrixError::InvalidSlice`] (tagged with `axis`) when out of
41    /// bounds.
42    pub fn output_extent(&self, axis: &brix::AxisName, extent: usize) -> brix::BrixResult<usize> {
43        if self.start > self.end || self.end > extent {
44            return Err(brix::BrixError::InvalidSlice {
45                axis: axis.to_string(),
46                start: self.start,
47                end: self.end,
48                step: self.step,
49                extent,
50            });
51        }
52        Ok((self.end - self.start).div_ceil(self.step))
53    }
54}
