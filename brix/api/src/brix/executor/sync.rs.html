15/// Sequential executor — single-threaded interpreter of the schedule tree.
16#[derive(Debug, Default, Clone, Copy)]
17pub struct SyncExecutor;
18
19impl Executor for SyncExecutor {
20    fn execute(&self, plan: &PhysicalPlan, sink: &dyn Sink) -> BrixResult<()> {
21        execute(plan, 0, sink)
22    }
23}

25/// Execute a physical plan sequentially, writing results to a sink.
26///
27/// The plan is expected to contain exactly one `NodeInfo::Sink` node;
28/// the provided `sink` is bound to that node's id. Multi-sink plans
29/// (Dataset outputs) are not yet supported.
30pub fn execute(plan: &PhysicalPlan, _memory_budget: usize, sink: &dyn Sink) -> BrixResult<()> {
31    let mut sink_nodes =
32        plan.nodes.iter().filter(|(_, info)| matches!(info, NodeInfo::Sink { .. }));
33    let (sink_node_id, _) = sink_nodes.next().expect("invariant: plan has at least one Sink node");
34    if sink_nodes.next().is_some() {
35        return Err(BrixError::Unsupported(
36            "executor::sync currently supports only single-sink plans".into(),
37        ));
38    }
39
40    let mut sinks: HashMap<NodeName, &dyn Sink> = HashMap::new();
41    sinks.insert(sink_node_id.clone(), sink);
42
43    let mut state = ExecState {
44        indices: NamedCoord::default(),
45        workspace: HashMap::new(),
46        accumulators: HashMap::new(),
47    };
48    run(&plan.schedule, &plan.nodes, &mut state, &sinks, None)
49}
50
51struct ExecState {
52    /// Current iteration coordinate, keyed `(axis, mode_index)` → component.
53    indices: NamedCoord,
54    workspace: HashMap<NodeName, Brick>,
55    accumulators: HashMap<NodeName, Box<dyn State>>,
56}
57
58fn run(
59    schedule: &Schedule,
60    nodes: &HashMap<NodeName, NodeInfo>,
61    state: &mut ExecState,
62    sinks: &HashMap<NodeName, &dyn Sink>,
63    iter_space: Option<&IterationSpace>,
64) -> BrixResult<()> {
65    match schedule {
66        Schedule::Domain { iter_space: s, body } => {
67            run(body, nodes, state, sinks, Some(s))?;
68        }
69
70        Schedule::Loop { axis, body, mode_index } => {
71            // FIXME: can we dump this expect; why is it possibly None
72            let space = iter_space.expect("invariant: Loop not inside Domain");
73            let bound = space.extent(axis, *mode_index, &state.indices);
74            for idx in 0..bound {
75                state.indices.insert(axis, *mode_index, idx);
76                for step in body {
77                    run(step, nodes, state, sinks, iter_space)?;
78                }
79            }
80        }
81
82        Schedule::Zip { axes, coords, body } => {
83            for coord in coords.iter() {
84                // Zip axes are flat (no fold split), so they live at mode_index 0.
85                for (name, &v) in axes.iter().zip(coord.iter()) {
86                    state.indices.insert(name, 0, v as usize);
87                }
88                for step in body {
89                    run(step, nodes, state, sinks, iter_space)?;
90                }
91            }
92        }
93
94        Schedule::Sequence(body) => {
95            for step in body {
96                run(step, nodes, state, sinks, iter_space)?;
97            }
98        }
99
100        Schedule::Init { node, access } => {
101            let info = &nodes[node];
102            let coord = access.resolve_coord(info.space(), &state.indices)?;
103            match info {
104                NodeInfo::Reduce { accumulator, .. } => {
105                    state.accumulators.insert(node.clone(), accumulator.init(&coord)?);
106                }
107                _ => unreachable!("invariant: Init on non-stateful node"),
108            }
109        }
110
111        Schedule::Apply { node, inputs, access } => {
112            let info = &nodes[node];
113            let coord = access.resolve_coord(info.space(), &state.indices)?;
114
115            match info {
116                NodeInfo::Source { source, .. } => {
117                    let data = source.get_brick_payload(&coord)?;
118                    state.workspace.insert(node.clone(), Brick { coord, data });
119                }
120
121                NodeInfo::Blockwise { kernel, .. } => {
122                    let input_bricks: Vec<Brick> =
123                        inputs.iter().map(|name| state.workspace[name].clone()).collect();
124                    let input_refs: Vec<&Brick> = input_bricks.iter().collect();
125                    let data = kernel.apply(&input_refs, &coord)?;
126                    state.workspace.insert(node.clone(), Brick { coord, data });
127                }
128
129                NodeInfo::Reduce { accumulator, .. } => {
130                    let [input] = inputs.as_slice() else {
131                        unreachable!("invariant: Reduce Apply has exactly one input")
132                    };
133                    let input_brick = &state.workspace[input];
134                    let acc_state = state
135                        .accumulators
136                        .get_mut(node)
137                        .expect("invariant: Init precedes Apply for stateful node");
138                    accumulator.fold_left(acc_state.as_mut(), input_brick, &coord)?;
139                }
140
141                NodeInfo::Sink { .. } => {
142                    unreachable!("invariant: Apply must not target a Sink node")
143                }
144            }
145        }
146
147        Schedule::Finalize { node } => match &nodes[node] {
148            NodeInfo::Reduce { accumulator, output_space, .. } => {
149                let acc_state = state
150                    .accumulators
151                    .remove(node)
152                    .expect("invariant: Init precedes Finalize for stateful node");
153                // Derive output coord from current loop indices by name-
154                // matching the output space's axes. Finalize sits outside the
155                // reduced axes' loops, so the surviving indices fully pin
156                // the output coord.
157                let positional: Vec<usize> = output_space
158                    .names()
159                    .iter()
160                    // Surviving axis → its surviving sub-layout, mode_index 0.
161                    // Present by invariant: Finalize sits inside the surviving
162                    // axes' loops, which set these indices.
163                    .map(|n| {
164                        state
165                            .indices
166                            .get_leading(n)
167                            .expect("invariant: surviving output axis indexed by enclosing loop")
168                    })
169                    .collect();
170                let coord = BrickCoordinate::from_named(output_space.names(), &positional);
171                let brick = accumulator.finalize(acc_state, &coord)?;
172                state.workspace.insert(node.clone(), brick);
173            }
174            _ => unreachable!("invariant: Finalize on non-stateful node"),
175        },
176
177        Schedule::InitSink { sink } => {
178            sinks[sink].init()?;
179        }
180
181        Schedule::WriteSink { node, sink, access: _ } => {
182            let brick =
183                state.workspace.remove(node).expect("invariant: no fan-out in current planner");
184            sinks[sink].write_brick(brick)?;
185        }
186
187        Schedule::FinalizeSink { sink } => {
188            sinks[sink].finalize()?;
189        }
190    }
191    Ok(())
192}
