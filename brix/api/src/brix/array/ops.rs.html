1//! Array-level arithmetic operations.
2//!
3//! `BinaryOp` and `UnaryOp` are the closed enums the array-level optimizer
4//! pattern-matches on for algebraic simplification (e.g. `x + 0 = x`).
5//! The mid-IR (`Expr`) only sees them indirectly through
6//! [`crate::array::kernels::BinaryKernel`] / [`crate::array::kernels::UnaryKernel`].
7
8use std::{fmt, sync::Arc};
9
10use crate::{
11    array::{
12        brick::ArrayPayload,
13        dtype::{Binary, DType, Unary, Value, dispatch_dtype},
14    },
15    brix::{BrickPayload, BrixResult},
16};
17use ArrayPayload::*;

19// ---------------------------------------------------------------------------
20// Unary
21// ---------------------------------------------------------------------------
22
23#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
24pub enum UnaryOp {
25    Neg,
26    Abs,
27}
28
29impl UnaryOp {
30    pub fn apply(&self, a: &dyn BrickPayload) -> BrixResult<Arc<dyn BrickPayload>> {
31        let p = ArrayPayload::from_payload(a);
32        dispatch_dtype!(p.dtype(), T => Ok(Arc::new(self.apply_typed::<T>(p))))
33    }
34
35    /// Monomorphic unary body — one implementation for every signed dtype.
36    /// `Unary` supplies `Neg` (`-x`) and `abs`.
37    fn apply_typed<T: Unary>(&self, p: &ArrayPayload) -> ArrayPayload {
38        let f: fn(T) -> T = match self {
39            UnaryOp::Neg => |x| -x,
40            UnaryOp::Abs => |x| x.abs(),
41        };
42        match p {
43            Fill(fill) => {
44                let v = fill.value.get::<T>().expect("payload dtype matches dispatch");
45                ArrayPayload::fill(T::from_scalar(f(v)), fill.shape.clone())
46            }
47            Dense(d) => {
48                let view = d.data.view().as_typed::<T>().expect("payload dtype matches dispatch");
49                ArrayPayload::dense(T::from_array(view.mapv(f)))
50            }
51        }
52    }
53
54    pub fn update_fill(&self, v: Value) -> Value {
55        match (self, v) {
56            (UnaryOp::Neg, Value::F64(x)) => Value::F64(-x),
57            (UnaryOp::Abs, Value::F64(x)) => Value::F64(x.abs()),
58            (UnaryOp::Neg, Value::I64(x)) => Value::I64(-x),
59            (UnaryOp::Abs, Value::I64(x)) => Value::I64(x.abs()),
60        }
61    }
62}
63
64impl fmt::Display for UnaryOp {
65    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
66        match self {
67            UnaryOp::Neg => write!(f, "-"),
68            UnaryOp::Abs => write!(f, "abs"),
69        }
70    }
71}

77/// Elementwise binary operation.
78///
79/// Enum by choice: the optimizer needs to pattern-match on specific ops for
80/// egglog algebraic rules (e.g., `x + 0 = x`), and the set of binary ops is
81/// small and fixed. Contrast with [`crate::expr::Accumulator`], which is a
82/// trait because reductions are open-ended.
83#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
84pub enum BinaryOp {
85    Add,
86    Mul,
87    Sub,
88}
89
90impl BinaryOp {
91    /// Apply the operation to two payloads.
92    ///
93    /// I anticipate two levels of short-circuit to be useful:
94    /// - **Plan time (optimizer):** rewrites `Blockwise(Binary(Mul), [Fill(0), x]) →
95    ///   Fill(0)` — eliminates the expression node entirely.
96    /// - **Runtime (here):** when a source emits Fill bricks at read time,
97    ///   `apply` can skip computation using identity/absorber properties.
98    pub fn apply(
99        &self,
100        a: &Arc<dyn BrickPayload>,
101        b: &Arc<dyn BrickPayload>,
102    ) -> BrixResult<Arc<dyn BrickPayload>> {
103        let pa = ArrayPayload::from_payload(a.as_ref());
104        let pb = ArrayPayload::from_payload(b.as_ref());
105
106        // Fill identity/absorber short-circuits return the surviving operand's
107        // `Arc` untouched — no buffer clone on the hot path.
108        match self.short_circuit(pa, pb) {
109            Some(true) => Ok(a.clone()),
110            Some(false) => Ok(b.clone()),
111            // Both inputs share a dtype at runtime (plan-time promotion inserts
112            // implicit casts), so dispatch on the left and assume right matches.
113            None => dispatch_dtype!(pa.dtype(), T => Ok(Arc::new(self.apply_typed::<T>(pa, pb)))),
114        }
115    }
116
117    /// Payload-level entry point: the same elementwise op as `apply`, but on
118    /// owned [`ArrayPayload`]s with no `Arc`/`dyn` wrapper — used by accumulator
119    /// merge. The Fill identity/absorber short-circuit clones the surviving
120    /// operand (unlike `apply`, which can hand back a cheap `Arc`).
121    pub fn apply_payloads(&self, a: &ArrayPayload, b: &ArrayPayload) -> ArrayPayload {
122        match self.short_circuit(a, b) {
123            Some(true) => a.clone(),
124            Some(false) => b.clone(),
125            None => dispatch_dtype!(a.dtype(), T => self.apply_typed::<T>(a, b)),
126        }
127    }
128
129    /// Resolve a Fill identity/absorber short-circuit, if one applies. `Some(true)`
130    /// → the left operand is the result; `Some(false)` → the right; `None` → must
131    /// compute. The fill value is compared exactly against the identity/absorber
132    /// built at *its own* dtype — no lossy `f64` round-trip — so the check stays
133    /// correct as non-numeric dtypes (e.g. a future `Bool` with `And`/`Or`) land.
134    fn short_circuit(&self, a: &ArrayPayload, b: &ArrayPayload) -> Option<bool> {
135        if let Fill(fa) = a {
136            if self.absorber(fa.value.dtype()) == Some(fa.value) {
137                return Some(true);
138            }
139            if self.identity(fa.value.dtype()) == Some(fa.value) {
140                return Some(false);
141            }
142        }
143        if let Fill(fb) = b {
144            if self.is_commutative() && self.absorber(fb.value.dtype()) == Some(fb.value) {
145                return Some(false);
146            }
147            if self.identity(fb.value.dtype()) == Some(fb.value) {
148                return Some(true);
149            }
150        }
151        None
152    }
153
154    /// Monomorphic elementwise body — one implementation per primitive dtype.
155    fn apply_typed<T: Binary>(&self, pa: &ArrayPayload, pb: &ArrayPayload) -> ArrayPayload {
156        let op: fn(T, T) -> T = match self {
157            BinaryOp::Add => |x, y| x + y,
158            BinaryOp::Mul => |x, y| x * y,
159            BinaryOp::Sub => |x, y| x - y,
160        };
161        let scalar = |p: &ArrayPayload| match p {
162            Fill(f) => f.value.get::<T>().expect("payload dtype matches dispatch"),
163            _ => unreachable!(),
164        };
165        let m = "payload dtype matches dispatch";
166        match (pa, pb) {
167            (Fill(x), Fill(_)) => {
168                ArrayPayload::fill(T::from_scalar(op(scalar(pa), scalar(pb))), x.shape.clone())
169            }
170            (Fill(_), Dense(_)) => {
171                let f = scalar(pa);
172                ArrayPayload::dense(T::from_array(
173                    pb.dense_view::<T>().expect(m).mapv(|y| op(f, y)),
174                ))
175            }
176            (Dense(_), Fill(_)) => {
177                let f = scalar(pb);
178                ArrayPayload::dense(T::from_array(
179                    pa.dense_view::<T>().expect(m).mapv(|x| op(x, f)),
180                ))
181            }
182            (Dense(_), Dense(_)) => {
183                let (va, vb) = (pa.dense_view::<T>().expect(m), pb.dense_view::<T>().expect(m));
184                ArrayPayload::dense(T::from_array(
185                    ndarray::Zip::from(&va).and(&vb).map_collect(|&x, &y| op(x, y)),
186                ))
187            }
188        }
189    }
190
191    /// The op's two-sided identity element at `dtype`, if it has one:
192    /// `apply(v, x) == x` and `apply(x, v) == x`.
193    /// Enables: `Blockwise(op, [x, Fill(identity)]) → x`.
194    ///
195    /// `Sub` is `None`: it has only a *right* identity (`x - 0 == x`, but
196    /// `0 - x == -x`), which this two-sided contract can't represent —
197    /// modelling left/right identity separately is the follow-up.
198    pub fn identity(&self, dtype: DType) -> Option<Value> {
199        match self {
200            BinaryOp::Add => Some(Value::zero(dtype)),
201            BinaryOp::Sub => None,
202            BinaryOp::Mul => Some(Value::one(dtype)),
203        }
204    }
205
206    /// The op's absorber (annihilator) at `dtype`, if any: `apply(v, x) == v`
207    /// (for commutative ops).
208    /// Enables: `Blockwise(op, [Fill(absorber), x]) → Fill(absorber)`.
209    pub fn absorber(&self, dtype: DType) -> Option<Value> {
210        match self {
211            BinaryOp::Mul => Some(Value::zero(dtype)),
212            BinaryOp::Add | BinaryOp::Sub => None,
213        }
214    }
215
216    pub fn is_commutative(&self) -> bool {
217        match self {
218            BinaryOp::Add | BinaryOp::Mul => true,
219            BinaryOp::Sub => false,
220        }
221    }
222}
223
224impl fmt::Display for BinaryOp {
225    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
226        match self {
227            BinaryOp::Add => write!(f, "+"),
228            BinaryOp::Mul => write!(f, "*"),
229            BinaryOp::Sub => write!(f, "-"),
230        }
231    }
232}

234#[cfg(test)]
235mod tests {
236    use super::*;
237
238    #[test]
239    fn binary_op_properties() {
240        assert_eq!(BinaryOp::Add.identity(DType::F64), Some(Value::F64(0.0)));
241        assert_eq!(BinaryOp::Add.identity(DType::I64), Some(Value::I64(0)));
242        assert_eq!(BinaryOp::Mul.identity(DType::I64), Some(Value::I64(1)));
243        assert_eq!(BinaryOp::Mul.absorber(DType::I64), Some(Value::I64(0)));
244        assert_eq!(BinaryOp::Sub.identity(DType::F64), None);
245        assert!(BinaryOp::Add.is_commutative());
246        assert!(!BinaryOp::Sub.is_commutative());
247    }
248}
