1//! Lower the array-level IR (`ArrayExpr`) to the mid-IR (`Expr`).
2//!
3//! Walks `ArrayExpr` post-order and emits the corresponding `Expr`.
4//!
5//! To do so, we need to also make concrete `ChunkGrid` choices for execution.
6//! That information is embedded in the [`crate::brix::Source`], [`crate::brix::Kernel`],
7//! [`crate::brix::Accumulator`] passed on to `brix`.
8//!
9//! Note that the lowering here is a mechanical translation from [`ArrayExpr`] to [`Expr`].
10//! We simply add concrete `ChunkGrid` information to the outputs of each [`Expr`]
11//! The planner in brix/planner.rs is responsible for wiring this up
12//! and accounting for broadcasting semantics etc.
13//!
14//! We have `expect` in this file, because we expect those invariants
15//! to be enforced higher up the call chain.
16

45/// A lowered array expression: the mid-IR DAG plus the element-sized
46/// chunk grid for the root node. The root grid is exposed because the
47/// sink isn't part of the lowered `Expr` DAG — it's passed to
48/// `execute()` separately — so the chosen sink has no other way to learn the
49/// grid the planner committed to.
50/// FIXME: This decision to keep Sink separate should be revisited.
51///        This awkwardness is also a consequence of keeping ChunkGrid off the Expr layer
52///        But we need it when lowering; see how many times we use `.root_grid` below.
53///
54/// `grids` caches the per-node element-sized `ChunkGrid` for every node
55/// in the lowered DAG, keyed by [`ExprId`]. Internal consumers
56/// (render in particular) can look this up to recover the rich
57/// `ChunkGrid` (with `AxisChunks` encoding and occupancy) without
58/// re-running the lowering math. Downstream of lowering, brix only sees
59/// the counts-only [`crate::brix::IterationSpace`] on each `Expr` — the rich grid
60/// lives only here and on the concrete `Source`/`Kernel`/`Accumulator`
61/// impls baked into each `Op`.
62// ANCHOR: lowered
63#[derive(Clone)]
64pub struct Lowered {
65    pub expr: Arc<Expr>,
66    // FIXME: This `root_grid` is a wart that exists only because the final sink
67    // is not in the `expr`. It is passed separately, and so we need to carry the
68    // explicit array-level chunk grid for that final Sink.
69    pub root_grid: ChunkGrid,
70    // This sidecar Map is simply to enable nice visualization/explainers.
71    // It is *not* used for planning or execution at the brix layer.
72    pub grids: HashMap<ExprId, ChunkGrid>,
73}
74// ANCHOR_END: lowered

76// ANCHOR: lower_code
77/// Lower an [`ArrayLike`] to a mid-IR `Arc<Expr>` DAG.
78/// The per-axis chunk target is derived from root shape + dtype + memory budget;
79/// child targets inherit it.
80/// FIXME: `memory_budget` needs to become a broader "ResourceConstraints" or similar.
81pub fn lower(arr: &impl ArrayLike, memory_budget: usize) -> ArrayResult<Lowered> {
82    let expr = arr.expr();
83    ////////////////////////////////////////////////////////////
84    // FIXME: this is extremely preliminary.
85    // Derive the chunk target from the *widest* shape in the DAG, not
86    // the root: a reduce drops axes from the root, but the inputs
87    // still carry those axes and we want them chunked under the same
88    // budget. Per-position max across all nodes gives chunk_target the
89    // length of the deepest axis stack we'll see.
90    let chunk_target = choose_chunk_target(expr, memory_budget);
91    ////////////////////////////////////////////////////////////
92    let mut memo: HashMap<*const ArrayExpr, (Arc<Expr>, ChunkGrid)> = HashMap::new();
93    let mut grids: HashMap<ExprId, ChunkGrid> = HashMap::new();
94    let (root_expr, root_grid) = lower_inner(expr, &chunk_target, &mut memo, &mut grids)?;
95    Ok(Lowered { expr: root_expr, root_grid, grids })
96}
97// ANCHOR_END: lower_code

99/// Per-position max across the entire ArrayExpr DAG. Used by [`lower`]
100/// to derive a chunk_target that covers every axis the lowering will
101/// touch — including axes dropped by reductions in the root. Axis order
102/// is positional: callers assume that the axis at position `d` in any
103/// node corresponds to the same logical axis (true for current
104/// workloads, which don't transpose mid-DAG).
105pub(crate) fn max_shape_in_tree(expr: &Arc<ArrayExpr>) -> Vec<usize> {
106    use std::collections::HashSet;
107    let mut max_shape: Vec<usize> = Vec::new();
108    let mut seen: HashSet<*const ArrayExpr> = HashSet::new();
109    fn walk(
110        expr: &Arc<ArrayExpr>,
111        max_shape: &mut Vec<usize>,
112        seen: &mut HashSet<*const ArrayExpr>,
113    ) {
114        if !seen.insert(Arc::as_ptr(expr)) {
115            return;
116        }
117        for (i, &dim) in expr.shape().iter().enumerate() {
118            if i < max_shape.len() {
119                max_shape[i] = max_shape[i].max(dim);
120            } else {
121                max_shape.push(dim);
122            }
123        }
124        for input in &expr.inputs {
125            walk(input, max_shape, seen);
126        }
127    }
128    walk(expr, &mut max_shape, &mut seen);
129    max_shape
130}

132/// Derive a per-axis chunk target from the widest shape in the DAG, the
133/// root's dtype, and the memory budget. The widest-shape derivation
134/// covers axes that the root drops (e.g. reductions) but inputs still
135/// carry — every node downstream chunks under the same budget.
136pub fn choose_chunk_target(expr: &Arc<ArrayExpr>, memory_budget: usize) -> Vec<usize> {
137    let shape = max_shape_in_tree(expr);
138    let bytes_per_element = expr.dtype.itemsize();
139    let ndim = shape.len();
140    let max_elems = memory_budget / bytes_per_element;
141    let extent = (max_elems as f64).powf(1.0 / ndim as f64).floor().max(1.0) as usize;
142    let target = shape.iter().map(|&s| extent.min(s)).collect::<Vec<_>>();
143
144    debug!(budget = memory_budget, ?shape, ?target, "choose_chunk_target");
145    target
146}

148/// Translate a framed reduce's element-space [`Frame`]s onto the chunk grids
149/// to build the brick-space per-axis access map.
150fn build_reduce_access(
151    view: &HashMap<AxisName, Frame>,
152    input_grid: &ChunkGrid,
153    output_grid: &ChunkGrid,
154) -> ChunkGridResult<HashMap<AxisName, AxisAccess>> {
155    view.iter()
156        .map(|(axis, frame)| -> ChunkGridResult<Option<(AxisName, AxisAccess)>> {
157            Ok(match frame {
158                Frame::WholeAxis { keepdims } => Some((
159                    axis.clone(),
160                    AxisAccess::Composed(BrixFrame::WholeAxis { keepdims: *keepdims }),
161                )),
162                // Coarsen folds `w = ceil(n_in / n_out)` input bricks per output brick.
163                Frame::Window { size, stride } if size == stride => {
164                    let n_in = input_grid.get(axis)?.0.n_chunks();
165                    let n_out = output_grid.get(axis)?.0.n_chunks();
166                    match n_in.div_ceil(n_out) {
167                        1 => None, // identity: no framing on this axis
168                        // Window's semantics allow a possibly smaller last chunk.
169                        w => Some((
170                            axis.clone(),
171                            AxisAccess::Composed(BrixFrame::Window { size: w, stride: w }),
172                        )),
173                    }
174                }
175                other => {
176                    return Err(ChunkGridError::Unsupported(format!(
177                        "lowering Frame {other:?} to brix Frame is not implemented",
178                    )));
179                }
180            })
181        })
182        .collect::<ChunkGridResult<Vec<_>>>()
183        .map(|entries| entries.into_iter().flatten().collect())
184}

186// Note: This was vibe-coded but generally appears OK.
187// 1. The choice to memo-ize may be premature.
188fn lower_inner(
189    expr: &Arc<ArrayExpr>,
190    chunk_target: &[usize],
191    memo: &mut HashMap<*const ArrayExpr, (Arc<Expr>, ChunkGrid)>,
192    grids: &mut HashMap<ExprId, ChunkGrid>,
193) -> LoweringResult<(Arc<Expr>, ChunkGrid)> {
194    let key = Arc::as_ptr(expr);
195    if let Some(hit) = memo.get(&key) {
196        return Ok(hit.clone());
197    }
198
199    let lowered_inputs: Vec<(Arc<Expr>, ChunkGrid)> = expr
200        .inputs
201        .iter()
202        .map(|inp| lower_inner(inp, chunk_target, memo, grids))
203        .collect::<LoweringResult<_>>()?;
204
205    let (node, grid): (Expr, ChunkGrid) = match &expr.op {
206        ArrayOp::TransparentSource(kind) => {
207            let names: Vec<AxisName> = expr.dims().to_vec();
208            let grid = ChunkGrid::new(kind.shape(), &names, chunk_target);
209            let op = kind.lower(grid.clone(), expr.dtype);
210            (
211                Expr {
212                    id: ExprId::fresh(),
213                    op,
214                    inputs: vec![],
215                    space: grid.to_iteration_space(),
216                    label: expr.label.clone(),
217                },
218                grid,
219            )
220        }
221        ArrayOp::OpaqueSource(src) => {
222            // V0: chosen grid == source's native grid (the trivially
223            // native-aligned choice).
224            // FIXME: real choice goes through `chunk_target` and picks the largest axis-aligned multiple
225            // of native that fits;
226            let grid: ChunkGrid = src.native_grid().0.clone();
227            let op = Op::Source(Arc::new(LoweredOpaque { inner: src.clone(), grid: grid.clone() }));
228            (
229                Expr {
230                    id: ExprId::fresh(),
231                    op,
232                    inputs: vec![],
233                    space: grid.to_iteration_space(),
234                    label: expr.label.clone(),
235                },
236                grid,
237            )
238        }
239        ArrayOp::Cast(dst) => {
240            let input_grid = lowered_inputs[0].1.clone();
241            let kernel = Arc::new(CastKernel {
242                src: expr.inputs[0].dtype,
243                dst: *dst,
244                grid: input_grid.clone(),
245            });
246            (
247                Expr {
248                    id: ExprId::fresh(),
249                    op: Op::Blockwise(kernel),
250                    inputs: vec![Input {
251                        expr: lowered_inputs[0].0.clone(),
252                        access: HashMap::new(),
253                    }],
254                    space: input_grid.to_iteration_space(),
255                    label: expr.label.clone(),
256                },
257                input_grid,
258            )
259        }
260        ArrayOp::Unary(op) => {
261            let input_grid = lowered_inputs[0].1.clone();
262            let kernel =
263                Arc::new(UnaryKernel { op: *op, dtype: expr.dtype, grid: input_grid.clone() });
264            (
265                Expr {
266                    id: ExprId::fresh(),
267                    op: Op::Blockwise(kernel),
268                    inputs: vec![Input {
269                        expr: lowered_inputs[0].0.clone(),
270                        access: HashMap::new(),
271                    }],
272                    space: input_grid.to_iteration_space(),
273                    label: expr.label.clone(),
274                },
275                input_grid,
276            )
277        }
278        ArrayOp::Binary(op) => {
279            let input_grids: Vec<&ChunkGrid> = lowered_inputs.iter().map(|l| &l.1).collect();
280            // construct the output chunk grid by broadcasting
281            // FIXME: This should error for non-aligned inputs
282            let grid = broadcast_grids(&input_grids, expr.dims());
283            let kernel = Arc::new(BinaryKernel { op: *op, dtype: expr.dtype, grid: grid.clone() });
284            // All inputs have ChunkAccess::identity since we will be accessing all input chunks
285            // Note that the actual broadcasting semantics are generated by brix/planner.rs
286            // which looks at output grid and input grids and wires up the broadcasting needed
287            // to get the right output.
288            let inputs = vec![
289                Input { expr: lowered_inputs[0].0.clone(), access: HashMap::new() },
290                Input { expr: lowered_inputs[1].0.clone(), access: HashMap::new() },
291            ];
292            (
293                Expr {
294                    id: ExprId::fresh(),
295                    op: Op::Blockwise(kernel),
296                    inputs,
297                    space: grid.to_iteration_space(),
298                    label: expr.label.clone(),
299                },
300                grid,
301            )
302        }
303        ArrayOp::Reduce { acc, view } => {
304            let (input_expr, input_grid) = &lowered_inputs[0];
305            // Reducing along a sparse axis would need the `SparseAccumulator` wrapper.
306            if let Some(occ) = &input_grid.occupancy
307                && let Some(ax) = view.keys().find(|a| occ.sparse_axes.contains(*a))
308            {
309                return Err(ChunkGridError::Unsupported(format!(
310                    "reduce along sparse axis {ax:?} is V1; \
311                     V0 supports reductions along dense axes only",
312                ))
313                .into());
314            }
315            let output_grid = input_grid.after_reduce(view)?;
316            let access = build_reduce_access(view, input_grid, &output_grid)?;
317            // this is a bit of a hack to maintain internal State with `keepdims=True` consistently regardless of the user's preference.
318            // the reduction axis is not squeezed out on the intermediates; we squeeze it out in `finalize` if `keepdims` is `true`.
319            let kept_views: HashMap<_, _> =
320                view.iter().map(|(k, v)| (k.clone(), v.force_keepdims())).collect();
321            let accumulator = Arc::new(LoweredAccumulator {
322                inner: acc.clone(),
323                grid: input_grid.after_reduce(&kept_views)?,
324                view: view.clone(),
325            });
326            (
327                Expr {
328                    id: ExprId::fresh(),
329                    op: Op::Reduce(accumulator),
330                    inputs: vec![Input { expr: input_expr.clone(), access }],
331                    space: output_grid.to_iteration_space(),
332                    label: expr.label.clone(),
333                },
334                output_grid,
335            )
336        }
337        ArrayOp::Slice(slicers) => {
338            let input_grid = lowered_inputs[0].1.clone();
339            let (output_grid, axis_offsets) =
340                chunk_grid::select::output_grid_and_offsets(&input_grid, slicers)?;
341
342            let kernel = Arc::new(SliceKernel {
343                slicers: slicers.clone(),
344                dtype: expr.dtype,
345                grid: output_grid.clone(),
346                input_grid: input_grid.clone(),
347            });
348
349            // Per-edge access — pick up the per-axis chunk offsets the
350            // composition computed. Identity axes (offset 0) contribute
351            // nothing.
352            let access: HashMap<AxisName, AxisAccess> = input_grid
353                .names
354                .iter()
355                .zip(axis_offsets.iter())
356                .map(|(name, &offset)| {
357                    (
358                        name.clone(),
359                        AxisAccess::Translated(TranslatedAccess::Independent(
360                            AxisOverride::Offset(offset),
361                        )),
362                    )
363                })
364                .collect();
365
366            (
367                Expr {
368                    id: ExprId::fresh(),
369                    op: Op::Blockwise(kernel),
370                    inputs: vec![Input { expr: lowered_inputs[0].0.clone(), access }],
371                    space: output_grid.to_iteration_space(),
372                    label: expr.label.clone(),
373                },
374                output_grid,
375            )
376        }
377        ArrayOp::Oindex(indexers) => {
378            let input_grid = lowered_inputs[0].1.clone();
379            let (output_grid, selection) =
380                chunk_grid::oindex::output_grid_and_selection(&input_grid, indexers)?;
381
382            let kernel = Arc::new(GatherKernel {
383                indices: indexers.clone(),
384                dtype: expr.dtype,
385                grid: output_grid.clone(),
386                input_grid: input_grid.clone(),
387            });
388
389            let access: HashMap<AxisName, AxisAccess> = input_grid
390                .names
391                .iter()
392                .zip(selection.iter())
393                .filter_map(|(name, sel)| {
394                    sel.as_ref().map(|r| {
395                        let table: Vec<usize> = r.runs.iter().map(|run| run.source_chunk).collect();
396                        (
397                            name.clone(),
398                            AxisAccess::Translated(TranslatedAccess::Independent(
399                                AxisOverride::Lookup(table),
400                            )),
401                        )
402                    })
403                })
404                .collect();
405
406            (
407                Expr {
408                    id: ExprId::fresh(),
409                    op: Op::Blockwise(kernel),
410                    inputs: vec![Input { expr: lowered_inputs[0].0.clone(), access }],
411                    space: output_grid.to_iteration_space(),
412                    label: expr.label.clone(),
413                },
414                output_grid,
415            )
416        }
417        ArrayOp::Vindex(indexer) => {
418            let input_grid = &lowered_inputs[0].1;
419            let (output_grid, selection) = chunk_grid::vindex::output_grid_and_selection(
420                indexer.produced_axis.clone(),
421                &indexer.indices,
422                input_grid,
423            )?;
424            let followers: HashMap<AxisName, AxisOverride> = selection
425                .iter()
426                .map(|(axis, axis_sel)| {
427                    let table: Vec<usize> = axis_sel.iter().map(|x| x.source_chunk).collect();
428                    (axis.clone(), AxisOverride::Lookup(table))
429                })
430                .collect();
431            let access = HashMap::from([(
432                indexer.produced_axis.clone(),
433                AxisAccess::Translated(TranslatedAccess::Zipped(followers)),
434            )]);
435            let kernel = Arc::new(VIndexKernel {
436                produces: indexer.produced_axis.clone(),
437                selection,
438                dtype: expr.dtype,
439                grid: output_grid.clone(),
440                input_grid: input_grid.clone(),
441            });
442            (
443                Expr {
444                    id: ExprId::fresh(),
445                    op: Op::Blockwise(kernel),
446                    inputs: vec![Input { expr: lowered_inputs[0].0.clone(), access }],
447                    space: output_grid.to_iteration_space(),
448                    label: expr.label.clone(),
449                },
450                output_grid,
451            )
452        }
453    };
454
455    let expr = Arc::new(node);
456    grids.insert(expr.id, grid.clone());
457    let result = (expr, grid);
458    memo.insert(key, result.clone());
459    Ok(result)
460}

467    fn one_slice(axis: &str, range: std::ops::Range<usize>) -> HashMap<AxisName, Indexer> {
468        let mut m = HashMap::new();
469        m.insert(axis.into(), Indexer::Slicer(Slice::contiguous(range.start, range.end)));

473    /// Source → Elementwise → Select → Reduce lowers in one pass; the
474    /// resulting mid-IR has matching dims at every node.
475    #[test]
476    fn end_to_end_lower() {
477        let a = Array::fill(1.0, vec![10, 4], &["i", "j"]);
478        let b = Array::fill(2.0, vec![10, 4], &["i", "j"]);
479        let sum = (a + b).isel(one_slice("i", 0..6)).unwrap();
480        let reduced = sum.over(&["j"]).unwrap().sum().unwrap();
481        let mid = lower(&reduced, 1024).unwrap().expr;
482        assert_eq!(mid.space.names(), &["i"]);
483        assert_eq!(reduced.shape(), vec![6]);
484    }
