11/// Pre-allocates an array of the output shape and writes bricks directly
12/// into it by coordinate. Dtype-generic: the backing buffer is a
13/// [`TypedArray`] sized to the sink's dtype.
14// ANCHOR: in_memory_sink
15pub struct InMemorySink {
16    array: Mutex<TypedArray>,
17    grid: ChunkGrid,
18}
19// ANCHOR_END: in_memory_sink
20
21impl InMemorySink {
22    /// `fill`: pre-fill value for absent / unwritten regions. `None`
23    /// means the dtype's zero. For `BlockSparseArray` lowerings the caller
24    /// passes `Some(arr.fill_value())` so absent bricks resolve to the fill
25    /// value without an explicit write. `fill`'s dtype must match `dtype`.
26    pub fn new(grid: ChunkGrid, dtype: DType, fill: Option<Value>) -> BrixResult<Self> {
27        let array = dispatch_dtype!(dtype, T => {
28            let prefill: T = match fill {
29                Some(v) => v.get::<T>().ok_or_else(|| BrixError::Unsupported(format!(
30                    "InMemorySink: fill value {v:?} does not match sink dtype {dtype:?}"
31                )))?,
32                None => T::zero(),
33            };
34            T::from_array(Array::from_elem(IxDyn(&grid.shape), prefill))
35        });
36        Ok(InMemorySink { array: Mutex::new(array), grid })
37    }
38
39    /// Take the materialized buffer as `ArrayD<T>`. Panics if `T` doesn't
40    /// match the sink's dtype.
41    pub fn into_array<T: Element>(self) -> ArrayD<T> {
42        self.array
43            .into_inner()
44            .unwrap()
45            .into_array::<T>()
46            .expect("into_array: T does not match sink dtype")
47    }
48
49    /// Take the materialized buffer as a dtype-tagged [`TypedArray`].
50    pub fn into_typed(self) -> TypedArray {
51        self.array.into_inner().unwrap()
52    }
53}
54
55impl Sink for InMemorySink {
56    fn init(&self) -> BrixResult<()> {
57        Ok(())
58    }
59
60    fn write_brick(&self, brick: Brick) -> BrixResult<()> {
61        let src = ArrayPayload::from_payload(brick.data.as_ref()).to_dense_dyn();
62        let ranges = self.grid.slice_ranges(&brick.coord).expect("sink coord must match grid axes");
63        let mut array = self.array.lock().unwrap();
64        with_typed_array!(&mut *array, |arr, T| {
65            let mut view = arr.view_mut();
66            for (d, range) in ranges.iter().enumerate() {
67                view.slice_axis_inplace(
68                    Axis(d),
69                    ndarray::Slice::from(range.start as isize..range.end as isize),
70                );
71            }
72            let src_view =
73                src.view().as_typed::<T>().expect("written brick dtype must match sink dtype");
74            view.assign(&src_view);
75        });
76        Ok(())
77    }
78
79    fn finalize(&self) -> BrixResult<()> {
80        Ok(())
81    }
82}
