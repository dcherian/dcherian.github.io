34fn esc(s: &str) -> String {
35    s.replace('&', "&amp;").replace('<', "&lt;").replace('>', "&gt;").replace('"', "&quot;")
36}

38/// Escape a multi-line string for use inside a mermaid `"..."` node label.
39///
40/// Why not `<br/>`: when the mermaid source sits inside `<pre class="mermaid">`,
41/// the browser parses `<br/>` as a real `<br>` element, so `.textContent`
42/// — what mermaid reads — strips the tag and concatenates the surrounding
43/// text. Mermaid's own escape (`\n` as two literal chars, backslash + n) is
44/// rendered as a newline inside the label and survives `textContent`.
45///
46/// Quote characters are sanitized to `#quot;` — mermaid's HTML-entity-like
47/// escape inside quoted labels.
48fn to_mermaid_label(s: &str) -> String {
49    s.replace('"', "#quot;").replace('\n', "\\n")
50}

52/// Lightweight mermaid-source validator. Catches the regressions we've hit:
53/// unclosed node label quotes, unbalanced `[`/`]` outside quoted labels,
54/// edges that reference an undefined node id.
55///
56/// Returns `Err(msg)` describing the first failure, with the offending line
57/// included. Cheap enough to call on every emitted diagram.
58pub fn validate_mermaid(src: &str) -> Result<(), String> {
59    let first = src.lines().next().unwrap_or("").trim();
60    if !first.starts_with("flowchart") && !first.starts_with("graph") {
61        return Err(format!("mermaid source missing header (saw {first:?})"));
62    }
63    let mut node_ids: std::collections::HashSet<String> = std::collections::HashSet::new();
64    let mut edge_refs: Vec<(usize, String, String)> = Vec::new();
65    for (i, raw_line) in src.lines().enumerate() {
66        let line = raw_line.trim();
67        if line.is_empty()
68            || line.starts_with("flowchart")
69            || line.starts_with("graph")
70            || line.starts_with("class ")
71            || line.starts_with("classDef")
72            || line.starts_with("%%")
73        {
74            continue;
75        }
76        // Reject unbalanced quotes.
77        let n_quotes = line.chars().filter(|c| *c == '"').count();
78        if n_quotes % 2 != 0 {
79            return Err(format!("line {}: unbalanced quotes in {raw_line:?}", i + 1));
80        }
81        // Reject unbalanced brackets outside quoted regions.
82        let (mut depth, mut in_quote) = (0i32, false);
83        for c in line.chars() {
84            match c {
85                '"' => in_quote = !in_quote,
86                '[' if !in_quote => depth += 1,
87                ']' if !in_quote => depth -= 1,
88                _ => {}
89            }
90            if depth < 0 {
91                return Err(format!("line {}: stray ']' in {raw_line:?}", i + 1));
92            }
93        }
94        if depth != 0 {
95            return Err(format!("line {}: unbalanced brackets in {raw_line:?}", i + 1));
96        }
97        // Track node declarations vs edge references for later cross-check.
98        // A node declaration looks like `id["label"]` (or with brackets).
99        // An edge looks like `a --> b` or `a -- "label" --> b`.
100        if let Some(rest) = line.split_once('[').map(|(id, _)| id.trim().to_string())
101            && !rest.contains("--")
102            && !rest.is_empty()
103        {
104            node_ids.insert(rest);
105        }
106        if line.contains("-->") {
107            let (lhs, rhs) = line.rsplit_once("-->").unwrap();
108            let src_id = lhs.split("--").next().map(str::trim).unwrap_or("").to_string();
109            let dst_id = rhs.split_whitespace().next().unwrap_or("").to_string();
110            edge_refs.push((i + 1, src_id, dst_id));
111        }
112    }
113    for (lineno, src_id, dst_id) in &edge_refs {
114        for id in [src_id, dst_id] {
115            if !node_ids.contains(id) {
116                return Err(format!("line {lineno}: edge references undeclared node {id:?}"));
117            }
118        }
119    }
120    Ok(())
121}

127/// Mermaid `flowchart TD` source for the ArrayExpr DAG rooted at `arr`.
128/// Nodes carry op kind, axis layout, dtype, and brick kind.
129pub fn array_expr_mermaid(arr: &impl ArrayLike) -> String {
130    let mut out = String::new();
131    // LR (left-right) avoids the mermaid v10/v11 "Could not find a suitable
132    // point" failure that hits TD layouts with edge labels between two
133    // tightly-stacked nodes — there's more horizontal room to place the
134    // label glyph than vertical.
135    out.push_str("flowchart LR\n");
136    out.push_str(CLASS_DEFS);
137    let mut ids: HashMap<*const ArrayExpr, String> = HashMap::new();
138    walk_array_expr(arr.expr(), &mut ids, &mut out);
139    out
140}

142/// Mermaid `classDef` block shared by both DAG renderers. The names match
143/// what the per-node `class id name;` lines emit so mermaid v10 doesn't
144/// error on parse (an undefined class halts the entire diagram).
145///
146/// Palette: Earthmover brand. Source = brand blue, blockwise = light-violet
147/// over violet stroke (the brand's primary accent, applied to the most
148/// common op), reduce = brand green, select = brand orange. Text is
149/// midnight (`#201F2C`) for legibility on all four fills.
150const CLASS_DEFS: &str = "    classDef source fill:#5EC4F7,stroke:#0E7BBF,color:#201F2C;\n    classDef blockwise fill:#C396F9,stroke:#A653FF,color:#201F2C;\n    classDef reduce fill:#31D495,stroke:#0F8A5F,color:#201F2C;\n    classDef select fill:#FF9E0D,stroke:#B25F00,color:#201F2C;\n    classDef clickable stroke:#201F2C,stroke-width:3px,stroke-dasharray:0;\n";

152/// Build the stable id used on both sides of the DAG-node ↔ NodeCard link.
153/// HTML ids only accept `[A-Za-z0-9_-]`; node labels can carry spaces,
154/// parens, etc. (e.g. `Fill(1.0)`), so we slug everything else to `_`.
155/// Kept here as the single source of truth so [`brix_expr_mermaid`]'s
156/// `click ... call openCard("...")` argument always matches the id
157/// [`brix_node_cards_html`] writes onto the corresponding `<details>`.
158pub fn card_id_for_label(label: &str) -> String {
159    let slug: String = label
160        .chars()
161        .map(|c| if c.is_ascii_alphanumeric() || c == '_' || c == '-' { c } else { '_' })
162        .collect();
163    format!("card-{slug}")
164}

166fn walk_array_expr(
167    node: &Arc<ArrayExpr>,
168    ids: &mut HashMap<*const ArrayExpr, String>,
169    out: &mut String,
170) -> String {
171    let key = Arc::as_ptr(node);
172    if let Some(id) = ids.get(&key) {
173        return id.clone();
174    }
175    let id = format!("a{}", ids.len());
176    ids.insert(key, id.clone());
177
178    let label = format_array_expr_node(node);
179    let _ = writeln!(out, "    {id}[\"{}\"]", to_mermaid_label(&label));
180    let class = array_op_class(&node.op);
181    let _ = writeln!(out, "    class {id} {class};");
182
183    for input in &node.inputs {
184        let child = walk_array_expr(input, ids, out);
185        let _ = writeln!(out, "    {child} --> {id}");
186    }
187    id
188}

190fn format_array_expr_node(node: &ArrayExpr) -> String {
191    let mut lines = vec![array_op_short(&node.op)];
192    let dims_extents: Vec<String> = node
193        .layout
194        .0
195        .names
196        .iter()
197        .zip(node.layout.extents().iter())
198        .map(|(n, &e)| format!("{n}:{e}"))
199        .collect();
200    if !dims_extents.is_empty() {
201        lines.push(format!("[{}]", dims_extents.join(", ")));
202    } else {
203        lines.push("[scalar]".into());
204    }
205    lines.push(format!("{:?} / {:?}", node.dtype, node.brick_kind));
206    if let Some(lbl) = &node.label {
207        lines.push(format!("label: {lbl}"));
208    }
209    lines.join("\n")
210}

212fn array_op_short(op: &ArrayOp) -> String {
213    match op {
214        ArrayOp::TransparentSource(TransparentSource::Fill { value, .. }) => {
215            format!("Fill({value})")
216        }
217        ArrayOp::TransparentSource(TransparentSource::Dense(_)) => "Dense".into(),
218        ArrayOp::OpaqueSource(_) => "Opaque".into(),
219        ArrayOp::Binary(b) => format!("Binary({b:?})"),
220        ArrayOp::Unary(u) => format!("Unary({u:?})"),
221        ArrayOp::Reduce { view, .. } => {
222            let axes: Vec<&AxisName> = view.keys().collect();
223            format!("Reduce(axes={axes:?})")
224        }
225        ArrayOp::Slice(s) => format!("Slice({} axes)", s.len()),
226        ArrayOp::Oindex(o) => format!("Oindex({} axes)", o.len()),
227        ArrayOp::Vindex(_) => "Vindex".into(),
228        ArrayOp::Cast(dtype) => format!("Cast({dtype:?})"),
229    }
230}
231
232fn array_op_class(op: &ArrayOp) -> &'static str {
233    match op {
234        ArrayOp::TransparentSource(_) | ArrayOp::OpaqueSource(_) => "source",
235        ArrayOp::Reduce { .. } => "reduce",
236        ArrayOp::Slice(_) | ArrayOp::Oindex(_) | ArrayOp::Vindex(_) => "select",
237        ArrayOp::Binary(_) | ArrayOp::Unary(_) | ArrayOp::Cast(_) => "blockwise",
238    }
239}

245/// Mermaid `flowchart TD` source for the BrixExpr DAG, using
246/// `build_expr_graph` for identity dedup. Edges are annotated with core
247/// dims, the access-relation token form, and any axes that broadcast from
248/// producer to consumer.
249///
250/// Labeled nodes get a mermaid `click ... call openCard("<sanitized>")`
251/// binding so the host page can scroll to the corresponding [`NodeCard`].
252/// `card_id` must match the `id` attribute used by [`brix_node_cards_html`].
253pub fn brix_expr_mermaid(expr: &Expr) -> String {
254    let (graph, _root) = build_expr_graph(expr);
255    let mut out = String::new();
256    out.push_str("flowchart LR\n");
257    out.push_str(CLASS_DEFS);
258
259    for nidx in graph.node_indices() {
260        let n = &graph[nidx];
261        let label = format!("{}\ngrid: {}", op_short(&n.op), grid_summary(&n.space),);
262        let label = if let Some(lbl) = &n.label { format!("{label}\nlabel: {lbl}") } else { label };
263        let id = format!("b{}", nidx.index());
264        let _ = writeln!(&mut out, "    {id}[\"{}\"]", to_mermaid_label(&label));
265        // Mermaid's `class A,B name` syntax expects comma-separated
266        // *node ids*, not classes — multiple classes need separate
267        // lines. The `click` directive below adds its own `clickable`
268        // class to the rendered SVG node, which our `classDef clickable`
269        // styles (thick dark outline + cursor:pointer), so we don't
270        // emit a redundant `class {id} clickable;` line ourselves.
271        let class = op_class(&n.op);
272        let _ = writeln!(&mut out, "    class {id} {class};");
273        if let Some(lbl) = &n.label {
274            // openCard takes the card id; the host page resolves it
275            // *within the currently-visible pane* so duplicate ids
276            // across panes (every workload uses "A"/"B"/"add" etc.)
277            // don't all collapse to the first match in document order.
278            let _ =
279                writeln!(&mut out, "    click {id} call openCard(\"{}\")", card_id_for_label(lbl),);
280        }
281    }
282
283    for eref in (&graph).edge_references() {
284        let src = eref.source();
285        let dst = eref.target();
286        let edge = eref.weight();
287        let prod_grid = &graph[src].space;
288        let cons_grid = &graph[dst].space;
289        let label = edge_label(
290            &fold_axes(&edge.access),
291            &translated_overrides(&edge.access),
292            prod_grid,
293            cons_grid,
294        );
295        let _ = if label.is_empty() {
296            writeln!(&mut out, "    b{} --> b{}", src.index(), dst.index())
297        } else {
298            writeln!(&mut out, "    b{} -- \"{}\" --> b{}", src.index(), esc(&label), dst.index(),)
299        };
300    }
301    out
302}

314/// DOT source for the BrixExpr DAG. Pair with [`brix_node_cards_html`]
315/// (cards share their `id` with the DAG node id when the node carries a
316/// label) so a click on a node can jump to its card. Edges carry a
317/// `tooltip=` derived from the same `(core_dims, access, broadcast)`
318/// triple shown on the card — hover text matches the card row.
319pub fn brix_expr_dot(expr: &Expr) -> String {
320    let (graph, _root) = build_expr_graph(expr);
321    // Build a "this axis is conventionally a row axis here" set by
322    // observing which axes occupy `names[0]` of any 2-D peer in this
323    // DAG. A 1-D node whose axis is in this set then renders as a
324    // column (N rows × 1 col), preserving the i/j orientation
325    // end-to-end through reductions and broadcasts.
326    let mut row_axes: HashSet<AxisName> = HashSet::new();
327    for nidx in graph.node_indices() {
328        let names = graph[nidx].space.names();
329        if names.len() >= 2 {
330            row_axes.insert(names[0].clone());
331        }
332    }
333    let mut out = String::new();
334    out.push_str("digraph G {\n");
335    out.push_str("  rankdir=LR;\n");
336    out.push_str("  bgcolor=\"transparent\";\n");
337    // node defaults — class is set per-node so CSS can pick up
338    // .source/.blockwise/.reduce/.select fills. We still pass fillcolor
339    // here so the DOT renders correctly even before the host CSS loads.
340    out.push_str(
341        "  node [shape=box, style=\"rounded,filled\", \
342         fontname=\"-apple-system,BlinkMacSystemFont,Segoe UI,sans-serif\", \
343         fontsize=11, margin=\"0.18,0.10\", penwidth=1.5];\n",
344    );
345    out.push_str(
346        "  edge [fontname=\"-apple-system,BlinkMacSystemFont,Segoe UI,monospace\", \
347         fontsize=10, color=\"#201F2C\", arrowsize=0.7];\n",
348    );
349
350    for nidx in graph.node_indices() {
351        let n = &graph[nidx];
352        let class = op_class(&n.op);
353        let (fill, stroke) = dot_node_colors(class);
354        // Labeled nodes anchor a card and get the click affordance —
355        // graphviz draws the outer rounded rectangle thick for them so
356        // they read as "named, anchorable" at a glance.
357        let (id_attr, extra_class, penwidth) = if let Some(lbl) = &n.label {
358            (format!(", id=\"{}\"", card_id_for_label(lbl)), " clickable", 3.0_f32)
359        } else {
360            (String::new(), "", 1.5_f32)
361        };
362        // shape=box (rounded,filled) lets graphviz draw the OUTER node
363        // outline with `penwidth`, separate from the HTML label's inner
364        // table borders. That separation is what lets the chunk-grid
365        // cells render hairline-thin while the node body stays bold —
366        // a single CSS `stroke-width` rule would catch both polygons.
367        let html_label = dot_node_html_label(n, fill, &row_axes);
368        let _ = writeln!(
369            &mut out,
370            "  b{} [label=<{html_label}>, shape=box, style=\"rounded,filled\", \
371             fillcolor=\"{fill}\", color=\"{stroke}\", penwidth={penwidth}, \
372             class=\"node {class}{extra_class}\"{id_attr}];",
373            nidx.index(),
374        );
375    }
376
377    for eref in (&graph).edge_references() {
378        let src = eref.source();
379        let dst = eref.target();
380        let edge = eref.weight();
381        let prod_grid = &graph[src].space;
382        let cons_grid = &graph[dst].space;
383        let core = fold_axes(&edge.access);
384        let overrides = translated_overrides(&edge.access);
385        let label = edge_label(&core, &overrides, prod_grid, cons_grid);
386        let tooltip = edge_tooltip(
387            &graph[src].label,
388            &graph[dst].label,
389            &core,
390            &overrides,
391            prod_grid,
392            cons_grid,
393        );
394        // Graphviz emits the tooltip as an SVG <title> child — browsers
395        // render it as a native hover tooltip on the edge. No JS needed.
396        let _ = writeln!(
397            &mut out,
398            "  b{} -> b{} [label=\"{}\", tooltip=\"{}\", labeltooltip=\"{}\", id=\"e{}-{}\"];",
399            src.index(),
400            dst.index(),
401            dot_escape(&label),
402            dot_escape(&tooltip),
403            dot_escape(&tooltip),
404            src.index(),
405            dst.index(),
406        );
407    }
408
409    out.push_str("}\n");
410    out
411}

413/// Graphviz HTML label for a BrixExpr DAG node — a two-column layout
414/// with the op/grid/label text on the left and a TABLE-based mini
415/// chunk-grid on the right. Each grid cell is a `<TD>` with a
416/// `BGCOLOR=` matching the node fill; sparse-absent cells get a muted
417/// fill. Self-contained: no base64/SVG embedding, no IMG dimensions to
418/// negotiate with graphviz's layout engine.
419fn dot_node_html_label(
420    n: &crate::brix::expr_graph::ExprNode,
421    _node_fill: &str,
422    row_axes: &HashSet<AxisName>,
423) -> String {
424    let mut html = String::new();
425    // Outer 1-row × 2-col table: text | mini-grid. BORDER="0" — the
426    // node body outline is drawn by graphviz via shape=box + penwidth,
427    // so the HTML label only contributes its inner structure.
428    // CELLPADDING gives breathing room between the text block, the row
429    // label, and the grid — without it the `i: 3` annotation slams
430    // against the text on its left and the grid on its right.
431    html.push_str("<TABLE BORDER=\"0\" CELLBORDER=\"0\" CELLSPACING=\"0\" CELLPADDING=\"10\">");
432    html.push_str("<TR>");
433    html.push_str("<TD ALIGN=\"LEFT\" BALIGN=\"LEFT\">");
434    // Op header + grid summary + optional label (each row left-aligned
435    // via the explicit <BR ALIGN="LEFT"/> graphviz needs).
436    html.push_str(&format!(
437        "<FONT POINT-SIZE=\"12\"><B>{}</B></FONT><BR ALIGN=\"LEFT\"/>",
438        html_label_escape(&op_short(&n.op)),
439    ));
440    html.push_str(&format!(
441        "<FONT POINT-SIZE=\"10\">grid: {}</FONT><BR ALIGN=\"LEFT\"/>",
442        html_label_escape(&grid_summary(&n.space)),
443    ));
444    if let Some(lbl) = &n.label {
445        html.push_str(&format!(
446            "<FONT POINT-SIZE=\"10\">label: {}</FONT><BR ALIGN=\"LEFT\"/>",
447            html_label_escape(lbl),
448        ));
449    }
450    html.push_str("</TD>");
451    html.push_str("<TD ALIGN=\"CENTER\" VALIGN=\"MIDDLE\">");
452    html.push_str(&chunk_grid_html_table(&n.space, row_axes));
453    html.push_str("</TD>");
454    html.push_str("</TR></TABLE>");
455    html
456}

458/// Mini chunk-grid as a Graphviz HTML TABLE — dask-array-repr style.
459/// One TD per chunk, fill color reflecting present/absent occupancy,
460/// axes annotated outside the grid as `name: n_chunks`. 1-D grids
461/// render as a single row, 2-D as a matrix, ≥3-D collapses to a "ND"
462/// marker (graphviz HTML tables don't have a tidy way to draw rank-3+).
463fn chunk_grid_html_table(grid: &IterationSpace, row_axes: &HashSet<AxisName>) -> String {
464    let counts = grid.max_extents();
465    let names = grid.names();
466    // Present cells use white (high contrast against the colored node
467    // body); absent cells use a muted neutral so sparsity reads at a
468    // glance. Borders are midnight everywhere.
469    let present_fill = "#FFFFFF";
470    let absent_fill = "#E5E2EC";
471    if names.is_empty() {
472        return format!(
473            "<TABLE BORDER=\"1\" CELLBORDER=\"0\" CELLSPACING=\"0\" CELLPADDING=\"3\" \
474             COLOR=\"#201F2C\"><TR><TD BGCOLOR=\"{present_fill}\">\
475             <FONT POINT-SIZE=\"8\">scalar</FONT></TD></TR></TABLE>",
476        );
477    }
478    if names.len() > 2 {
479        // High-rank fallback — outline the rank rather than try to
480        // render a 3+ D table in 2-D.
481        let chunks: Vec<String> =
482            names.iter().zip(counts.iter()).map(|(n, c)| format!("{n}:{c}")).collect();
483        return format!(
484            "<TABLE BORDER=\"1\" CELLBORDER=\"0\" CELLSPACING=\"0\" CELLPADDING=\"3\" \
485             COLOR=\"#201F2C\"><TR><TD BGCOLOR=\"{present_fill}\">\
486             <FONT POINT-SIZE=\"8\">{}D · {}</FONT></TD></TR></TABLE>",
487            names.len(),
488            chunks.join(", "),
489        );
490    }
491    let is_1d = names.len() == 1;
492    // For 1-D, orient as a column if this axis is the row axis of any
493    // 2-D peer in the DAG — keeps i/j orientation consistent through
494    // reductions ("j on rows" stays "j on rows" in the reduced output).
495    let as_column = is_1d && row_axes.contains(&names[0]);
496    let (n_rows, n_cols) = if is_1d {
497        if as_column { (counts[0].max(1), 1) } else { (1, counts[0].max(1)) }
498    } else {
499        (counts[0].max(1), counts[1].max(1))
500    };
501    let (row_name, col_name) = if is_1d {
502        if as_column { (names[0].as_str(), "") } else { ("", names[0].as_str()) }
503    } else {
504        (names[0].as_str(), names[1].as_str())
505    };
506    let occupancy = grid.occupancy();
507    // Cell extent scaled so the grid stays compact: 14px when small,
508    // shrinking as the count grows. A 6×6 grid still fits comfortably
509    // next to the text block.
510    let cell_px = (14usize).saturating_sub((n_rows.max(n_cols)).saturating_sub(3));
511    let cell_px = cell_px.max(8);
512
513    // Inner grid table — borders only, no padding, fixed-size cells.
514    let mut inner = String::new();
515    inner.push_str(
516        "<TABLE BORDER=\"1\" CELLBORDER=\"1\" CELLSPACING=\"0\" CELLPADDING=\"0\" \
517         COLOR=\"#201F2C\">",
518    );
519    for r in 0..n_rows {
520        inner.push_str("<TR>");
521        for c in 0..n_cols {
522            let present = match occupancy {
523                None => true,
524                Some(occ) => grid_chunk_present(occ, names, r, c, is_1d),
525            };
526            let fill = if present { present_fill } else { absent_fill };
527            inner.push_str(&format!(
528                "<TD BGCOLOR=\"{fill}\" WIDTH=\"{cell_px}\" HEIGHT=\"{cell_px}\" \
529                 FIXEDSIZE=\"TRUE\"> </TD>",
530            ));
531        }
532        inner.push_str("</TR>");
533    }
534    inner.push_str("</TABLE>");
535
536    // Two-level wrapper so the column label centres directly above the
537    // grid, not above the combined `row-label | grid` strip:
538    //
539    //   [ row_label ][ inner_wrap ]
540    //                ┌─────────────┐
541    //                │   col_label  │  ← inner_wrap is its own table:
542    //                ├─────────────┤    col label sits in a single-column
543    //                │    grid      │    cell above the grid, so ALIGN=CENTER
544    //                └─────────────┘    means "centred on the grid".
545    //
546    // For 1-D rendered as a row (e.g., `j: 4`) only the col label is
547    // populated; for 1-D rendered as a column (e.g., `i: 4` after a
548    // reduce kept `i` on rows) only the row label is populated.
549    let row_label_html = if row_name.is_empty() {
550        String::new()
551    } else {
552        format!("<FONT POINT-SIZE=\"8\">{}: {}</FONT>", html_label_escape(row_name), n_rows)
553    };
554    let col_label_html = if col_name.is_empty() {
555        String::new()
556    } else {
557        format!("<FONT POINT-SIZE=\"8\">{}: {}</FONT>", html_label_escape(col_name), n_cols)
558    };
559    let inner_wrap = format!(
560        "<TABLE BORDER=\"0\" CELLBORDER=\"0\" CELLSPACING=\"0\" CELLPADDING=\"1\">\
561         <TR><TD ALIGN=\"CENTER\">{col_label_html}</TD></TR>\
562         <TR><TD>{inner}</TD></TR></TABLE>",
563    );
564    format!(
565        "<TABLE BORDER=\"0\" CELLBORDER=\"0\" CELLSPACING=\"0\" CELLPADDING=\"2\">\
566         <TR><TD ALIGN=\"RIGHT\" VALIGN=\"MIDDLE\">{row_label_html}</TD>\
567         <TD VALIGN=\"BOTTOM\">{inner_wrap}</TD></TR></TABLE>",
568    )
569}

571/// `Occupancy` membership probe used by [`chunk_grid_html_table`].
572/// Mirrors [`chunk_in_occupancy`] but operates on the brix-layer
573/// [`IterationSpace::names`] / chunk-count plane (no element extents).
574fn grid_chunk_present(
575    occ: &crate::brix::iteration_space::Occupancy,
576    grid_names: &[AxisName],
577    r: usize,
578    c: usize,
579    is_1d: bool,
580) -> bool {
581    let coord: Vec<usize> = if is_1d { vec![c] } else { vec![r, c] };
582    let mut probe = Vec::with_capacity(occ.sparse_axes.len());
583    for axis in &occ.sparse_axes {
584        let pos = match grid_names.iter().position(|n| n == axis) {
585            Some(p) => p,
586            None => return true,
587        };
588        if pos >= coord.len() {
589            return true;
590        }
591        probe.push(coord[pos] as u32);
592    }
593    occ.as_coo_coords().contains(&probe)
594}

621/// `"` and `\` are special; newlines are conveyed as the literal `\n`

636/// Tooltip text shown when hovering an edge in the DOT-rendered DAG.
637/// Same shape as the `incoming edges` row inside a [`NodeCard`]:
638/// producer → consumer · core dims · access tokens · broadcast axes.
639fn edge_tooltip(
640    producer_label: &Option<String>,
641    consumer_label: &Option<String>,
642    core_dims: &[String],
643    overrides: &HashMap<AxisName, TranslatedAccess>,
644    producer_grid: &IterationSpace,
645    consumer_grid: &IterationSpace,
646) -> String {
647    let producer = producer_label.clone().unwrap_or_else(|| "<unlabeled>".into());
648    let consumer = consumer_label.clone().unwrap_or_else(|| "<unlabeled>".into());
649    let mut parts: Vec<String> = vec![format!("{producer} → {consumer}")];
650    if !core_dims.is_empty() {
651        let core: Vec<String> = core_dims.iter().map(|d| d.to_string()).collect();
652        parts.push(format!("core dims: {{{}}}", core.join(", ")));
653    }
654    let acc = access_relation_inline(overrides);
655    if !acc.is_empty() {
656        parts.push(format!("access: {acc}"));
657    } else if core_dims.is_empty() {
658        // Identity edges produce an empty `acc` and no core dims — say
659        // so explicitly in the tooltip so the hover isn't just the
660        // producer→consumer line. (Non-empty `core_dims` already
661        // surface above.)
662        parts.push("access: Id — output chunk i ⇐ producer chunk i (identity)".to_string());
663    }
664    // Broadcast axes are the consumer-iter axes the producer doesn't
665    // carry, mirroring the logic on EdgeCard.broadcast_axes.
666    let bcast: Vec<AxisName> = consumer_grid
667        .names()
668        .iter()
669        .filter(|n| !producer_grid.names().contains(n) && !overrides.contains_key(n.as_str()))
670        .cloned()
671        .collect();
672    if !bcast.is_empty() {
673        parts.push(format!(
674            "{} broadcast across axis {} — stride 0 on the consumer side; \
675             the same producer chunk is reused as the consumer iterates that axis.",
676            producer,
677            bcast.join(", "),
678        ));
679    }
680    parts.join("\n")
681}

723fn edge_label(
724    core_dims: &[String],
725    overrides: &HashMap<AxisName, TranslatedAccess>,
726    producer: &IterationSpace,
727    consumer: &IterationSpace,
728) -> String {
729    let mut parts: Vec<String> = Vec::new();
730    if !core_dims.is_empty() {
731        parts.push(format!("core: {{{}}}", core_dims.join(",")));
732    }
733    for (axis, entry) in overrides {
734        parts.push(match entry {
735            TranslatedAccess::Independent(a) => format!("{axis}={}", axis_access_token(a)),
736            TranslatedAccess::Zipped(followers) => {
737                let inner: Vec<String> = followers
738                    .iter()
739                    .map(|(fa, ov)| format!("{fa}={}", axis_access_token(ov)))
740                    .collect();
741                format!("{axis}=Zip{{{}}}", inner.join(","))
742            }
743        });
744    }
745    // Implicit broadcast: axes the consumer iterates that the producer does
746    // not carry. These are stride-0 (Bcast) in the producer's chunk space.
747    let prod_names: Vec<&str> = producer.names().iter().map(|s| s.as_str()).collect();
748    for cname in consumer.names() {
749        if !prod_names.contains(&cname.as_str()) && !overrides.contains_key(cname) {
750            parts.push(format!("{cname}=Bcast"));
751        }
752    }
753    // Identity edges still get an "Id" label so the DAG never has an
754    // unlabeled arrow — easier to read at a glance.
755    if parts.is_empty() {
756        return "Id".to_string();
757    }
758    parts.join(" | ")
759}

905// ===========================================================================
906// Brix-layer grid SVG — uniform cells, one cell per chunk. The mid-IR
907// `brix::IterationSpace` only knows per-axis chunk counts (and, for sparse, which
908// coords are present). Element extents and the rich AxisChunks encoding
909// (Uniform/RLE/Variable) live on the array-layer `ChunkGrid` and are
910// rendered separately in panel ③.
911// ===========================================================================
912
913pub fn brix_grid_svg(grid: &IterationSpace, payload_kind: ArrayPayloadKind) -> String {
914    let counts = grid.max_extents();
915    if counts.is_empty() {
916        return "<svg><text>scalar</text></svg>".into();
917    }
918    let is_1d = counts.len() == 1;
919    let (n_rows, n_cols, row_name, col_name) = if is_1d {
920        (1usize, counts[0].max(1), "·", grid.names()[0].as_str())
921    } else {
922        (counts[0].max(1), counts[1].max(1), grid.names()[0].as_str(), grid.names()[1].as_str())
923    };
924    let cell = 28usize;
925    let pad = 32usize;
926    let w = n_cols * cell + pad * 2;
927    let h = n_rows * cell + pad * 2 + 14;
928
929    let occupancy = grid.occupancy();
930    let mut out = String::new();
931    let _ = writeln!(
932        &mut out,
933        "<svg viewBox=\"0 0 {w} {h}\" width=\"{w}\" height=\"{h}\" \
934         xmlns=\"http://www.w3.org/2000/svg\" class=\"brix-grid-svg\">",
935    );
936    let _ = writeln!(
937        &mut out,
938        "  <text x=\"{}\" y=\"{}\" class=\"bg-axis\">{}</text>",
939        pad,
940        pad - 10,
941        esc(col_name),
942    );
943    let _ = writeln!(
944        &mut out,
945        "  <text x=\"{}\" y=\"{}\" class=\"bg-axis\" transform=\"rotate(-90 {} {})\">{}</text>",
946        pad - 20,
947        pad + (n_rows * cell) / 2,
948        pad - 20,
949        pad + (n_rows * cell) / 2,
950        esc(row_name),
951    );
952
953    let names = grid.names();
954    for r in 0..n_rows {
955        for c in 0..n_cols {
956            let coord: Vec<usize> = if is_1d { vec![c] } else { vec![r, c] };
957            // Sparse occupancy check — project (r,c) into the sparse_axes order.
958            let present = match occupancy {
959                None => true,
960                Some(occ) => {
961                    let mut probe = Vec::with_capacity(occ.sparse_axes.len());
962                    let mut ok = true;
963                    for ax in &occ.sparse_axes {
964                        let pos = names.iter().position(|n| n == ax);
965                        match pos {
966                            Some(p) if p < coord.len() => probe.push(coord[p] as u32),
967                            _ => {
968                                ok = false;
969                                break;
970                            }
971                        }
972                    }
973                    if !ok { true } else { occ.as_coo_coords().contains(&probe) }
974                }
975            };
976            let x = pad + c * cell;
977            let y = pad + r * cell;
978            let class = if present { "bg-cell-present" } else { "bg-cell-absent" };
979            let _ = writeln!(
980                &mut out,
981                "  <rect class=\"{class}\" x=\"{x}\" y=\"{y}\" width=\"{cell}\" height=\"{cell}\"/>",
982            );
983            let coord_text = if is_1d { format!("({c})") } else { format!("({r},{c})") };
984            let _ = writeln!(
985                &mut out,
986                "  <text x=\"{}\" y=\"{}\" class=\"bg-cell-coord\" text-anchor=\"middle\" \
987                 dominant-baseline=\"middle\">{}</text>",
988                x + cell / 2,
989                y + cell / 2,
990                esc(&coord_text),
991            );
992        }
993    }
994
995    let _ = writeln!(
996        &mut out,
997        "  <rect class=\"bg-frame\" x=\"{pad}\" y=\"{pad}\" width=\"{}\" height=\"{}\"/>",
998        n_cols * cell,
999        n_rows * cell,
1000    );
1001
1002    let caption = match occupancy {
1003        Some(occ) => format!(
1004            "Sparse · {} present coords on axes [{}]",
1005            occ.len(),
1006            occ.sparse_axes.join(", ")
1007        ),
1008        None => format!(
1009            "Dense · {} bricks (n_chunks={:?}), kind={:?}",
1010            grid.total_bricks(),
1011            counts,
1012            payload_kind,
1013        ),
1014    };
1015    let _ = writeln!(
1016        &mut out,
1017        "  <text x=\"{pad}\" y=\"{}\" class=\"bg-caption\">{}</text>",
1018        h - 4,
1019        esc(&caption),
1020    );
1021    out.push_str("</svg>\n");
1022    out
1023}

1072/// Like [`chunk_grid_svg`] but with a caller-chosen target for the longer
1073/// side of the drawn grid (in CSS pixels). Useful when packing several
1074/// grids into a fixed column.
1075pub fn chunk_grid_svg_sized(
1076    grid: &ChunkGrid,
1077    payload_kind: ArrayPayloadKind,
1078    target_px: usize,
1079) -> String {
1080    if grid.shape.is_empty() {
1081        return "<svg><text>scalar</text></svg>".into();
1082    }
1083    let is_1d = grid.shape.len() == 1;
1084    let row_strip = AxisChunks::Uniform { size: 1, n_chunks: 1 };
1085    let (row_chunks, col_chunks, row_name, col_name) = if is_1d {
1086        (&row_strip, &grid.chunks[0], "·", grid.names[0].as_str())
1087    } else {
1088        (&grid.chunks[0], &grid.chunks[1], grid.names[0].as_str(), grid.names[1].as_str())
1089    };
1090    let row_extent: usize = row_chunks.extent().max(1);
1091    let col_extent: usize = col_chunks.extent().max(1);
1092    let n_row_chunks = row_chunks.n_chunks().max(1);
1093    let n_col_chunks = col_chunks.n_chunks().max(1);
1094
1095    // Scale so the longer side of the drawn area is exactly `target_px`.
1096    let longer = row_extent.max(col_extent).max(1);
1097    let elem_px = (target_px / longer).max(8);
1098
1099    let w = col_extent * elem_px + PAD * 2;
1100    let h = row_extent * elem_px + PAD * 2;
1101
1102    let mut out = String::new();
1103    // Explicit width/height: without these, SVGs inside `display: flex`
1104    // containers (the node cards) collapse to 0 because they have no
1105    // intrinsic dimensions.
1106    let _ = writeln!(
1107        &mut out,
1108        "<svg viewBox=\"0 0 {w} {h}\" width=\"{w}\" height=\"{h}\" \
1109         xmlns=\"http://www.w3.org/2000/svg\" class=\"cg-svg\">",
1110    );
1111
1112    // Axis name labels.
1113    let _ = writeln!(
1114        &mut out,
1115        "  <text x=\"{}\" y=\"{}\" class=\"cg-axis-label\">{}</text>",
1116        PAD,
1117        PAD - 12,
1118        esc(col_name),
1119    );
1120    let _ = writeln!(
1121        &mut out,
1122        "  <text x=\"{}\" y=\"{}\" class=\"cg-axis-label\" transform=\"rotate(-90 {} {})\">{}</text>",
1123        PAD - 22,
1124        PAD + (row_extent * elem_px) / 2,
1125        PAD - 22,
1126        PAD + (row_extent * elem_px) / 2,
1127        esc(row_name),
1128    );
1129
1130    // Per-chunk rects + labels.
1131    let occupancy = grid.occupancy.as_ref();
1132    for ri in 0..n_row_chunks {
1133        for ci in 0..n_col_chunks {
1134            // Element-space bounds for this chunk.
1135            let (r_start, r_end, c_start, c_end) = if grid.shape.len() == 1 {
1136                let r = 0..1;
1137                let c = col_chunks.slice_range(ci);
1138                (r.start, r.end, c.start, c.end)
1139            } else {
1140                let r = row_chunks.slice_range(ri);
1141                let c = col_chunks.slice_range(ci);
1142                (r.start, r.end, c.start, c.end)
1143            };
1144            let x = PAD + c_start * elem_px;
1145            let y = PAD + r_start * elem_px;
1146            let cw = (c_end - c_start) * elem_px;
1147            let ch = (r_end - r_start) * elem_px;
1148
1149            let absent = if let Some(occ) = occupancy {
1150                // Map (ri, ci) into the sparse-axis coord schema; sparse_axes
1151                // is a subset of grid.names so we look up by axis name.
1152                !chunk_in_occupancy(occ, grid, ri, ci)
1153            } else {
1154                false
1155            };
1156
1157            let (fill_class, label_text) = if absent {
1158                ("cg-cell cg-cell-absent", "absent".to_string())
1159            } else {
1160                let kind = chunk_label(payload_kind);
1161                ("cg-cell cg-cell-present", kind)
1162            };
1163
1164            let _ = writeln!(
1165                &mut out,
1166                "  <rect class=\"{fill_class}\" x=\"{x}\" y=\"{y}\" width=\"{cw}\" height=\"{ch}\"/>",
1167            );
1168            let _ = writeln!(
1169                &mut out,
1170                "  <text x=\"{}\" y=\"{}\" class=\"cg-cell-label\" text-anchor=\"middle\">{}</text>",
1171                x + cw / 2,
1172                y + ch / 2 + 4,
1173                esc(&label_text),
1174            );
1175            // Chunk coord tag in the upper-left.
1176            let coord_text =
1177                if grid.shape.len() == 1 { format!("({ci})") } else { format!("({ri},{ci})") };
1178            let _ = writeln!(
1179                &mut out,
1180                "  <text x=\"{}\" y=\"{}\" class=\"cg-cell-coord\">{}</text>",
1181                x + 4,
1182                y + 12,
1183                esc(&coord_text),
1184            );
1185        }
1186    }
1187
1188    // Outer frame.
1189    let _ = writeln!(
1190        &mut out,
1191        "  <rect class=\"cg-frame\" x=\"{PAD}\" y=\"{PAD}\" width=\"{}\" height=\"{}\"/>",
1192        col_extent * elem_px,
1193        row_extent * elem_px,
1194    );
1195
1196    if let Some(occ) = occupancy {
1197        let _ = writeln!(
1198            &mut out,
1199            "  <text x=\"{PAD}\" y=\"{}\" class=\"cg-caption\">Occupancy: {} present coords on axes {{{}}}</text>",
1200            h - 6,
1201            occ.len(),
1202            occ.sparse_axes.join(","),
1203        );
1204    } else {
1205        let _ = writeln!(
1206            &mut out,
1207            "  <text x=\"{PAD}\" y=\"{}\" class=\"cg-caption\">Dense grid · {} bricks · element shape {:?}</text>",
1208            h - 6,
1209            grid.total_bricks(),
1210            grid.shape,
1211        );
1212    }
1213
1214    out.push_str("</svg>\n");
1215    out
1216}

1219// Broadcast illustration — shows each operand's native shape and how it
1220// tiles into the consumer's iter space. Stride-0 (broadcast) axes appear
1221// as repeated cells with dashed borders; native axes get solid borders.
1222// ===========================================================================

1224/// One operand to a broadcast illustration: its display label and the
1225/// axes it natively carries (`(name, extent)` pairs).
1226pub struct BroadcastOperand {
1227    pub label: String,
1228    pub dims: Vec<(String, usize)>,
1229}

1231/// Render an SVG showing each operand's native shape laid out within the
1232/// consumer's full output dims, with broadcast (stride-0) axes drawn as
1233/// dashed-border repeated cells. Works for 1-D and 2-D operands /
1234/// outputs; higher ranks fall back to the textual edge tokens.
1235///
1236/// `cell_label` controls what's drawn in each cell — "elem" stamps the
1237/// element coord (array-layer view), "chunk" stamps the chunk coord
1238/// (brix-layer view).
1239pub fn broadcast_diagram_svg(
1240    operands: &[BroadcastOperand],
1241    output: &[(String, usize)],
1242    cell_label: &str,
1243) -> String {
1244    if output.is_empty() || output.len() > 2 {
1245        return String::new();
1246    }
1247    let (row_name, row_extent, col_name, col_extent) = match output.len() {
1248        1 => ("·".to_string(), 1, output[0].0.clone(), output[0].1),
1249        _ => (output[0].0.clone(), output[0].1, output[1].0.clone(), output[1].1),
1250    };
1251
1252    let cell_px = 26usize;
1253    let pad = 32usize;
1254    let n_operands = operands.len();
1255    let block_w = col_extent * cell_px;
1256    let block_h = row_extent * cell_px;
1257    // Big enough that the centered +/= glyph at gap-midpoint sits well
1258    // clear of the next block's left-axis label (text-anchor=end at
1259    // `x0 - LABEL_OFFSET`). With `block_gap = 60` and `LABEL_OFFSET = 14`,
1260    // the glyph and the label live in clearly separate horizontal bands.
1261    let block_gap = 60usize;
1262    let block_total = block_w + block_gap;
1263    let w = pad * 2 + block_total * (n_operands + 1).max(1) + 40;
1264    let h = pad * 2 + block_h + 56;
1265
1266    let mut out = String::new();
1267    let _ = writeln!(
1268        &mut out,
1269        "<svg viewBox=\"0 0 {w} {h}\" width=\"{w}\" height=\"{h}\" \
1270         xmlns=\"http://www.w3.org/2000/svg\" class=\"bc-svg\">",
1271    );
1272
1273    // Draw operand blocks left-to-right, then "=" then output.
1274    for (idx, op) in operands.iter().enumerate() {
1275        let x0 = pad + idx * block_total;
1276        let op_dims: HashMap<&str, usize> = op.dims.iter().map(|(n, e)| (n.as_str(), *e)).collect();
1277        draw_broadcast_block(
1278            &mut out, x0, pad, &row_name, row_extent, &col_name, col_extent, cell_px, &op.label,
1279            &op_dims,
1280        );
1281        // "+" between operands (except after the last one)
1282        if idx + 1 < n_operands {
1283            let xc = pad + (idx + 1) * block_total - block_gap / 2;
1284            let yc = pad + block_h / 2;
1285            let _ = writeln!(
1286                &mut out,
1287                "  <text x=\"{xc}\" y=\"{yc}\" class=\"bc-op\" text-anchor=\"middle\" \
1288                 dominant-baseline=\"middle\">+</text>",
1289            );
1290        }
1291    }
1292
1293    // "=" then output block (full, no broadcast)
1294    let eq_x = pad + n_operands * block_total - block_gap / 2;
1295    let eq_y = pad + block_h / 2;
1296    let _ = writeln!(
1297        &mut out,
1298        "  <text x=\"{eq_x}\" y=\"{eq_y}\" class=\"bc-op\" text-anchor=\"middle\" \
1299         dominant-baseline=\"middle\">=</text>",
1300    );
1301    let out_x = pad + n_operands * block_total;
1302    let out_dims: HashMap<&str, usize> = output.iter().map(|(n, e)| (n.as_str(), *e)).collect();
1303    draw_broadcast_block(
1304        &mut out, out_x, pad, &row_name, row_extent, &col_name, col_extent, cell_px, "output",
1305        &out_dims,
1306    );
1307
1308    // Caption.
1309    let _ = writeln!(
1310        &mut out,
1311        "  <text x=\"{pad}\" y=\"{}\" class=\"bc-caption\">{} view — faded cells = broadcast copies (same value as the anchor row/column); ↻ marks a broadcast axis</text>",
1312        h - 10,
1313        cell_label,
1314    );
1315    out.push_str("</svg>\n");
1316    out
1317}

1319#[allow(clippy::too_many_arguments)]
1320fn draw_broadcast_block(
1321    out: &mut String,
1322    x0: usize,
1323    y0: usize,
1324    row_name: &str,
1325    row_extent: usize,
1326    col_name: &str,
1327    col_extent: usize,
1328    cell: usize,
1329    label: &str,
1330    op_dims: &HashMap<&str, usize>,
1331) {
1332    let has_row = op_dims.contains_key(row_name);
1333    let has_col = op_dims.contains_key(col_name);
1334    // VanderPlas-style: the operand's native data lives at coord 0 on
1335    // each missing axis; cells off that anchor on a missing axis are
1336    // broadcast COPIES — drawn with reduced opacity so the eye can read
1337    // "same value, tiled". Cells whose missing-axis coord is 0 (the
1338    // anchor row/column) stay solid.
1339    for r in 0..row_extent {
1340        for c in 0..col_extent {
1341            let row_is_copy = !has_row && r > 0;
1342            let col_is_copy = !has_col && c > 0;
1343            let is_copy = row_is_copy || col_is_copy;
1344            let opacity = if is_copy { "0.30" } else { "1.0" };
1345            let x = x0 + c * cell;
1346            let y = y0 + r * cell;
1347            let _ = writeln!(
1348                out,
1349                "  <rect class=\"bc-cell\" x=\"{x}\" y=\"{y}\" width=\"{cell}\" height=\"{cell}\" \
1350                 fill-opacity=\"{opacity}\"/>",
1351            );
1352            // Cell text: the value/coord at the native position. If row
1353            // is missing, the native row is 0; same for col. The TEXT
1354            // doesn't fade (so it's still readable), only the fill does.
1355            let text = if has_row && has_col {
1356                format!("{r},{c}")
1357            } else if has_col {
1358                format!("{c}")
1359            } else if has_row {
1360                format!("{r}")
1361            } else {
1362                String::from("·")
1363            };
1364            let _ = writeln!(
1365                out,
1366                "  <text class=\"bc-cell-coord\" x=\"{}\" y=\"{}\" text-anchor=\"middle\" \
1367                 dominant-baseline=\"middle\" fill-opacity=\"{}\">{text}</text>",
1368                x + cell / 2,
1369                y + cell / 2,
1370                if is_copy { "0.55" } else { "1.0" },
1371            );
1372        }
1373    }
1374    // Block label below.
1375    let label_y = y0 + row_extent * cell + 14;
1376    let cx = x0 + (col_extent * cell) / 2;
1377    let _ = writeln!(
1378        out,
1379        "  <text x=\"{cx}\" y=\"{label_y}\" class=\"bc-label\" text-anchor=\"middle\">{}</text>",
1380        esc(label),
1381    );
1382    // Axis annotations: row name on the left, col name on top. Use just
1383    // "↻" (no " bcast" suffix) when an axis is broadcast — the suffix
1384    // pushed the row label wide enough to spill leftward across the gap
1385    // into the previous block / the "+" glyph. The single glyph fits in
1386    // any reasonable gap, and the SVG caption explains what "↻" means.
1387    let row_label = if has_row { row_name } else { "↻" };
1388    let col_label = if has_col { col_name } else { "↻" };
1389    let _ = writeln!(
1390        out,
1391        "  <text x=\"{}\" y=\"{}\" class=\"bc-axis\" text-anchor=\"end\" \
1392         dominant-baseline=\"middle\">{row_label}</text>",
1393        x0 - 4,
1394        y0 + (row_extent * cell) / 2,
1395    );
1396    let _ = writeln!(
1397        out,
1398        "  <text x=\"{}\" y=\"{}\" class=\"bc-axis\" text-anchor=\"middle\">{col_label}</text>",
1399        x0 + (col_extent * cell) / 2,
1400        y0 - 4,
1401    );
1402}

1404fn chunk_in_occupancy(
1405    occ: &crate::brix::iteration_space::Occupancy,
1406    grid: &ChunkGrid,
1407    ri: usize,
1408    ci: usize,
1409) -> bool {
1410    let coord: Vec<usize> = if grid.shape.len() == 1 { vec![ci] } else { vec![ri, ci] };
1411    let mut probe = Vec::with_capacity(occ.sparse_axes.len());
1412    for axis in &occ.sparse_axes {
1413        let pos = match grid.names.iter().position(|n| n == axis) {
1414            Some(p) => p,
1415            None => return true,
1416        };
1417        if pos >= coord.len() {
1418            return true;
1419        }
1420        probe.push(coord[pos] as u32);
1421    }
1422    occ.as_coo_coords().contains(&probe)
1423}

1438/// One mid-IR node, fully rendered for side-by-side display: op kind,
1439/// label, brick payload kind, the element-sized `ChunkGrid` baked at
1440/// lowering time, and each incoming edge's per-axis [`AxisAccess`] map with
1441/// core dims.
1442///
1443/// Card ordering mirrors a post-order walk of the ArrayExpr DAG — every
1444/// producer appears before its consumers, matching the DAG drawn by
1445/// [`brix_expr_mermaid`].
1446pub struct NodeCard {
1447    pub label: String,
1448    pub op_short: String,
1449    pub grid: ChunkGrid,
1450    pub kind: ArrayPayloadKind,
1451    pub incoming: Vec<EdgeCard>,
1452}

1454/// One incoming edge into a `NodeCard` — producer name plus the
1455/// consumption metadata stamped on the corresponding mid-IR `Input`.
1456pub struct EdgeCard {
1457    pub producer_label: String,
1458    pub core_dims: Vec<String>,
1459    /// `Translated` overrides on this edge (slice offset, gather lookup,
1460    /// zipped followers). `Composed` axes appear in [`Self::core_dims`].
1461    pub overrides: HashMap<AxisName, TranslatedAccess>,
1462    /// Axes the consumer iterates that the producer doesn't carry on its
1463    /// chunk grid. Pre-computed so the per-card view shows broadcast
1464    /// edges without re-deriving from `IterationSpace::names`.
1465    pub broadcast_axes: Vec<String>,
1466    /// The `NamedLayout` from `access_layout(producer_grid, iter_space)` —
1467    /// per-axis `(name, shape, stride)`. Stride 0 entries are exactly the
1468    /// broadcast/transposed-broadcast axes: as the consumer iterates that
1469    /// axis, the producer's chunk coord stays put. Populated when a
1470    /// `PhysicalPlan` is supplied to `brix_node_cards`.
1471    pub access_layout: Option<Vec<AccessAxis>>,
1472    /// Per-run lookup table for vindex edges — the materialized form of
1473    /// the `Zip{i=Lookup(...), j=Lookup(...)}` access entry: each entry
1474    /// pairs the leader axis's iter index with the source chunk and the
1475    /// in-chunk positions on each consumed axis. `None` for non-vindex
1476    /// edges.
1477    pub vindex_lookup: Option<VindexLookup>,
1478}

1480/// One row in an edge's access layout — `(axis_name, iter_shape, producer_stride)`.
1481/// Stride 0 means broadcast on that axis.
1482pub struct AccessAxis {
1483    pub name: String,
1484    pub shape: usize,
1485    pub stride: usize,
1486}
1487
1488/// Materialized form of the `VectorizedSelection` the lowering built for
1489/// a vindex consumer's input edge. One `runs` entry per produced-axis
1490/// chunk; each run lists the source chunk coord (one entry per consumed
1491/// axis) and the in-chunk positions picked from that chunk.
1492pub struct VindexLookup {
1493    /// Canonical (sorted) name order of the axes this vindex consumes —
1494    /// drives the column order in [`VindexRun::source_chunk`] /
1495    /// [`VindexRun::positions`].
1496    pub consumed_axes: Vec<String>,
1497    pub runs: Vec<VindexRun>,
1498}
1499
1500pub struct VindexRun {
1501    /// Source chunk coord along each consumed axis, in
1502    /// `consumed_axes` order.
1503    pub source_chunk: Vec<usize>,
1504    /// Per consumed axis (outer index), the list of in-chunk positions
1505    /// picked from that source chunk by this run (inner index = position
1506    /// within the produced-axis chunk).
1507    pub positions: Vec<Vec<usize>>,
1508}

1510/// Walk the ArrayExpr DAG post-order and produce one [`NodeCard`] per
1511/// unique node. The per-node `ChunkGrid` and access layouts are read
1512/// back from the supplied [`Lowered`] / [`PhysicalPlan`] — no chunk-grid
1513/// math is recomputed here.
1514pub fn brix_node_cards(
1515    arr: &impl ArrayLike,
1516    lowered: &Lowered,
1517    plan: &PhysicalPlan,
1518) -> Vec<NodeCard> {
1519    let mut cards: Vec<NodeCard> = Vec::new();
1520    let mut visited: HashMap<*const ArrayExpr, usize> = HashMap::new();
1521    walk_cards(arr.expr(), &lowered.expr, &lowered.grids, &mut visited, &mut cards);
1522
1523    // Merge `access_layout` for each (consumer, producer) edge by walking
1524    // the schedule's Apply steps. The planner has already built the
1525    // NamedLayout via `access_layout(producer_grid, iter_space)`; we just
1526    // surface it on the card for display.
1527    let apply_inputs = collect_apply_inputs(&plan.schedule);
1528    for card in &mut cards {
1529        let Some(inputs) = apply_inputs.get(card.label.as_str()) else {
1530            continue;
1531        };
1532        for edge in &mut card.incoming {
1533            let Some(input) = inputs.iter().find(|i| i.source.0 == edge.producer_label) else {
1534                continue;
1535            };
1536            edge.access_layout = Some(named_layout_axes(input.access.affine.layout()));
1537        }
1538    }
1539    cards
1540}

1542/// Walk the schedule and return `consumer_name -> Vec<&InputEdge>`. Used
1543/// to look up per-edge access layouts the planner already computed via
1544/// `access_layout`.
1545fn collect_apply_inputs(schedule: &Schedule) -> HashMap<String, Vec<&InputEdge>> {
1546    let mut out: HashMap<String, Vec<&InputEdge>> = HashMap::new();
1547    walk_schedule(schedule, &mut out);
1548    out
1549}
1550
1551fn walk_schedule<'a>(s: &'a Schedule, out: &mut HashMap<String, Vec<&'a InputEdge>>) {
1552    match s {
1553        Schedule::Apply { node, inputs, .. } => {
1554            out.entry(node.0.clone()).or_default().extend(inputs.iter());
1555        }
1556        Schedule::Domain { body, .. } => walk_schedule(body, out),
1557        Schedule::Loop { body, .. } | Schedule::Zip { body, .. } => {
1558            for s in body {
1559                walk_schedule(s, out);
1560            }
1561        }
1562        Schedule::Sequence(body) => {
1563            for s in body {
1564                walk_schedule(s, out);
1565            }
1566        }
1567        Schedule::Init { .. }
1568        | Schedule::Finalize { .. }
1569        | Schedule::InitSink { .. }
1570        | Schedule::WriteSink { .. }
1571        | Schedule::FinalizeSink { .. } => {}
1572    }
1573}

1575fn selection_to_lookup(sel: crate::array::chunk_grid::vindex::VectorizedSelection) -> VindexLookup {
1576    let n_runs = sel.per_axis.first().map(|v| v.len()).unwrap_or(0);
1577    let runs: Vec<VindexRun> = (0..n_runs)
1578        .map(|r| {
1579            let source_chunk: Vec<usize> =
1580                sel.per_axis.iter().map(|axis_sels| axis_sels[r].source_chunk).collect();
1581            let positions: Vec<Vec<usize>> =
1582                sel.per_axis.iter().map(|axis_sels| axis_sels[r].positions.clone()).collect();
1583            VindexRun { source_chunk, positions }
1584        })
1585        .collect();
1586    VindexLookup { consumed_axes: sel.consumed_axes.iter().map(Into::into).collect(), runs }
1587}

1589fn named_layout_axes(nl: &crate::brix::layout::NamedLayout) -> Vec<AccessAxis> {
1590    let shapes = nl.layout.shape.flat();
1591    let strides = nl.layout.stride.flat();
1592    nl.names
1593        .iter()
1594        .enumerate()
1595        .map(|(i, n)| AccessAxis {
1596            name: n.into(),
1597            shape: *shapes.get(i).unwrap_or(&0),
1598            stride: *strides.get(i).unwrap_or(&0),
1599        })
1600        .collect()
1601}

1603fn walk_cards(
1604    a_node: &Arc<ArrayExpr>,
1605    e_node: &Arc<Expr>,
1606    grids: &HashMap<ExprId, ChunkGrid>,
1607    visited: &mut HashMap<*const ArrayExpr, usize>,
1608    out: &mut Vec<NodeCard>,
1609) {
1610    let key = Arc::as_ptr(a_node);
1611    if visited.contains_key(&key) {
1612        return;
1613    }
1614    debug_assert_eq!(
1615        a_node.inputs.len(),
1616        e_node.inputs.len(),
1617        "ArrayExpr and lowered Expr must mirror each other (render walkers zip their inputs)"
1618    );
1619    for (a_input, e_input) in a_node.inputs.iter().zip(e_node.inputs.iter()) {
1620        walk_cards(a_input, &e_input.expr, grids, visited, out);
1621    }
1622    let grid = grids
1623        .get(&e_node.id)
1624        .cloned()
1625        .expect("lowered.grids must carry every node in the lowered DAG");
1626    let label = a_node.label.clone().unwrap_or_else(|| format!("node@{:p}", Arc::as_ptr(a_node)));
1627    let op_short = array_op_short(&a_node.op);
1628    // Build the per-run lookup table for a vindex consumer, once. Attached
1629    // to the (single) incoming edge below.
1630    let mut vindex_lookup: Option<VindexLookup> = match &a_node.op {
1631        ArrayOp::Vindex(indexer) => {
1632            e_node.inputs.first().and_then(|i| grids.get(&i.expr.id)).and_then(|input_grid| {
1633                crate::array::chunk_grid::vindex::output_grid_and_selection(
1634                    indexer.produced_axis.clone(),
1635                    &indexer.indices,
1636                    input_grid,
1637                )
1638                .ok()
1639                .map(|(_, sel)| selection_to_lookup(sel))
1640            })
1641        }
1642        _ => None,
1643    };
1644
1645    let mut incoming: Vec<EdgeCard> = Vec::with_capacity(e_node.inputs.len());
1646    for (e_in, a_in) in e_node.inputs.iter().zip(a_node.inputs.iter()) {
1647        let producer_label = a_in.label.clone().unwrap_or_else(|| array_op_short(&a_in.op));
1648        let producer_dims: Vec<String> = a_in.layout.0.names.iter().map(Into::into).collect();
1649        let consumer_dims: Vec<String> = a_node.layout.0.names.iter().map(Into::into).collect();
1650        let broadcast_axes: Vec<String> = consumer_dims
1651            .iter()
1652            .filter(|c| !producer_dims.contains(c) && !e_in.access.contains_key(c.as_str()))
1653            .cloned()
1654            .collect();
1655        // Vindex always has exactly one input edge; attach the lookup to it.
1656        let lookup = vindex_lookup.take();
1657        incoming.push(EdgeCard {
1658            producer_label,
1659            core_dims: fold_axes(&e_in.access),
1660            overrides: translated_overrides(&e_in.access),
1661            broadcast_axes,
1662            access_layout: None,
1663            vindex_lookup: lookup,
1664        });
1665    }
1666    out.push(NodeCard { label, op_short, grid, kind: a_node.brick_kind, incoming });
1667    visited.insert(key, out.len() - 1);
1668}

1954fn collect_sources(
1955    a_node: &Arc<ArrayExpr>,
1956    e_node: &Arc<Expr>,
1957    grids: &HashMap<ExprId, ChunkGrid>,
1958    visited: &mut HashMap<*const ArrayExpr, ()>,
1959    out: &mut Vec<(String, ChunkGrid, ArrayPayloadKind)>,

1992// ===========================================================================
1993// Chunk renumbering / selection — for Slice / Oindex / Vindex / Reduce.
1994//
1995// Each of those ops carries a structural "which input chunks become which
1996// output chunks?" story that's invisible in the DAG node + edge tooltip
1997// alone. This emits one collapsible section per such node showing the
1998// input grid (filtered-out chunks at low opacity) → output grid
1999// (renumbered, with cell labels back to source chunks).
2000// ===========================================================================
2001
2002/// Per-axis selection record used by the renumbering renderer. All
2003/// chunk-space data — derived directly from the lowered brix `Expr`
2004/// rather than re-walking the array layer.
2005#[derive(Debug, Clone)]
2006enum AxisSelection {
2007    /// Axis untouched — every input chunk passes through unchanged.
2008    PassThrough,
2009    /// Slice: output chunk `c` ⇐ input chunk `c + offset`. `n_out` is
2010    /// the resulting output chunk count.
2011    Slice { offset: usize, n_out: usize },
2012    /// Outer / vectorized indexer: explicit list of source chunk
2013    /// indices, one per output chunk.
2014    Pick(Vec<usize>),
2015    /// Reduce folds input chunks into output chunks along this axis,
2016    /// shaped by the per-axis array-layer [`Frame`] on the edge:
2017    /// - `WholeAxis` — every input chunk contributes to one output
2018    ///   position (the axis disappears unless `keepdims`).
2019    /// - `Window { size, stride }` with `size == stride` (coarsen) —
2020    ///   `size` consecutive input chunks fold into each output chunk.
2021    Reduce(Frame),
2022}
2023
2024struct RenumberSection {
2025    label: String,
2026    op_short: String,
2027    description: String,
2028    input_grid: IterationSpace,
2029    output_grid: IterationSpace,
2030    /// One entry per axis name in `input_grid.names()`. Length matches.
2031    selection: Vec<AxisSelection>,
2032    /// When true, parallel `Pick` lists on multiple axes are read in
2033    /// lockstep — the i-th element of each pick is one `(c0, c1, ...)`
2034    /// chunk tuple that fires. Used by vindex's co-iterated gather. When
2035    /// false, axis picks are independent and the cell mask is their
2036    /// outer product (Slice/Oindex).
2037    zipped: bool,
2038}

2075fn collect_renumber_sections(
2076    a_node: &Arc<ArrayExpr>,
2077    e_node: &Arc<Expr>,
2078    grids: &HashMap<ExprId, ChunkGrid>,
2079    visited: &mut HashSet<*const ArrayExpr>,
2080    out: &mut Vec<RenumberSection>,
2081) {
2082    let key = Arc::as_ptr(a_node);
2083    if !visited.insert(key) {
2084        return;
2085    }
2086    debug_assert_eq!(
2087        a_node.inputs.len(),
2088        e_node.inputs.len(),
2089        "ArrayExpr and lowered Expr must mirror each other (render walkers zip their inputs)"
2090    );
2091    for (a_in, e_in) in a_node.inputs.iter().zip(e_node.inputs.iter()) {
2092        collect_renumber_sections(a_in, &e_in.expr, grids, visited, out);
2093    }
2094
2095    let Some(input_chunk_grid) = e_node.inputs.first().and_then(|i| grids.get(&i.expr.id)) else {
2096        return;
2097    };
2098    let output_chunk_grid =
2099        grids.get(&e_node.id).expect("lowered.grids must carry every node in the lowered DAG");
2100    let label = a_node.label.clone().unwrap_or_else(|| array_op_short(&a_node.op));
2101    let input_brix = input_chunk_grid.to_iteration_space();
2102    let output_brix = output_chunk_grid.to_iteration_space();
2103    let section = match &a_node.op {
2104        ArrayOp::Slice(slicers) => {
2105            let Ok((_out_cg, offsets)) = crate::array::chunk_grid::select::output_grid_and_offsets(
2106                input_chunk_grid,
2107                slicers,
2108            ) else {
2109                return;
2110            };
2111            let mut selection: Vec<AxisSelection> = Vec::with_capacity(input_brix.names().len());
2112            let out_counts = output_brix.max_extents();
2113            for (d, _name) in input_brix.names().iter().enumerate() {
2114                let n_out = out_counts[d];
2115                let offset = offsets.get(d).copied().unwrap_or(0);
2116                if offset == 0 && n_out == input_chunk_grid.chunks[d].n_chunks() {
2117                    selection.push(AxisSelection::PassThrough);
2118                } else {
2119                    selection.push(AxisSelection::Slice { offset, n_out });
2120                }
2121            }
2122            let desc = format!(
2123                "kept {} of {} input chunk{} along {{{}}}",
2124                output_brix.total_bricks(),
2125                input_brix.total_bricks(),
2126                if input_brix.total_bricks() == 1 { "" } else { "s" },
2127                slicers.keys().cloned().collect::<Vec<_>>().join(", "),
2128            );
2129            Some(RenumberSection {
2130                label,
2131                op_short: array_op_short(&a_node.op),
2132                description: desc,
2133                input_grid: input_brix,
2134                output_grid: output_brix,
2135                selection,
2136                zipped: false,
2137            })
2138        }
2139        ArrayOp::Oindex(indexers) => {
2140            let Ok((_out_cg, runs_per_axis)) =
2141                crate::array::chunk_grid::oindex::output_grid_and_selection(
2142                    input_chunk_grid,
2143                    indexers,
2144                )
2145            else {
2146                return;
2147            };
2148            let mut selection: Vec<AxisSelection> = Vec::with_capacity(input_brix.names().len());
2149            for axis_runs in &runs_per_axis {
2150                match axis_runs {
2151                    None => selection.push(AxisSelection::PassThrough),
2152                    Some(rs) => selection.push(AxisSelection::Pick(
2153                        rs.runs.iter().map(|r| r.source_chunk).collect(),
2154                    )),
2155                }
2156            }
2157            let desc = format!(
2158                "outer-product gather along {{{}}}",
2159                indexers.keys().cloned().collect::<Vec<_>>().join(", "),
2160            );
2161            Some(RenumberSection {
2162                label,
2163                op_short: array_op_short(&a_node.op),
2164                description: desc,
2165                input_grid: input_brix,
2166                output_grid: output_brix,
2167                selection,
2168                zipped: false,
2169            })
2170        }
2171        ArrayOp::Vindex(indexer) => {
2172            let Ok((_out_cg, sel)) = crate::array::chunk_grid::vindex::output_grid_and_selection(
2173                indexer.produced_axis.clone(),
2174                &indexer.indices,
2175                input_chunk_grid,
2176            ) else {
2177                return;
2178            };
2179            // Each run picks one (chunk per consumed axis) tuple. For
2180            // each consumed axis, build the picked list across runs.
2181            let n_runs = sel.per_axis.first().map(|v| v.len()).unwrap_or(0);
2182            let mut selection: Vec<AxisSelection> = Vec::with_capacity(input_brix.names().len());
2183            for (d, name) in input_brix.names().iter().enumerate() {
2184                if let Some(ax_pos) = sel.consumed_axes.iter().position(|c| c == name) {
2185                    let picks: Vec<usize> =
2186                        (0..n_runs).map(|r| sel.per_axis[ax_pos][r].source_chunk).collect();
2187                    selection.push(AxisSelection::Pick(picks));
2188                } else {
2189                    let _ = d;
2190                    selection.push(AxisSelection::PassThrough);
2191                }
2192            }
2193            let desc = format!(
2194                "co-iterated gather → produced axis `{}` ({} runs)",
2195                indexer.produced_axis, n_runs,
2196            );
2197            Some(RenumberSection {
2198                label,
2199                op_short: array_op_short(&a_node.op),
2200                description: desc,
2201                input_grid: input_brix,
2202                output_grid: output_brix,
2203                selection,
2204                zipped: true,
2205            })
2206        }
2207        ArrayOp::Reduce { view, .. } => {
2208            let selection: Vec<AxisSelection> = input_brix
2209                .names()
2210                .iter()
2211                .map(|name| match view.get(name) {
2212                    None => AxisSelection::PassThrough,
2213                    Some(frame) => AxisSelection::Reduce(frame.clone()),
2214                })
2215                .collect();
2216            let parts: Vec<String> = view
2217                .iter()
2218                .map(|(axis, frame)| match frame {
2219                    Frame::WholeAxis { .. } => format!("`{axis}` (all input chunks → 1)"),
2220                    Frame::Window { size, stride } if size == stride => {
2221                        format!("`{axis}` (coarsen, window {size})")
2222                    }

2226                    _ => format!("`{axis}` (variable framing)"),

2239        }
2240        _ => None,
2241    };
2242    if let Some(s) = section {
2243        out.push(s);

2301/// Per-run picks across all input axes (None on PassThrough axes —
2302/// any chunk on that axis matches). Empty when not zipped.
2303fn zipped_runs(ndim: usize, selection: &[AxisSelection], zipped: bool) -> Vec<Vec<Option<usize>>> {
2304    if !zipped {
2305        return Vec::new();
2306    }
2307    let picks_per_axis: Vec<Option<&Vec<usize>>> = (0..ndim)
2308        .map(|d| match selection.get(d) {
2309            Some(AxisSelection::Pick(p)) => Some(p),
2310            _ => None,
2311        })
2312        .collect();
2313    let n_runs = picks_per_axis.iter().flatten().map(|p| p.len()).max().unwrap_or(0);
2314    (0..n_runs).map(|r| picks_per_axis.iter().map(|p| p.map(|v| v[r])).collect()).collect()
2315}

2317fn input_rn_grid(
2318    input: &IterationSpace,
2319    selection: &[AxisSelection],
2320    zipped_runs: &[Vec<Option<usize>>],
2321    zipped: bool,
2322) -> RnGrid {
2323    let names = input.names();
2324    let counts = input.max_extents();
2325    let axes: Vec<(String, usize)> =
2326        names.iter().zip(counts.iter()).map(|(n, c)| (n.into(), *c)).collect();
2327    let selected_per_axis: Vec<std::collections::BTreeSet<usize>> = (0..names.len())
2328        .map(|d| match selection.get(d).unwrap_or(&AxisSelection::PassThrough) {
2329            AxisSelection::PassThrough | AxisSelection::Reduce(_) => (0..counts[d]).collect(),
2330            AxisSelection::Slice { offset, n_out } => (*offset..(*offset + *n_out)).collect(),
2331            AxisSelection::Pick(picks) => picks.iter().copied().collect(),
2332        })
2333        .collect();
2334    let (rows, cols) =
2335        if names.len() == 1 { (1, counts[0].max(1)) } else { (counts[0].max(1), counts[1].max(1)) };
2336    let cells = (0..rows)
2337        .map(|r| {
2338            (0..cols)
2339                .map(|c| {
2340                    let (a0, a1) = if names.len() == 1 { (c, c) } else { (r, c) };
2341                    let present = if zipped {
2342                        zipped_runs.iter().any(|run| {
2343                            let ok0 = run.first().and_then(|x| *x).is_none_or(|v| v == a0);
2344                            let ok1 = if names.len() == 1 {
2345                                true
2346                            } else {
2347                                run.get(1).and_then(|x| *x).is_none_or(|v| v == a1)
2348                            };
2349                            ok0 && ok1
2350                        })
2351                    } else if names.len() == 1 {
2352                        selected_per_axis[0].contains(&c)
2353                    } else {
2354                        selected_per_axis[0].contains(&r) && selected_per_axis[1].contains(&c)
2355                    };
2356                    let text = if names.len() == 1 { format!("{c}") } else { format!("{r},{c}") };
2357                    RnCell { text, present }
2358                })
2359                .collect()
2360        })
2361        .collect();
2362    RnGrid { axes, cells }
2363}

2365fn output_rn_grid(
2366    output: &IterationSpace,
2367    input: &IterationSpace,
2368    selection: &[AxisSelection],
2369    zipped_runs: &[Vec<Option<usize>>],
2370    zipped: bool,
2371) -> RnGrid {
2372    let names = output.names();
2373    let counts = output.max_extents();
2374    let in_names = input.names();
2375    let in_counts = input.max_extents();
2376    let axes: Vec<(String, usize)> =
2377        names.iter().zip(counts.iter()).map(|(n, c)| (n.into(), *c)).collect();
2378    // Per output axis: text label for each cell along that axis.
2379    let labels_per_axis: Vec<Vec<String>> = (0..names.len())
2380        .map(|d| {
2381            let count = counts[d];
2382            if zipped && !in_names.iter().any(|n| n == &names[d]) {
2383                // Produced axis from a co-iterated gather: each cell
2384                // shows the (axis_0, axis_1, ...) tuple of chunk coords.
2385                return (0..count)
2386                    .map(|i| {
2387                        zipped_runs[i]
2388                            .iter()
2389                            .map(|x| x.map(|v| v.to_string()).unwrap_or_else(|| "*".into()))
2390                            .collect::<Vec<_>>()
2391                            .join(",")
2392                    })
2393                    .collect();
2394            }
2395            let in_pos = in_names.iter().position(|n| n == &names[d]).unwrap_or(0);
2396            match selection.get(in_pos).unwrap_or(&AxisSelection::PassThrough) {
2397                AxisSelection::PassThrough | AxisSelection::Reduce(_) => {
2398                    // Coarsen keeps the axis on the output with a smaller
2399                    // count; label each output cell with the input chunk
2400                    // range it folds (window = ceil(n_in / n_out)).
2401                    if let Some(AxisSelection::Reduce(Frame::Window { size, stride })) =
2402                        selection.get(in_pos)
2403                        && size == stride
2404                    {
2405                        let n_in = in_counts[in_pos];
2406                        let w = n_in.div_ceil(count);
2407                        return (0..count)
2408                            .map(|i| {
2409                                let lo = i * w;
2410                                let hi = ((i + 1) * w).min(n_in);
2411                                if hi - lo == 1 { lo.to_string() } else { format!("{lo}..{hi}") }
2412                            })
2413                            .collect();
2414                    }
2415                    (0..count).map(|i| i.to_string()).collect()
2416                }
2417                AxisSelection::Slice { offset, .. } => {
2418                    (0..count).map(|i| (offset + i).to_string()).collect()
2419                }
2420                AxisSelection::Pick(picks) => picks.iter().map(|p| p.to_string()).collect(),
2421            }
2422        })
2423        .collect();
2424    let (rows, cols) = if names.is_empty() {
2425        (1, 1)
2426    } else if names.len() == 1 {
2427        (1, counts[0].max(1))
2428    } else {
2429        (counts[0].max(1), counts[1].max(1))
2430    };
2431    let cells = (0..rows)
2432        .map(|r| {
2433            (0..cols)
2434                .map(|c| {
2435                    let text = if names.is_empty() {
2436                        String::new()
2437                    } else if names.len() == 1 {
2438                        labels_per_axis[0][c].clone()
2439                    } else {
2440                        format!("{},{}", labels_per_axis[0][r], labels_per_axis[1][c])
2441                    };
2442                    RnCell { text, present: true }
2443                })
2444                .collect()
2445        })
2446        .collect();
2447    RnGrid { axes, cells }
2448}
