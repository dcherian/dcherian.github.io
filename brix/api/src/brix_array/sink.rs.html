15/// Array-level sink descriptor, lowered into a concrete [`brix::Sink`] once the grid is chosen
16/// (mirrors `OpaqueSource` → `LoweredOpaque`). The materialized result is
17/// returned by `execute` (keyed by the sink node's name), not held here.
18pub trait ArraySink: Debug + Send + Sync {
19    /// Check this sink can accept an output of `dtype`. Runs at
20    /// [`crate::expr::ArrayExpr::sink`] construction — the earliest point
21    /// the output dtype is known — so `lower`/`init` downstream treat
22    /// compatibility as an `expect`-backed invariant.
23    fn validate(&self, dtype: &DType) -> ArrayResult<()>;
24
25    fn lower(&self, grid: ChunkGrid, dtype: DType) -> LoweringResult<Arc<dyn brix::Sink>>;
26}

28/// In-memory output descriptor — just the fill.
29// ANCHOR: in_memory_sink
30#[derive(Debug)]
31pub struct InMemorySink {
32    fill: Option<ScalarValue>,
33}
34// ANCHOR_END: in_memory_sink
35
36impl InMemorySink {
37    /// `fill`: pre-fill for absent / unwritten regions (sparse outputs);
38    /// `None` selects the dtype zero (primitive) or `ExtensionDType::empty`
39    /// (extension). See [`crate::ArrayLike::sink_fill`].
40    pub fn new(fill: Option<ScalarValue>) -> Self {
41        Self { fill }
42    }
43}
44
45impl ArraySink for InMemorySink {
46    fn validate(&self, dtype: &DType) -> ArrayResult<()> {
47        match &self.fill {
48            None => Ok(()),
49            Some(fill) if fill.dtype() == *dtype => Ok(()),
50            Some(fill) => {
51                Err(ArrayError::SinkFillDTypeMismatch { fill: fill.dtype(), sink: dtype.clone() })
52            }
53        }
54    }
55
56    fn lower(&self, grid: ChunkGrid, dtype: DType) -> LoweringResult<Arc<dyn brix::Sink>> {
57        // No allocation here — the buffer is allocated at `InitSink` time
58        // (the array layer owns the element shape).
59        Ok(Arc::new(LoweredInMemorySink {
60            grid,
61            dtype,
62            fill: self.fill.clone(),
63            buffer: Mutex::new(None),
64        }))
65    }
66}

68enum SinkBuffer {
69    Dense(DenseArray),
70    Extension(Box<dyn ArrayData>),
71}

73// Concrete Sink with ChunkGrid for population.
74pub struct LoweredInMemorySink {
75    grid: ChunkGrid,
76    dtype: DType,
77    fill: Option<ScalarValue>,
78    buffer: Mutex<Option<SinkBuffer>>,
79}
80
81impl Sink for LoweredInMemorySink {
82    fn init(&self) -> BrixResult<()> {
83        let m = "fill dtype validated at ArrayExpr::sink";
84        let buffer = match &self.dtype {
85            DType::Primitive(p) => {
86                let array = dispatch_dtype!(*p, T => {
87                    let prefill: T = match &self.fill {
88                        Some(v) => v.get::<T>().expect(m),
89                        None => T::zero(),
90                    };
91                    T::from_array(Array::from_elem(IxDyn(&self.grid.shape), prefill))
92                });
93                SinkBuffer::Dense(array)
94            }
95            DType::Extension(e) => SinkBuffer::Extension(match &self.fill {
96                Some(ScalarValue::Extension(s)) => e.full(s.as_ref(), &self.grid.shape),
97                None => e.empty(&self.grid.shape),
98                Some(ScalarValue::Primitive(_)) => unreachable!("{}", m),
99            }),
100        };
101        *self.buffer.lock().unwrap() = Some(buffer);
102        Ok(())
103    }
104
105    fn write_brick(&self, brick: Brick) -> BrixResult<()> {
106        let payload = ArrayPayload::from_payload(brick.data.as_ref());
107        let ranges = self.grid.slice_ranges(&brick.coord).expect("sink coord must match grid axes");
108        let mut guard = self.buffer.lock().unwrap();
109        let buffer: &mut dyn ArrayData =
110            match guard.as_mut().expect("invariant: InitSink runs before WriteSink") {
111                SinkBuffer::Dense(array) => array,
112                SinkBuffer::Extension(buffer) => buffer.as_mut(),
113            };
114        buffer.write_range(&ranges, payload)
115    }
116
117    fn finalize(&self) -> BrixResult<Option<Arc<dyn BrickPayload>>> {
118        // Always `Some`: InitSink allocated the buffer before FinalizeSink runs.
119        // (The `Option` return is for sinks that commit externally, e.g. Zarr.)
120        let buffer = self
121            .buffer
122            .lock()
123            .unwrap()
124            .take()
125            .expect("invariant: InitSink runs before FinalizeSink");
126        let payload: Arc<dyn BrickPayload> = match buffer {
127            SinkBuffer::Dense(array) => Arc::new(ArrayPayload::Dense(array)),
128            SinkBuffer::Extension(buffer) => Arc::new(ArrayPayload::Extension(Arc::from(buffer))),
129        };
130        Ok(Some(payload))
131    }
132}
