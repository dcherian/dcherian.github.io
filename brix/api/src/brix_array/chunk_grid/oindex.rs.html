1//! Chunk-grid algebra for orthogonal indexing (`BlockwiseOp::Gather`).
2//!
3//! Plan-time math: given an input `ChunkGrid` and per-axis 1-D integer
4//! index arrays, partition each axis into runs by source chunk and build
5//! the output `ChunkGrid`. The brick-level kernel that gathers one
6//! output brick lives in [`crate::GatherKernel`].
7//!
8//! Output chunking is per-source-chunk-touched: each output chunk
9//! corresponds to one source chunk along the indexed axis; the chunk
10//! holds the in-chunk positions hitting that source chunk. This is the
11//! sparse, vindex-style runs model — only the source chunks the indices
12//! actually touch are read at execute time.
13
14use std::collections::HashMap;
15
16use crate::chunk_grid::{AxisChunks, AxisSelection, ChunkGrid, ChunkGridError, ChunkGridResult};
17use brix::AxisName;
18
19/// Per-axis partition of an oindex index array into runs by source chunk.
20/// Each run corresponds to one output chunk along this axis.
21#[derive(Clone, Debug, PartialEq, Eq)]
22pub struct OuterSelection {
23    pub runs: Vec<AxisSelection>,
24}
25
26impl OuterSelection {
27    /// Build runs from a sorted index array against the source axis's
28    /// chunking. Hard errors:
29    /// - any index is out of bounds (`>= source.extent()`)
30    /// - indices are not sorted by source chunk
31    /// - source axis is `AxisChunks::Variable` (mirror the select rule)
32    pub fn from_sorted_indices(indices: &[usize], source: &AxisChunks) -> ChunkGridResult<Self> {
33        if matches!(source, AxisChunks::Variable(_)) {
34            return Err(ChunkGridError::Unsupported(
35                "Oindex on AxisChunks::Variable axes not supported".into(),
36            ));
37        }
38        // Walk indices, group by source chunk, assert non-decreasing.
39        let mut runs: Vec<AxisSelection> = Vec::new();
40        let mut last_chunk: Option<usize> = None;
41        for &elt in indices {
42            let (chunk, pos) = source.locate(elt)?;
43            match last_chunk {
44                Some(c) if chunk < c => {
45                    return Err(ChunkGridError::Unsupported(format!(
46                        "Oindex indices not sorted by source chunk: index {elt} maps to \
47                         source chunk {chunk} but previous index hit chunk {c}"
48                    )));
49                }
50                Some(c) if chunk == c => {
51                    runs.last_mut().unwrap().positions.push(pos);
52                }
53                _ => {
54                    runs.push(AxisSelection { source_chunk: chunk, positions: vec![pos] });
55                    last_chunk = Some(chunk);
56                }
57            }
58        }
59        Ok(OuterSelection { runs })
60    }
61
62    /// Total number of indexed positions
63    pub fn len(&self) -> usize {
64        self.runs.iter().map(|r| r.positions.len()).sum()
65    }
66
67    pub fn is_empty(&self) -> bool {
68        self.len() == 0
69    }
70}
71
72/// Compute the output `ChunkGrid` and per-axis runs for an orthogonal-gather
73/// node. Axes not in `indexers` are passed through unchanged (None
74/// entry in the returned runs vector); for each indexed axis the runs
75/// drive both the output chunk shape and the source-chunk lookup at
76/// execute time.
77pub fn output_grid_and_selection(
78    input_grid: &ChunkGrid,
79    indexers: &HashMap<AxisName, Vec<usize>>,
80) -> ChunkGridResult<(ChunkGrid, Vec<Option<OuterSelection>>)> {
81    let n_axes = input_grid.ndim();
82    let mut output_shape = Vec::with_capacity(n_axes);
83    let mut output_chunks: Vec<AxisChunks> = Vec::with_capacity(n_axes);
84    let mut runs_per_axis: Vec<Option<OuterSelection>> = Vec::with_capacity(n_axes);
85
86    for (d, name) in input_grid.names.iter().enumerate() {
87        let input_axis = &input_grid.chunks[d];
88        match indexers.get(name) {
89            None => {
90                output_shape.push(input_grid.shape[d]);
91                output_chunks.push(input_axis.clone());
92                runs_per_axis.push(None);
93            }
94            Some(indices) => {
95                let runs = OuterSelection::from_sorted_indices(indices, input_axis)?;
96                output_shape.push(indices.len());
97                // One output chunk per run; size = run.positions.len(). Sizes are
98                // independent per run, so encode as Variable.
99                let sizes: Vec<usize> = runs.runs.iter().map(|r| r.positions.len()).collect();
100                output_chunks.push(AxisChunks::Variable(sizes));
101                runs_per_axis.push(Some(runs));
102            }
103        }
104    }
105
106    Ok((
107        ChunkGrid {
108            shape: output_shape,
109            chunks: output_chunks,
110            names: input_grid.names.clone(),
111            occupancy: None,
112        },
113        runs_per_axis,
114    ))
115}

117#[cfg(test)]
118mod tests {
119    use super::*;
120
121    #[test]
122    fn axisruns_from_sorted_uniform() {
123        // Source: shape 20, chunks of 5 (4 chunks).
124        // Indices [0, 3, 6, 7, 12] → chunks [0, 0, 1, 1, 2].
125        // Expected runs: chunk 0 with [0, 3], chunk 1 with [1, 2], chunk 2 with [2].
126        let source = AxisChunks::Uniform { size: 5, n_chunks: 4 };
127        let runs = OuterSelection::from_sorted_indices(&[0, 3, 6, 7, 12], &source).unwrap();
128        assert_eq!(
129            runs.runs,
130            vec![
131                AxisSelection { source_chunk: 0, positions: vec![0, 3] },
132                AxisSelection { source_chunk: 1, positions: vec![1, 2] },
133                AxisSelection { source_chunk: 2, positions: vec![2] },
134            ]
135        );
136    }
137
138    #[test]
139    fn axisruns_unsorted_errors() {
140        let source = AxisChunks::Uniform { size: 5, n_chunks: 4 };
141        // 7 (chunk 1) then 3 (chunk 0) — out of order across chunks.
142        let err = OuterSelection::from_sorted_indices(&[7, 3], &source).unwrap_err();
143        assert!(matches!(err, ChunkGridError::Unsupported(ref m) if m.contains("not sorted")));
144    }
145
146    #[test]
147    fn axisruns_within_chunk_unordered_allowed() {
148        // Within-chunk reordering (e.g. [3, 0] in chunk 0) is allowed —
149        // we only check non-decreasing source-chunk index.
150        let source = AxisChunks::Uniform { size: 5, n_chunks: 4 };
151        let runs = OuterSelection::from_sorted_indices(&[3, 0, 6], &source).unwrap();
152        assert_eq!(runs.runs[0].positions, vec![3, 0]);
153        assert_eq!(runs.runs[1].positions, vec![1]);
154    }
155
156    #[test]
157    fn axisruns_oob_errors() {
158        let source = AxisChunks::Uniform { size: 5, n_chunks: 4 };
159        let err = OuterSelection::from_sorted_indices(&[0, 99], &source).unwrap_err();
160        assert!(matches!(err, ChunkGridError::Unsupported(ref m) if m.contains("out of bounds")));
161    }
162
163    #[test]
164    fn axisruns_variable_errors() {
165        let source = AxisChunks::Variable(vec![2, 4, 1, 3]);
166        let err = OuterSelection::from_sorted_indices(&[0, 5], &source).unwrap_err();
167        assert!(matches!(err, ChunkGridError::Unsupported(ref m) if m.contains("Variable")));
168    }
169
170    #[test]
171    fn output_grid_passthrough_unindexed_axis() {
172        let input_grid = ChunkGrid {
173            shape: vec![20, 6],
174            chunks: vec![
175                AxisChunks::Uniform { size: 5, n_chunks: 4 },
176                AxisChunks::Uniform { size: 3, n_chunks: 2 },
177            ],
178            names: vec!["i".into(), "j".into()],
179            occupancy: None,
180        };
181        let mut indexers = HashMap::new();
182        indexers.insert("i".into(), vec![0, 3, 6, 12]);
183        let (out, runs) = output_grid_and_selection(&input_grid, &indexers).unwrap();
184        assert_eq!(out.shape, vec![4, 6]);
185        // "j" axis passes through.
186        assert_eq!(runs[1], None);
187        // "i" has runs.
188        let runs_i = runs[0].as_ref().unwrap();
189        assert_eq!(runs_i.runs.len(), 3); // chunks 0, 1, 2
190        assert_eq!(runs_i.len(), 4);
191    }
