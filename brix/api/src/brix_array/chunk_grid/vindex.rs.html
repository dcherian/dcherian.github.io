1//! Chunk-grid algebra for vectorized indexing (`ArrayOp::Vindex` →
2//! `BlockwiseOp::VIndex`).
3//!
4//! Plan-time math: given an input `ChunkGrid` and a set of per-axis 1-D
5//! integer index arrays *zipped* into a single produced axis `k`,
6//! partition `k` into runs by source chunk (multi-axis chunk-coord
7//! tuple) and build the output `ChunkGrid`.
8//!
9//! Contrast with [`super::oindex`]: oindex runs are independent per
10//! axis (the cartesian product reconstructs the output). Vindex zips
11//! the consumed axes — each `k` position picks one cell across all
12//! consumed axes simultaneously, so a "run" is a contiguous group of
13//! `k`s that map to the *same multi-axis input chunk*.
14//!
15//! The narrow slice (see `checkpoints/vindex.md`):
16//! - indices are literals, known at plan time,
17//! - pre-sorted so that the per-run input-chunk-coord tuple is
18//!   non-decreasing in lex order over the consumed axes,
19//! - hard error otherwise (no auto-sort here; see shuffle.md).
20
21use std::{collections::HashMap, iter::zip};
22
23use crate::chunk_grid::{AxisChunks, AxisSelection, ChunkGrid, ChunkGridError, ChunkGridResult};
24use brix::AxisName;

26// ---------------------------------------------------------------------------
27// VIndexRuns — partition of the produced axis
28// ---------------------------------------------------------------------------
29
30/// Partition of the produced `k` axis into multi-axis runs, stored
31/// per-axis-outer so that [`Self::iter`] is a literal `zip` against
32/// [`Self::consumed_axes`].
33///
34/// `per_axis[a][r]` is the selection along axis `consumed_axes[a]` for
35/// output run `r`. The "run width" — `per_axis[a][r].positions.len()` —
36/// is the same across all axes at a given `r` (by construction); k_start
37/// / k_end fall out of the cumulative sum of widths and are not stored.
38///
39/// `consumed_axes` is the canonical (sorted) name order and lives on the
40/// container so an empty `per_axis` still records which axes were consumed.
41#[derive(Clone, Debug, PartialEq, Eq)]
42pub struct VectorizedSelection {
43    pub consumed_axes: Vec<AxisName>,
44    /// `per_axis[a]` is the sequence of per-run selections for axis
45    /// `consumed_axes[a]`. All entries have the same length (= number of
46    /// runs); `per_axis[*][r].positions.len()` is constant across axes.
47    pub per_axis: Vec<Vec<AxisSelection>>,
48}
49
50impl VectorizedSelection {
51    /// `(axis_name, &per-run-selections-for-that-axis)` — well-typed zip.
52    pub fn iter(&self) -> impl Iterator<Item = (&AxisName, &Vec<AxisSelection>)> {
53        zip(self.consumed_axes.iter(), self.per_axis.iter())
54    }
55
56    /// Total number of indexed positions (= length of the produced `k` axis).
57    pub fn len(&self) -> usize {
58        self.per_axis.first().map_or(0, |axis| axis.iter().map(|s| s.positions.len()).sum())
59    }
60
61    pub fn is_empty(&self) -> bool {
62        self.len() == 0
63    }
64
65    /// One entry per run: the run's width (= the produced axis chunk size).
66    pub fn chunk_sizes(&self) -> impl Iterator<Item = usize> + '_ {
67        self.per_axis
68            .first()
69            .into_iter()
70            .flat_map(|axis| axis.iter().map(|sel| sel.positions.len()))
71    }
72
73    /// Build runs by zipping the per-consumed-axis index arrays and
74    /// grouping by (multi-axis) input chunk coord.
75    ///
76    /// Preconditions (hard errors otherwise):
77    /// - all index arrays have the same length (= produced k extent),
78    /// - every consumed axis appears in `indices` keyed by name,
79    /// - no consumed axis is `AxisChunks::Variable` (mirror oindex),
80    /// - every index is in bounds along its axis,
81    /// - the resulting sequence of input-chunk-coord tuples is
82    ///   non-decreasing in lex order over `consumed_axes`.
83    ///
84    /// `consumed_axes` fixes the lex order used for the sortedness
85    /// check and is preserved on the returned [`VectorizedSelection`].
86    pub fn from_indexer(
87        indexer: &HashMap<AxisName, Vec<usize>>,
88        input_grid: &ChunkGrid,
89    ) -> ChunkGridResult<Self> {
90        // Canonical (deterministic) order over consumed axes — drives
91        // `per_axis` row ordering and the lex-sortedness check.
92        let mut consumed_axes: Vec<AxisName> = indexer.keys().cloned().collect();
93        consumed_axes.sort();
94
95        // Resolve each consumed axis to its AxisChunks, validating axis
96        // existence and rejecting Variable (mirror oindex).
97        let axis_chunks: Vec<&AxisChunks> = consumed_axes
98            .iter()
99            .map(|axis| {
100                let (chunks, _) = input_grid.get(axis)?;
101                if matches!(chunks, AxisChunks::Variable(_)) {
102                    return Err(ChunkGridError::Unsupported(format!(
103                        "VIndex on AxisChunks::Variable axis {axis:?} not supported"
104                    )));
105                }
106                Ok(chunks)
107            })
108            .collect::<ChunkGridResult<_>>()?;
109
110        // All index arrays must agree on length (= produced k extent).
111        let k_extent = indexer[&consumed_axes[0]].len();
112        for axis in &consumed_axes {
113            let len = indexer[axis].len();
114            if len != k_extent {
115                return Err(ChunkGridError::Unsupported(format!(
116                    "VIndex: index array for axis {axis:?} has length {len} but \
117                     expected {k_extent} (length of axis {:?}'s array)",
118                    consumed_axes[0]
119                )));
120            }
121        }
122
123        // Walk k-positions, group by multi-axis chunk-coord (canonical
124        // order), hard-error on any lex decrease across the sequence.
125        // Each axis grows its own per-run Vec in lockstep.
126        let mut per_axis: Vec<Vec<AxisSelection>> =
127            (0..consumed_axes.len()).map(|_| Vec::new()).collect();
128        let mut last_coord: Option<Vec<usize>> = None;
129
130        #[allow(clippy::needless_range_loop)]
131        for k in 0..k_extent {
132            let resolved: Vec<(usize, usize)> = consumed_axes
133                .iter()
134                .enumerate()
135                .map(|(a, axis)| axis_chunks[a].locate(indexer[axis][k]))
136                .collect::<ChunkGridResult<_>>()?;
137            let chunk_coord: Vec<usize> = resolved.iter().map(|&(c, _)| c).collect();
138
139            if let Some(prev) = &last_coord {
140                if chunk_coord < *prev {
141                    return Err(ChunkGridError::Unsupported(format!(
142                        "VIndex indices not sorted by input chunk at k={k}: \
143                         coord {chunk_coord:?} < previous {prev:?}"
144                    )));
145                }
146                if chunk_coord == *prev {
147                    // Extend the current (last) run on every axis.
148                    for (axis_sels, &(_, pos)) in per_axis.iter_mut().zip(resolved.iter()) {
149                        axis_sels.last_mut().unwrap().positions.push(pos);
150                    }
151                    continue;
152                }
153            }
154            // Start a new run: push a fresh AxisSelection on every axis.
155            for (axis_sels, &(c, p)) in per_axis.iter_mut().zip(resolved.iter()) {
156                axis_sels.push(AxisSelection { source_chunk: c, positions: vec![p] });
157            }
158            last_coord = Some(chunk_coord);
159        }
160
161        Ok(VectorizedSelection { consumed_axes, per_axis })
162    }
163}

165// ---------------------------------------------------------------------------
166// output_grid_and_runs — driver
167// ---------------------------------------------------------------------------
168
169/// Compute the output `ChunkGrid` and the multi-axis run partition for
170/// a vindex node.
171///
172/// `produced_axis` is the single output axis name (Phase 1 restricts
173/// `VectorizedIndexer.produces` to one axis at lowering — see
174/// `checkpoints/vindex.md`). `consumed_axes` is the canonical
175/// input-axis order used both to key into `indices` and to interpret
176/// each run's multi-axis chunk coord.
177///
178/// The returned grid has one axis (`produced_axis`) whose chunks are
179/// non-uniform: one chunk per run, sized by the run's k-extent.
180pub fn output_grid_and_selection(
181    produced_axis: AxisName,
182    indexer: &HashMap<AxisName, Vec<usize>>,
183    input_grid: &ChunkGrid,
184) -> ChunkGridResult<(ChunkGrid, VectorizedSelection)> {
185    let runs = VectorizedSelection::from_indexer(indexer, input_grid)?;
186    let new_axis = AxisChunks::Variable(runs.chunk_sizes().collect());
187    let output_grid =
188        input_grid.without_axes(indexer.keys())?.extend([(produced_axis, new_axis)])?;
189    Ok((output_grid, runs))
190}
