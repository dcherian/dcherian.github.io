1//! Lower the array-level IR (`ArrayExpr`) to the mid-IR (`Expr`).
2//!
3//! Walks `ArrayExpr` post-order and emits the corresponding `Expr`.
4//!
5//! To do so, we need to also make concrete `ChunkGrid` choices for execution.
6//! That information is embedded in the [`brix::Source`], [`brix::Kernel`],
7//! [`brix::Accumulator`] passed on to `brix`.
8//!
9//! Note that the lowering here is a mechanical translation from [`ArrayExpr`] to [`Expr`].
10//! We simply add concrete `ChunkGrid` information to the outputs of each [`Expr`]
11//! The planner in brix/planner.rs is responsible for wiring this up
12//! and accounting for broadcasting semantics etc.
13//!
14//! We have `expect` in this file, because we expect those invariants
15//! to be enforced higher up the call chain.
16

48/// A lowered array expression: the mid-IR DAG.
49///
50/// `grids` caches the per-node element-sized `ChunkGrid` for every node
51/// in the lowered DAG, keyed by [`ExprId`]. Internal consumers
52/// (render in particular) can look this up to recover the rich
53/// `ChunkGrid` (with `AxisChunks` encoding and occupancy) without
54/// re-running the lowering math. Downstream of lowering, brix only sees
55/// the counts-only [`brix::IterationSpace`] on each `Expr` — the rich grid
56/// lives only here and on the concrete `Source`/`Kernel`/`Accumulator`/`Sink`
57/// impls baked into each `Op`.
58// ANCHOR: lowered
59#[derive(Clone)]
60pub struct Lowered {
61    pub expr: Arc<Expr>,
62    // This sidecar Map is simply to enable nice visualization/explainers.
63    // It is *not* used for planning or execution at the brix layer.
64    pub grids: HashMap<ExprId, ChunkGrid>,
65}
66// ANCHOR_END: lowered

70/// Lower an [`ArrayLike`] to a mid-IR `Arc<Expr>` DAG.
71/// The per-axis chunk target is derived from root shape + dtype + memory budget;
72/// child targets inherit it.
73/// FIXME: `memory_budget` needs to become a broader "ResourceConstraints" or similar.
74pub fn lower(arr: &impl ArrayLike, memory_budget: usize) -> ArrayResult<Lowered> {
75    // ANCHOR_END: lower_sig
76    let expr = arr.expr();
77    ////////////////////////////////////////////////////////////
78    // FIXME: this is extremely preliminary.
79    // Derive the chunk target from the *widest* shape in the DAG, not
80    // the root: a reduce drops axes from the root, but the inputs
81    // still carry those axes and we want them chunked under the same
82    // budget. Per-position max across all nodes gives chunk_target the
83    // length of the deepest axis stack we'll see.
84    let chunk_target = choose_chunk_target(expr, memory_budget);
85    ////////////////////////////////////////////////////////////
86    let mut memo: HashMap<*const ArrayExpr, (Arc<Expr>, ChunkGrid)> = HashMap::new();
87    let mut grids: HashMap<ExprId, ChunkGrid> = HashMap::new();
88    let (root_expr, _) = lower_inner(expr, &chunk_target, &mut memo, &mut grids)?;
89    Ok(Lowered { expr: root_expr, grids })
90}
91// ANCHOR_END: lower_code

93/// Per-position max across the entire ArrayExpr DAG. Used by [`lower`]
94/// to derive a chunk_target that covers every axis the lowering will
95/// touch — including axes dropped by reductions in the root. Axis order
96/// is positional: callers assume that the axis at position `d` in any
97/// node corresponds to the same logical axis (true for current
98/// workloads, which don't transpose mid-DAG).
99pub(crate) fn max_shape_in_tree(expr: &Arc<ArrayExpr>) -> Vec<usize> {
100    use std::collections::HashSet;
101    let mut max_shape: Vec<usize> = Vec::new();
102    let mut seen: HashSet<*const ArrayExpr> = HashSet::new();
103    fn walk(
104        expr: &Arc<ArrayExpr>,
105        max_shape: &mut Vec<usize>,
106        seen: &mut HashSet<*const ArrayExpr>,
107    ) {
108        if !seen.insert(Arc::as_ptr(expr)) {
109            return;
110        }
111        for (i, &dim) in expr.shape().iter().enumerate() {
112            if i < max_shape.len() {
113                max_shape[i] = max_shape[i].max(dim);
114            } else {
115                max_shape.push(dim);
116            }
117        }
118        for input in &expr.inputs {
119            walk(input, max_shape, seen);
120        }
121    }
122    walk(expr, &mut max_shape, &mut seen);
123    max_shape
124}

126/// Derive a per-axis chunk target from the widest shape in the DAG, the
127/// root's dtype, and the memory budget. The widest-shape derivation
128/// covers axes that the root drops (e.g. reductions) but inputs still
129/// carry — every node downstream chunks under the same budget.
130pub fn choose_chunk_target(expr: &Arc<ArrayExpr>, memory_budget: usize) -> Vec<usize> {
131    let shape = max_shape_in_tree(expr);
132    let bytes_per_element = expr.dtype.itemsize();
133    let ndim = shape.len();
134    let max_elems = memory_budget / bytes_per_element;
135    let extent = (max_elems as f64).powf(1.0 / ndim as f64).floor().max(1.0) as usize;
136    let target = shape.iter().map(|&s| extent.min(s)).collect::<Vec<_>>();
137
138    debug!(budget = memory_budget, ?shape, ?target, "choose_chunk_target");
139    target
140}

142/// Translate a framed reduce's element-space [`Frame`]s onto the chunk grids
143/// to build the brick-space per-axis access map.
144fn build_reduce_access(
145    view: &HashMap<AxisName, Frame>,
146    input_grid: &ChunkGrid,
147    output_grid: &ChunkGrid,
148) -> ChunkGridResult<HashMap<AxisName, AxisAccess>> {
149    view.iter()
150        .map(|(axis, frame)| -> ChunkGridResult<Option<(AxisName, AxisAccess)>> {
151            Ok(match frame {
152                Frame::WholeAxis { keepdims } => Some((
153                    axis.clone(),
154                    AxisAccess::Composed(BrixFrame::WholeAxis { keepdims: *keepdims }),
155                )),
156                // Coarsen folds `w = ceil(n_in / n_out)` input bricks per output brick.
157                Frame::Window { size, stride } if size == stride => {
158                    let n_in = input_grid.get(axis)?.0.n_chunks();
159                    let n_out = output_grid.get(axis)?.0.n_chunks();
160                    match n_in.div_ceil(n_out) {
161                        1 => None, // identity: no framing on this axis
162                        // Window's semantics allow a possibly smaller last chunk.
163                        w => Some((
164                            axis.clone(),
165                            AxisAccess::Composed(BrixFrame::Window { size: w, stride: w }),
166                        )),
167                    }
168                }
169                other => {
170                    return Err(ChunkGridError::Unsupported(format!(
171                        "lowering Frame {other:?} to brix Frame is not implemented",
172                    )));
173                }
174            })
175        })
176        .collect::<ChunkGridResult<Vec<_>>>()
177        .map(|entries| entries.into_iter().flatten().collect())
178}

180// Note: This was vibe-coded but generally appears OK.
181// 1. The choice to memo-ize may be premature.
182fn lower_inner(
183    expr: &Arc<ArrayExpr>,
184    chunk_target: &[usize],
185    memo: &mut HashMap<*const ArrayExpr, (Arc<Expr>, ChunkGrid)>,
186    grids: &mut HashMap<ExprId, ChunkGrid>,
187) -> LoweringResult<(Arc<Expr>, ChunkGrid)> {
188    let key = Arc::as_ptr(expr);
189    if let Some(hit) = memo.get(&key) {
190        return Ok(hit.clone());
191    }
192
193    let lowered_inputs: Vec<(Arc<Expr>, ChunkGrid)> = expr
194        .inputs
195        .iter()
196        .map(|inp| lower_inner(inp, chunk_target, memo, grids))
197        .collect::<LoweringResult<_>>()?;
198
199    let (node, grid): (Expr, ChunkGrid) = match &expr.op {
200        ArrayOp::TransparentSource(kind) => {
201            let names: Vec<AxisName> = expr.dims().to_vec();
202            let grid = ChunkGrid::new(kind.shape(), &names, chunk_target);
203            let op = Op::Source(Arc::new(LoweredTransparent {
204                inner: kind.clone(),
205                grid: grid.clone(),
206            }));
207            (
208                Expr {
209                    id: ExprId::fresh(),
210                    op,
211                    inputs: vec![],
212                    space: grid.to_iteration_space(),
213                    label: expr.label.clone(),
214                },
215                grid,
216            )
217        }
218        ArrayOp::Literal(value) => {
219            // Rank-0: an empty grid is one brick; the constant source
220            // materializes the value on demand.
221            let grid = ChunkGrid::new(&[], &[], &[]);
222            let op = Op::Source(Arc::new(LoweredLiteral { value: value.clone() }));
223            (
224                Expr {
225                    id: ExprId::fresh(),
226                    op,
227                    inputs: vec![],
228                    space: grid.to_iteration_space(),
229                    label: expr.label.clone(),
230                },
231                grid,
232            )
233        }
234        ArrayOp::OpaqueSource(src) => {
235            // V0: chosen grid == source's native grid (the trivially
236            // native-aligned choice).
237            // FIXME: real choice goes through `chunk_target` and picks the largest axis-aligned multiple
238            // of native that fits;
239            let grid: ChunkGrid = src.native_grid().0.clone();
240            let op = Op::Source(Arc::new(LoweredOpaque { inner: src.clone(), grid: grid.clone() }));
241            (
242                Expr {
243                    id: ExprId::fresh(),
244                    op,
245                    inputs: vec![],
246                    space: grid.to_iteration_space(),
247                    label: expr.label.clone(),
248                },
249                grid,
250            )
251        }
252        ArrayOp::AsType(dst) => {
253            let input_grid = lowered_inputs[0].1.clone();
254            let (src, dst) = match (&expr.inputs[0].dtype, dst) {
255                (DType::Primitive(s), DType::Primitive(d)) => (*s, *d),
256                // Extension casts are rejected at `ArrayExpr::cast` construction
257                // (a TODO: they'll lower through the dtype's own capability over
258                // its storage primitive, not this monomorphized `AsTypeKernel`).
259                _ => unreachable!("extension-dtype cast rejected at ArrayExpr::cast"),
260            };
261            let kernel = Arc::new(AsTypeKernel { src, dst, grid: input_grid.clone() });
262            (
263                Expr {
264                    id: ExprId::fresh(),
265                    op: Op::Blockwise(kernel),
266                    inputs: vec![Input {
267                        expr: lowered_inputs[0].0.clone(),
268                        access: HashMap::new(),
269                    }],
270                    space: input_grid.to_iteration_space(),
271                    label: expr.label.clone(),
272                },
273                input_grid,
274            )
275        }
276        ArrayOp::Unary(op) => {
277            let input_grid = lowered_inputs[0].1.clone();
278            let kernel = Arc::new(UnaryKernel {
279                op: *op,
280                dtype: expr.dtype.clone(),
281                grid: input_grid.clone(),
282            });
283            (
284                Expr {
285                    id: ExprId::fresh(),
286                    op: Op::Blockwise(kernel),
287                    inputs: vec![Input {
288                        expr: lowered_inputs[0].0.clone(),
289                        access: HashMap::new(),
290                    }],
291                    space: input_grid.to_iteration_space(),
292                    label: expr.label.clone(),
293                },
294                input_grid,
295            )
296        }
297        ArrayOp::Binary(op) => {
298            let input_grids: Vec<&ChunkGrid> = lowered_inputs.iter().map(|l| &l.1).collect();
299            // construct the output chunk grid by broadcasting
300            // FIXME: This should error for non-aligned inputs
301            let grid = broadcast_grids(&input_grids, expr.dims());
302            let kernel =
303                Arc::new(BinaryKernel { op: *op, dtype: expr.dtype.clone(), grid: grid.clone() });
304            // All inputs have ChunkAccess::identity since we will be accessing all input chunks
305            // Note that the actual broadcasting semantics are generated by brix/planner.rs
306            // which looks at output grid and input grids and wires up the broadcasting needed
307            // to get the right output.
308            let inputs = lowered_inputs
309                .iter()
310                .map(|(expr, _)| Input { expr: expr.clone(), access: HashMap::new() })
311                .collect();
312            (
313                Expr {
314                    id: ExprId::fresh(),
315                    op: Op::Blockwise(kernel),
316                    inputs,
317                    space: grid.to_iteration_space(),
318                    label: expr.label.clone(),
319                },
320                grid,
321            )
322        }
323        ArrayOp::Reduce { acc, view } => {
324            let (input_expr, input_grid) = &lowered_inputs[0];
325            // Reducing along a sparse axis would need the `SparseAccumulator` wrapper.
326            if let Some(occ) = &input_grid.occupancy
327                && let Some(ax) = view.keys().find(|a| occ.sparse_axes.contains(*a))
328            {
329                return Err(ChunkGridError::Unsupported(format!(
330                    "reduce along sparse axis {ax:?} is V1; \
331                     V0 supports reductions along dense axes only",
332                ))
333                .into());
334            }
335            let output_grid = input_grid.after_reduce(view)?;
336            let access = build_reduce_access(view, input_grid, &output_grid)?;
337            // this is a bit of a hack to maintain internal State with `keepdims=True` consistently regardless of the user's preference.
338            // the reduction axis is not squeezed out on the intermediates; we squeeze it out in `finalize` if `keepdims` is `true`.
339            let kept_views: HashMap<_, _> =
340                view.iter().map(|(k, v)| (k.clone(), v.force_keepdims())).collect();
341            let accumulator = Arc::new(LoweredAccumulator {
342                inner: acc.clone(),
343                grid: input_grid.after_reduce(&kept_views)?,
344                view: view.clone(),
345            });
346            (
347                Expr {
348                    id: ExprId::fresh(),
349                    op: Op::Reduce(accumulator),
350                    inputs: vec![Input { expr: input_expr.clone(), access }],
351                    space: output_grid.to_iteration_space(),
352                    label: expr.label.clone(),
353                },
354                output_grid,
355            )
356        }
357        ArrayOp::Slice(slicers) => {
358            let input_grid = lowered_inputs[0].1.clone();
359            let (output_grid, axis_offsets) =
360                chunk_grid::select::output_grid_and_offsets(&input_grid, slicers)?;
361
362            let kernel = Arc::new(SliceKernel {
363                slicers: slicers.clone(),
364                dtype: expr.dtype.clone(),
365                grid: output_grid.clone(),
366                input_grid: input_grid.clone(),
367            });
368
369            // Per-edge access — pick up the per-axis chunk offsets the
370            // composition computed. Identity axes (offset 0) contribute
371            // nothing.
372            let access: HashMap<AxisName, AxisAccess> = input_grid
373                .names
374                .iter()
375                .zip(axis_offsets.iter())
376                .map(|(name, &offset)| {
377                    (
378                        name.clone(),
379                        AxisAccess::Translated(TranslatedAccess::Independent(
380                            AxisOverride::Offset(offset),
381                        )),
382                    )
383                })
384                .collect();
385
386            (
387                Expr {
388                    id: ExprId::fresh(),
389                    op: Op::Blockwise(kernel),
390                    inputs: vec![Input { expr: lowered_inputs[0].0.clone(), access }],
391                    space: output_grid.to_iteration_space(),
392                    label: expr.label.clone(),
393                },
394                output_grid,
395            )
396        }
397        ArrayOp::Oindex(indexers) => {
398            let input_grid = lowered_inputs[0].1.clone();
399            let (output_grid, selection) =
400                chunk_grid::oindex::output_grid_and_selection(&input_grid, indexers)?;
401
402            let kernel = Arc::new(GatherKernel {
403                indices: indexers.clone(),
404                dtype: expr.dtype.clone(),
405                grid: output_grid.clone(),
406                input_grid: input_grid.clone(),
407            });
408
409            let access: HashMap<AxisName, AxisAccess> = input_grid
410                .names
411                .iter()
412                .zip(selection.iter())
413                .filter_map(|(name, sel)| {
414                    sel.as_ref().map(|r| {
415                        let table: Vec<usize> = r.runs.iter().map(|run| run.source_chunk).collect();
416                        (
417                            name.clone(),
418                            AxisAccess::Translated(TranslatedAccess::Independent(
419                                AxisOverride::Lookup(table),
420                            )),
421                        )
422                    })
423                })
424                .collect();
425
426            (
427                Expr {
428                    id: ExprId::fresh(),
429                    op: Op::Blockwise(kernel),
430                    inputs: vec![Input { expr: lowered_inputs[0].0.clone(), access }],
431                    space: output_grid.to_iteration_space(),
432                    label: expr.label.clone(),
433                },
434                output_grid,
435            )
436        }
437        ArrayOp::Vindex(indexer) => {
438            let input_grid = &lowered_inputs[0].1;
439            let (output_grid, selection) = chunk_grid::vindex::output_grid_and_selection(
440                indexer.produced_axis.clone(),
441                &indexer.indices,
442                input_grid,
443            )?;
444            let followers: HashMap<AxisName, AxisOverride> = selection
445                .iter()
446                .map(|(axis, axis_sel)| {
447                    let table: Vec<usize> = axis_sel.iter().map(|x| x.source_chunk).collect();
448                    (axis.clone(), AxisOverride::Lookup(table))
449                })
450                .collect();
451            let access = HashMap::from([(
452                indexer.produced_axis.clone(),
453                AxisAccess::Translated(TranslatedAccess::Zipped(followers)),
454            )]);
455            let kernel = Arc::new(VIndexKernel {
456                produces: indexer.produced_axis.clone(),
457                selection,
458                dtype: expr.dtype.clone(),
459                grid: output_grid.clone(),
460                input_grid: input_grid.clone(),
461            });
462            (
463                Expr {
464                    id: ExprId::fresh(),
465                    op: Op::Blockwise(kernel),
466                    inputs: vec![Input { expr: lowered_inputs[0].0.clone(), access }],
467                    space: output_grid.to_iteration_space(),
468                    label: expr.label.clone(),
469                },
470                output_grid,
471            )
472        }
473        ArrayOp::Sink(sink) => {
474            let (input_expr, input_grid) = &lowered_inputs[0];
475            let op = Op::Sink(sink.lower(input_grid.clone(), expr.dtype.clone())?);
476            (
477                Expr {
478                    id: ExprId::fresh(),
479                    op,
480                    inputs: vec![Input { expr: input_expr.clone(), access: HashMap::new() }],
481                    space: input_grid.to_iteration_space(),
482                    label: expr.label.clone(),
483                },
484                input_grid.clone(),
485            )
486        }
487    };
488
489    let expr = Arc::new(node);
490    grids.insert(expr.id, grid.clone());
491    let result = (expr, grid);
492    memo.insert(key, result.clone());
493    Ok(result)
494}

502    fn one_slice(axis: &str, range: std::ops::Range<usize>) -> HashMap<AxisName, Indexer> {
503        let mut m = HashMap::new();
504        m.insert(axis.into(), Indexer::Slicer(Slice::contiguous(range.start, range.end)));

508    /// Source → Elementwise → Select → Reduce lowers in one pass; the
509    /// resulting mid-IR has matching dims at every node.
510    #[test]
511    fn end_to_end_lower() {
512        let a = Array::fill(1.0, vec![10, 4], &["i", "j"]);
513        let b = Array::fill(2.0, vec![10, 4], &["i", "j"]);
514        let sum = (a + b).isel(one_slice("i", 0..6)).unwrap();
515        let reduced = sum.over(&["j"]).unwrap().sum().unwrap();
516        let mid = lower(&reduced, 1024).unwrap().expr;
517        assert_eq!(mid.space.names(), &["i"]);
518        assert_eq!(reduced.shape(), vec![6]);
519    }

521    /// Pointer-identity memoization: a single `Arc<ArrayExpr>` shared by
522    /// two consumers lowers to a single `Arc<Expr>`.
523    #[test]
524    fn fanout_is_preserved() {
525        let leaf = Array::fill(1.0, vec![4], &["i"]);
526        let lowered = lower(&(leaf.clone() + leaf), 1024).unwrap().expr;
527        assert_eq!(lowered.inputs.len(), 2);
528        assert!(Arc::ptr_eq(&lowered.inputs[0].expr, &lowered.inputs[1].expr));
529    }
