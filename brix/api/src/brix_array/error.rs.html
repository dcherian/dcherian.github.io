1//! Array-layer error type.
2
3use thiserror::Error;
4
5use crate::dtype::DType;
6use crate::lower::LoweringError;
7use crate::ops::BinaryOp;
8use brix::{BrixError, LayoutError};
9
10#[derive(Debug, Error)]
11pub enum ArrayError {
12    #[error("shape mismatch: {0}")]
13    ShapeMismatch(String),
14
15    /// Both operands declined the op (the analogue of Python's `TypeError`
16    /// after `__op__` and `__rop__` both return `NotImplemented`).
17    #[error("unsupported operand dtypes for {op}: {left} and {right}")]
18    InvalidBinaryOpOperands { op: BinaryOp, left: DType, right: DType },
19
20    /// A satellite downcast failed: a value advertised an extension dtype but
21    /// its concrete payload/scalar type was not the `expected` one (the dtype
22    /// tag and the concrete type disagree). A malformed-operand error a
23    /// capability returns, distinct from a clean op decline (`None`).
24    #[error("extension downcast failed: expected concrete type `{expected}`")]
25    ExtensionDowncastError { expected: &'static str },
26
27    /// Default-deny for reductions a dtype doesn't structurally support
28    /// (e.g. `sum` over an extension dtype).
29    #[error("reduction {reduction:?} unsupported for dtype {dtype}")]
30    UnsupportedReduction { reduction: &'static str, dtype: DType },
31
32    /// A sink's fill value doesn't match the output dtype — caught at
33    /// [`crate::expr::ArrayExpr::sink`] construction.
34    #[error("sink fill dtype {fill} does not match sink dtype {sink}")]
35    SinkFillDTypeMismatch { fill: DType, sink: DType },
36
37    #[error("unknown axis {name:?}; expected one of {available:?}")]
38    UnknownAxis { name: String, available: Vec<String> },
39
40    #[error("Use of known axis {name:?} not allowed")]
41    KnownAxis { name: String },
42
43    #[error("{0}")]
44    Unsupported(String),
45
46    #[error(transparent)]
47    Layout(#[from] LayoutError),
48
49    #[error(transparent)]
50    Brix(#[from] BrixError),
51
52    #[error(transparent)]
53    Lowering(#[from] LoweringError),
54}
55
56pub type ArrayResult<T> = std::result::Result<T, ArrayError>;
